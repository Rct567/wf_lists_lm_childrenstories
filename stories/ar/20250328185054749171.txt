فَوْقُ: (بصوتٍ عالٍ ومرح) "صحيحٌ! أتذكَّرُ كيفَ كنَّا نلعبُ في العُشبِ الأخضرِ."

لِيلا: (تومئُ برأسِها بحكمةٍ) "لكنَّ الأرضَ أخذتْ تتجمّدُ، والأشجارُ اكتستْ بالبياضِ."

أُورْس: "هلْ تتذكرينَ عندما بدأتِ القِصَّةُ يا لِيلا؟"

لِيلا: (تنظرُ إلى بعيدٍ) "نعم، كانتْ هناكَ عاصفةٌ ثلجيّةٌ كبرى. كلُّ شيءٍ اختفى تحتَ غطاءٍ أبيضٍ ناصعٍ."

فَوْقُ: "وعندما هَدَأتِ العاصفةُ، كانَ كلُّ شيءٍ مختلفًا تمامًا."

سُون: "أذكرُ كيفَ خفنا جميعًا. لمْ نعرِفْ ما الذي سيحدُثُ."

أُورْس: "تذكرون عندما التقينا بِـ"سَايْمُون"، سنجابِ الثّلجِ العجوز؟"

سُون: "نعم! قالَ لنا أنَّ الثَّلجَ هديّةٌ ثمينةٌ."

لِيلا: "وأخبرَنا أيضًا أنَّ الثَّلجَ يحملُ سرًّا... سرَّ الحكاياتِ."

لِيلا: "قالَ أنَّ كلَّ ندفةِ ثلجٍ تحملُ حكايةً... حكايةً عنْ عالمٍ سحريّ."

فَوْقُ: (يبدو متحمِّسًا) "هلْ هذا صحيحٌ؟ هلْ للثَّلجِ حكاياتٌ حقيقيّة؟"

سُون: "بالتأكيد! وإلّا لماذا كانَ سَايْمُون يتحدَّثُ عن ذلكَ؟"

فَوْقُ: "ماذا حدثَ بعدَ ذلكَ؟"

فَوْقُ: "يا لها من بومةٍ شُجاعةٍ!"

أُورْس: "ذاتَ ليلةٍ، وبينما كانتْ تطيرُ، رأتْ أضواءً غريبةً في السماءِ."

لِيلا: (تتّسعُ عيناها) "أضواءَ شماليّةً!"

أُورْس: "بالضبط! أخذتْ آيسِلَا تحلِّقُ نحوِها، مفتونةً بجمالِها."

سُون: "ماذا رأتْ يا أُورْس؟"

أُورْس: "رأتْ الأضواءَ تتراقصُ بألوانٍ سحريّةٍ: الأخضرِ، والأزرقِ، والأرجوانيِّ."

أُورْس: "قرّرتْ آيسِلَا أنْ تنضمَّ إلى رقصةِ الأضواءِ. أصبحتْ تطيرُ معها وتدورُ في السماءِ."

لِيلا: "يا لهُ مِن حلمٍ جميلٍ!"

أُورْس: "تعلَّمتْ آيسِلَا أنَّ الجمالَ موجودٌ في كلِّ مكانٍ، حتى في أشدِّ الليالي ظلامًا."

سُون: "يا لها من حكايةٍ مدهشةٍ!"

فَوْقُ: "أحَبَبْتُها حقًّا!"

لِيلا: (تبتسم) "يبدو أنَّ كلَّ ندفةِ ثلجٍ تحملُ حكايةً تستحقُّ السَّماعَ."

سُون: "هذا صحيحٌ! أتمنى أنْ يخبّرَنا الثَّلجُ المزيدَ من الحكاياتِ."

أُورْس: "وأنا أيضًا! لِأَقُلْ هذه المرَّة: "يا ثَلجَنا الأبيَضَ، أخْبِرْنا بالمزيدِ من القصصِ!"
