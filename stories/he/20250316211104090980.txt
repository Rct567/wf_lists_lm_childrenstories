וְכָךְ הַמִּשְׁפָּחָה חָזְרָה לְבֵיתָהּ הָאֲמִתִּי, שֶׁלָּהּ מְגִדִּית גָּדוֹל, אֲבָל הֵם לֹא שָׁכְחוּ אֶת מִירְדָּק. הֵם הָיוּ יוֹדְעִים מְהַנּוֹת עִם הַגְּבִינָה שֶׁלּוֹ! וּלְכָל הַיֶּתֶר, הֵם כּוֹכְבִים אוֹתוֹ כְּבֶן מִשְׁפָּחָה.

כָּל שָׁבוּעַ הֵם בָּאוּ לְבַקֵּר אוֹתוֹ, לְהַשְׁאִיר לוֹ דְּלִי גְּדוֹל שֶׁל גְּבִינָה, וְלִשְׁמֹעַ אֶת הַסִּיפּוּרִים שֶׁלּוֹ עַל הַהַרְפַּתְקָאוֹת הַחֲדָשׁוֹת שֶׁלּוֹ, וּלְשַׂחֵק אִתּוֹ. מִירְדָּק לֹא הָיָה יָשֵׁן שׁוּב, לְאַחַר מְהִירוּת הוּא הַנְּסִיכָה! בִּקֵּר הַרְבֵּה מְקוֹמוֹת שׁוֹנִים, שֵׁנָּה תְּנִי בְּגֶטְּלָה בִּנְיָמִינוּ הִתְחַמְּשָׁה עִם זֶה בְּצֶדֶק בִּמְיֵה, אֲבָל תָּמִיד חָזַר לְאוֹתוֹ תָּל, וּבְנֹכְחוּת הַדֶּרֶךְ הָמַסְתִּי וְסִכְּיַת שֶׁלּוֹ הַיְּלָדִים וְכָךְ עַל חֲיִו שׁוֹקֶק וּמַצְלִיחַ.

כָּל מִי שֶׁמֵּעַז לָקַח אֶת הַחִירְת מִמִּשְׁפָּחָה, הוּא רָאוּי לָלֶכֶת לָקוּנָה! אָמְנָם זֶה סוֹפוֹ שֶׁל הַסִּיפּוּר, אֲבָל כְּשֶׁאַתֶּם רְצוֹת לִמְצֹא הֲכִי כִּיּוּפִי אוֹכֶל, מִירְדָּק מְחַכֶּה לָכֶם. גַּם בִּגְבִינָה טְעִימָה!
