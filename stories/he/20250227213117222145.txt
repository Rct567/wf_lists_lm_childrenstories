תַּלְמִי הִתְחִילָה לְהִשְׁתַּתֵּף בַּסִּפּוּר. הִיא עָזְרָה לַנָּסִיךְ לִמְצֹא אֶת הַתַּנִּין, הִשְׁתַּתְּפָה בַּמִּלְחָמָה תּוֹרָנִית וּמִלְּחָמָה וְקָרְבָנִי עִם הַתַּנִּין. הִיא מְאֹד נֶהֶנְתָּה מִן הַסִּפּוּר, בְּמֵחִיר דִּי לֹא יוּכַל לִשְׁכֹּחַ.

הִיא טִפְּלָה בְּסִירָה תּוֹךְ יָם שֶׁל אֶשֶׁת, הִתְיַדְּדָה עִם דָּבִּים מְדַבְּרִיא. הִיא בָּלְשָׁה אַחֲרֵי תַּעֲלוּמוּתיוֹת עִם חוֹפְשְׁטָן. עַל יְדֵי הַקְּרִיאָה, תַּלְמִי לָמְדָה עַל הָעֵינַיִם הַגּוֹרְעוֹת שֶׁל הָעוֹלָם וְשֶׁל הָאֲנָשִׁים סְבִיבָהּ.

אֶחָד מִן הַיָּמִים, תַּלְמִי קָרְאָה סִפּוּר עַל יַלְדָּה שֶׁלָּה הַמְּאַהֲרִיס. הַסִּפּוּר עַל הָעֲצָבִּי מֵאָז חוֹרוֹת עַל הַקִּירוֹת הַמְּצַמְצֵר, חוֹשְׁבִים עַל שָׁלוֹם. כְּשֶׁתַּלְמִי נִכְנְסָה לְסִפּוּר, הִיא הִשִּׂיגָה אֶת הַיַּלְדָּה, הִיא הַחֲבֵרָה הֲכִי טוֹבָה שֶׁאֶפְשָׁר לְדַמְיֵן. הֵן שָׁעוֹת שֶׁאָגְדוּ עַל הָעוֹלָם, עַד שֶׁסִּבּוּב הַסִּפּוּר גָּרַם לְדַרְדְּרְלַן בְּלִי לְהִסְתַּכֵּל.

בְּמֶשֶׁךְ הַזְּמַן, תַּלְמִי לָמְדָה שֶׁהִיא לֹא רַק חֲלוֹנִי מִן עַל סִפּוּרִים שֶׁל הַסֵּפֶר שֶׁל סָבְתָהּ. הִיא הִתְחִילָה לְהִשְׁתַּעְשֵׁעַ בְּסִפּוּרִים שֶׁל כָּל סְפָרִים.

בְּאַחַד הַיָּמִים, הִיא הִשְׁתַּמְּשָׁה בְּכִשּׁוּף הַסֵּפֶר שֶׁלָּהּ לְהִשְׁתַּתֵּף בְּסִפּוּרִים בַּסִּפְרִייה בְּעַצְמָהּ, הָפְכָה לְחֵלֶק מֵהֶם. הִיא חָשָׁה שֶׁאַף סִפּוּר אֵינֶנּוּ מְאֻכְּלָס יוֹתֵר יֵשׁ בִּתְנָאֵי "הַמַּתָּנָה הָעֲצוּמָה שֶׁל הָאִישְׁיוּת".

יוֹם אֶחָד, בְּעוֹד תַּלְמִי חוֹזֶרֶת מִבֵּית הַסֵּפֶר, עַל הַדֶּרֶךְ הִיא רָאֲתָה יֶלֶד קָטָן שֶׁלֹּא הֲלָקוּ אֶת הַכֶּלֶב שֶׁלָּהּ. הַיֶּלֶד הָיָה מַסְתִּיר אֶת עֵינָיו, גּוּפוֹ בִּידֵי כְּמוּשִׁי. תַּלְמִי יָדְעָה מִיָּד מַה עָלֶיהָ לַעֲשׁוֹת. הִיא קָרָאֵת לְצִפּוֹר, פָּתְחָה אֶת הַסֵּפֶר שֶׁלָּהּ, הָפַכְּתָּ לְסִפּוּר, וְהִיא הצליחה לְהַשְׁרִישׁ אֶת יְכוֹלְתָּהּ לְהָגִּיעַ לְעַצְמָהּ.

בְּאֹפֶן זֶה, הִיא לִמְּדָה אֶת הַיֶּלֶד הַקָּטָן עַל הַכֹּחַ שֶׁל הַדִּמְיוֹן וְשֶׁל הָאַחְרָיוּת. הִיא סִפְּרָה לָוֹ עַל כֵּיצַד הָיְנוּ נִכְלָלִים בְּהַקְשָׁבָה לְסִבָּתָנוּ בְּיַחַד וְלִפְעֹל בְּתוֹךְ יַחַד.

תַּלְמִי הִבִּיטָה סְבִיבָהּ. הַיֶּלֶד בָּרַח, עַל פָּנֶיהָ חוֹלְמָה גְּדוֹלָה. כְּשֶׁהִיא הִצְטָרְפָה לְסִפּוּר שֶׁל אַחֵר, הִיא זָכְרָה, הַסֵּפֶר מְלֵא בַּסִּפּוּרִים. עַתָּה הִיא יָדְעָה, שֶׁהִיא לֹא רַק קוֹרְאִים שׁוּב. הִיא הָיְתָה חֵלֶק מֵהֶם, מְשַׁנָּה אֶת הָעוֹלָם, סִפּוּר אֶחָד בְּכָל פַּעַם.

וּבְכֹךְ נִגְמַר הַסִּפּוּר שֶׁל תַּלְמִי, אֲשֶׁר גִּלְּתָה אֶת כּוֹחוֹ שֶׁל הַסֵּפֶר הַמְּדַבֵּר, הַכּוֹחַ שֶׁל קְסֶם, שֶׁל יְצִירָתִיּוּת וְשֶׁל אֱמוּן עַצְמִי. וְתָמִיד, כַּאֲשֶׁר הִיא חָשָׁה אֶחָד, הָיָה הַסֵּפֶר שֶׁלָּהּ שְׁמֵרָתָהּ מֻכָּנָה לְהַפְעִיל אֶת הַקְּסָם.
