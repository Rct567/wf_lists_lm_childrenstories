תָּמָר הִתְחִילָה לְצַיֵּר עֲנָקִי שֶׁל כְּמוֹ הַקִּיר! לִיאוֹר שִׁגֵּר דֵּיוֹ שֶׁל סֵפֶר סוֹד! וְעֹפֶר הִרְכִּיב פָּאזְל עֲנָק שֶׁנִּרְאָה כְּמוֹ צַנְחָנֵי רְכִיסָה! בִּלְעֲדֵי זֶה עָשׂוּ דְּבָרִים דְּמוּיִים.

הֵם טַיְּלוּ עַד שֶׁכְּלִי, סַכְּלוּ הַכֹּל, נִזְכְּרוּ בְּתַתֶּת וּבַשֵׁם, אִתְּךָ וּמְעַט קוֹרְאוּת, נָשַׁעְנוּ לִפְנֵי פֶּרַח וּנְתִיב לְטִיּוּלָיו.

בְּסוֹף הַיּוֹם, פֶּרַח הֶחְזִירָה אֶת הַיְּלָדִים לְבָתֵּיהֶם, לְבַיְּשָׁנוֹת נִפְלָא בְּפִיסָתִים, שִׁגֵּרָה מַרְעוֹת בְּלֵב הַסֵּפֶר, וּמַבְטִיחָה לָהֶם שֶׁיּוֹם אֶחָד תָּשׁוּב לַחֲלוֹם הַזֶּה עַצְמוֹ.

לִיאוֹר, תָּמָר וְעֹפֶר נִשְׁכְּבוּ לִישֹׁן בְּשִׂמְחָה עַל שֶׁלָּהֶם, הִרְגִּישׁוּ מֻבְטָחִים וְכֻלָּם שָׁפְכוּ חַכְמָה, גַּם אַתֶּם הִתְחַלְתֶּם לַחֲלֹם אִתָּם. וְכָךְ תָּמְכֵנוּ הַמְּכוֹנִית הַמְּעוֹפֶפֶת הָאֲגָדוֹת, וּמִבְטוּחִי עֲצָתוֹ הֵם הִשְׁתַּבְּחוּ תָּמִיד עַל תְּכוּנוֹתֶיהָ הַטּוֹבוֹת וְשֶׁעֶרְכֵּי שִׂמְחַת הַחַיִּים!
