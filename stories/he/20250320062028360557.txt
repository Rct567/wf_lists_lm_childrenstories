בְּעִיר קְטַנָּה וּשְׁקִיטָה, שֶׁיֵּשׁ בָּהּ בָּתִּים נְמוּכִים וְגַנִּים מְלֵאִים פְּרָחִים, גָּר יֶלֶד שֶׁקָּרְאוּ לוֹ אָרִי. אָרִי אָהַב לְהִתְבּוֹנֵן בַּשָּׁמַיִם, בְּמִיוּחַד בַּלֵּילוֹת כְּשֶׁהַכּוֹכָבִים הִבְרִיקוּ. כָּל לַיְלָה הוּא יָצָא לַגִּנָּה שֶׁלּוֹ וְהִתְבּוֹנֵן עַד שֶׁהוּא הָיָה נֵרְדָּם.

לְיָדוֹ הָיָה כּוּס סִיפּוּרִים יָשָׁן וְשָׁחִיק. הוּא קִבֵּל אוֹתוֹ מֵסַבָּא שֶׁלּוֹ. בַּכּוּס הָיָה כְּתוּב סִיפּוּר עַל יֶלֶד שֶׁיָּכֹל לְהָרִים כּוֹכָבִים מֵהַשָּׁמַיִם וְלַשֵּׂאת אוֹתָם בְּכִיסָיו. אָרִי חָשַׁב שֶׁזֶּה הָיָה סִיפּוּר נִפְלָא וְתָמִיד חָלַם לְהִתְנַסּוּת.

יוֹם אֶחָד, כְּשֶׁהַשֶּׁמֶשׁ הִתְחִילָה לִשְׁקֹעַ וְהַשָּׁמַיִם הָפְכוּ לִכְתֹּם, אָרִי הִרְגִּישׁ שֶׁמַּשֶּׁהוּ מוּזָר קוֹרֶה. לְעֵינָיו הוּא רָאָה כּוֹכָב נוֹפֵל מֵהַשָּׁמַיִם, זֹהֵר כְּמוֹ גַּחֶלֶת. הוּא הִתְעוֹרֵר בְּאַזְעָקָה, וְרָץ לַשָּׂדֶה.

לְתַדְהִימוֹנוֹ, הַכּוֹכָב נָפַל עַל הָאֲדָמָה, לֹא רָחוֹק מֵהַבַּיִת שֶׁלּוֹ. הוּא הָיָה בְּצוּרַת מַחְבֵּרֶת קְטַנָּה שֶׁל זָהָב, וְזֶהֵר בְּאוֹר רַךְ. אָרִי הִתְכּוֹפֵף, הִנְטָה אֶת רֹאשׁוֹ, וְנָגַע בַּכּוֹכָב. כְּשֶׁהוּא נָגַע, הוּא שָׁמַע שׁוֹפֵּר קָטָן שׁוֹמֵעַ, "שֵׁםִי הַכּוֹכָב הַזֶּה לֵאָה. אֲנִי שָׁלוּחָה מֵעוֹלָם הַכּוֹכָבִים לַעֲזֹר לְךָ."

אָרִי נִשְׁאָר קָפוּא בַּמָּקוֹם, שֶׁמַּשֶּׁהוּ לֹא הִאֲמִין בָּעֵינַיִם שֶׁלּוֹ. "לְהַעְזֹר לִי?" הוּא שָׁאַל.

"כֵּן," אָמְרָה לֵאָה בְּקוֹל קָטָן וְלֹא מֻרְגָּשׁ, "אֲנִי יוֹדַעַת עַל הַסִּיפּוּר שֶׁבַּכּוּס הַיָּשָׁן. אַתָּה יְכוֹל לֶאֱסֹף כּוֹכָבִים, אַךְ רַק עִם עֶזְרָתִי. אֲנִי אֲלַמֵּד אוֹתְךָ אֵיךְ."

אָרִי שָׂמַח עַד מְאֹד. הוּא תָּמִיד חָלַם עַל הַמִּפְעָל הַזֶּה. "אֵיךְ מִתְחִילִים?" הוּא שָׁאַל בְּהִתְלַהֲבוּת.

לֵאָה הֵשִׁיבָה, "קֹדֶם כָּל, עָלֶיךָ לְהַתְמִיד לִלְמֹד עַל הַכּוֹכָבִים וְעַל הַיְּקוּם. כַּמָּה שֶׁתֵּדַע יוֹתֵר, כָּךְ תִּהְיֶה לְךָ כֹּחַ יוֹתֵר."

וְכָךְ, הָחֵלָּה הַהַרְפַּתְקָה שֶׁל אָרִי. בְּכָל לַיְלָה, הוּא קָרָא סִיפּוּרִים עַל הַכּוֹכָבִים, הִתְבּוֹנֵן עֲלֵיהֶם בְּמַשָּׁה רַבָּה, וְלָמַד אֶת שֵׁמוֹתָם וְאֶת מַהוּתָם. לֵאָה, הַכּוֹכָב הַמְּמֻמֶּה שֶׁלּוֹ, הַדְרִיכָה אוֹתוֹ בְּכָל שְׁאֵלָה וְתָמִיךְ בּוֹ בְּכָל הַזְּמַן.

יוֹם אֶחָד, לֵאָה אָמְרָה לוֹ: "עַכְשָׁו, אַתָּה מוּכָן לְהַעֲבִיר כּוֹכָבִים אֵלֶיךָ. צָרִיךְ לִשְׁמֹעַ וּלְהִתְרַכֵּז בְּעֵצֵךְ. זֶה לֹא יִקְרֶה בִּמְהִירוּת, כִּי אַתָּה יֶלֶד קָטָן, אַךְ אֲנִי יוֹדַעַת שֶׁתַּצְלִיחַ!"

אָרִי עָצַם אֶת עֵינָיו וְהִתְמַקֵּד בַּשָּׁמַיִם. הוּא חָשַׁב עַל כָּל הַכּוֹכָבִים, עַל לֵאָה שֶׁעוֹזֶרֶת לוֹ, וְעַל הַכּוֹס שֶׁל הַסַּבָּא שֶׁלּוֹ. לְפֶתַע, הוּא הִרְגִּישׁ תְּחוּשָׁה מוּזָרָה בַּיָּד שֶׁלּוֹ. הוּא פָּתַח אֶת עֵינָיו וְרָאָה כּוֹכָב קָטָן וְזָהוּב בְּיָדוֹ. הוּא הִצְלִיחַ!

בִּמְשָׁךְ הַזְּמַן, אָרִי הָפַךְ לְמֵכִּיר יוֹתֵר בְּכִישׁוּרוֹ. הוּא הִתְחִיל לֶאֱסֹף כּוֹכָבִים יוֹתֵר וְיוֹתֵר, בְּצִדּוֹ שֶׁל לֵאָה. בְּכָל פַּעַם הוּא הָיָה צָרִיךְ לִלְמֹד שֵׁם שֶׁל כּוֹכָב. אָרִי שָׁמַע עַל כּוֹכָבִים שׁוֹנִים, גְּדוֹלִים וּקְטַנִּים, זוֹרְחִים וְעָמְלוּסִים, וּלְמַד תּוֹכְנוֹת עַל כְּמוּת מַסַּת הָאֵזוֹר.

כְּשֶׁאָרִי אָסַף יוֹתֵר וְיוֹתֵר כּוֹכָבִים, הוּא הִתְחִיל לְהַבְחִין שֶׁהַכּוֹכָבִים הָאֵלֶּה שׁוֹמְרִים עַל סוֹדוֹת. הָיָה לָהֶם כֹּחַ. כָּל כּוֹכָב יָכֹל לְהַגְשִׁים מַשְׁאָלָה אַחַת. אָרִי שָׁאַל אֶת לֵאָה לְשֵׁם הַדָּבָר, וְהִיא הֵשִׁיבָה, "כִּי הֵם תַּמִידוּשְׁמָעִים. כָּל כּוֹכָב יוֹדֵעַ מַה אַתָּה חָשׁ."

לְאַחַר מִכֵּן, לֵאָה עֲצַר הָרֶפֶשׁ. "אֲנִי צְרִיכָה לָלֶכֶת. הִשְׁלַמְתִּי אֶת מְשִׁימָתִי," אָמְרָה לְאָרִי בְּקוֹל רַךְ וְעָצִיב.

אָרִי הִצְטַעֵר. הוּא לֹא רָצָה שֶׁלֵּאָה תֵּלֵךְ. הֵם הָפְכוּ לִחֲבֵרִים טוֹבִים, וְהָיָה לוֹ קָשֶׁה לְוַתֵּר עַל הָרִגְשָׁה שֶׁהֵבִיאָה אִתָּהּ. "אֲבָל לָמָּה?" שָׁאַל.

"כִּי הַכּוֹכָבִים שֶׁלָּךְ עַכְשָׁו שׁוֹמְרִים עָלֶיךָ. הַכֹּל שָׁם." הִיא חִיְּכָה וְחִיּוּךְ לֹא מֻבְחָן.

בְּרֶגַע הַזֶּה, לֵאָה נֶעְלְמָה, כְּמוֹ נִצְנֵץ אוֹר בַּחֲשֵׁכָה. אָרִי הִסְתַּכֵּל עַל הַכּוֹכָבִים שֶׁבְּיָדָיו, וְהֵבִין שֶׁלֵּאָה צָרִיכָה לַחְזֹר לַעוֹלָם שֶׁלָּהּ. הוּא יָדַע שֶׁהֵם בְּטָחוֹן עִמּוֹ, וְשֶׁהוּא יוּכַל לְהִשְׁתַּמֵּשׁ בְּכֹּחַ שֶׁלָּהֶם לִבְנֵה שָׁמַיִם נְכוֹנִים.

הוּא הִחְלִיט לְנַתֵּן כֹּחוֹ לַאֲנָשִׁים שֶׁיָּכְלוּ לְהִשְׁתַּמֵּשׁ בּוֹ לְטַבְעוֹנֵי מַעֲשִׂים טוֹבִים. הוּא עָזַב אֶת הַכּוֹכָבִים לְכָל אֶחָד שֶׁעָבַר דֶּרֶךְ הַגִּנָּה שֶׁלּוֹ, עַד שֶׁכָּל הַכּוֹכָבִים חָזְרוּ לִשְׁמַיִם.

מֵאוֹתוֹ יוֹם, אָרִי לֹא עָצַר מֵהַתְמָדָה וּמִלִּלְמוּד הַשָּׁמַיִם. כָּל לַיְלָה הוּא הָיָה יוֹצֵא לַגִּנָּה וּמִתְבּוֹנֵן בַּכּוֹכָבִים, יוֹדֵעַ שֶׁלֵּאָה שָׁם, שׁוֹמֶרֶת עָלָיו וְעַל כָּל הַנּוֹרִים, וְשֶׁכָּל אוֹר כּוֹכָב הוּא אוֹר קָטָן שֶׁל תִּקְוָה. וְכָךְ, הָפַךְ אָרִי לְאוֹר בְּעַצְמוֹ, וּבְסִיפּוּרוֹ עָזְרוּ כּוֹכָבִים לְאוּשְׁרִים שֶׁל יֶלֶד לָכֹחַ.

וְהַסִּיפּוּר הַזֶּה מוֹשֵׁךְ וּמַרְגִּישׁ לִבֵּנוּ עַד הַיּוֹם.
