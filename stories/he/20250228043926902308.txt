הַצִּפּוֹר עָפָה לְמֶרְחַקִּים. הִיא עָפָה מֵעַל לַיָּם הַכְּחֻלָּה, חָלְפָה עַל פְּנֵי אִיִּים יְרֻקִּים, וְסוֹף סוֹף הִגִּיעָה לְמָקוֹם קָסוּם. שָׁם גָּרָה צִפּוֹר חֲכָמָה וְזְקֵנָה, שֶׁיָּדְעָה דְּבָרִים רַבִּים עַל הָעוֹלָם.

הַצִּפּוֹר סִפְּרָה לַצִּפּוֹר הַזְּקֵנָה עַל שׁוֹשָׁן וְעַל חֲלוֹמוֹ. הַצִּפּוֹר הַזְּקֵנָה הֶקְשִׁיבָה בְּקִשּׁוּב. כְּשֶׁהַצִּפּוֹר סִיְּמָה לְסַפֵּר, הַצִּפּוֹר הַזְּקֵנָה אָמְרָה, "כָּל דָּבָר אֶפְשָׁרִי, הַכֹּל תָּלוּי בֶּאֱמוּן וּבָרָצוֹן הָעַז."

הִיא שָׁלְפָה מִכִּיסָהּ מַגִּיעָה קְטַנָּה מְאֹד. "תְּנִי אֶת זֶה לְשׁוֹשָׁן," אָמְרָה, "הַמַּגִּיעָה תַּעֲזֹר לוֹ לִמְצֹא אֶת הַדֶּרֶךְ לַגָּן הַנִּסְתָּר."

הַצִּפּוֹר חָזְרָה לַיָּם וְנָתְנָה אֶת הַמַּגִּיעָה לְשׁוֹשָׁן. שׁוֹשָׁן שָׂמַח מְאוֹד. הוּא אָמַר, "אֲנִי לֹא יוֹדֵעַ אֵיךְ לְהוֹדוֹת לָךְ!" הַצִּפּוֹר חִיְּכָה וְאָמְרָה, "צָרִיךְ לִטְעוֹם נִסּוּיִים, עַד אֲשֶׁר יִבָּדֵק הַרָצוֹן הָעַז שֶׁלְּךָ."

שׁוֹשָׁן הִתְחִיל לִשְׂחוֹת עִם הַמַּגִּיעָה בַּיָּם. הַמַּגִּיעָה הוֹבִילָה אוֹתוֹ דֶּרֶךְ שְׁמוּרוֹת וְמִסְתּוֹרִיּוֹת. הוּא שָׂחָה לְמַעְלָה וּלְמַטָּה, עָבַר תְּעָלוֹת צָרוֹת, וְלֹא פָחַד. הַרְבֵּה פִּי עָבַר סַכָּנוֹת שׁוֹנוֹת.

לְבַסּוֹף, אַחֲרֵי הַרְבֵּה שְׁעוֹת שְׂחִיָּה, הוּא הִגִּיעַ לְמָקוֹם יָקָר מְאוֹד. לִפְנֵי הָיָה שַׁעַר גָּדוֹל, עָשׂוּי פְּנִינִים. שׁוֹשָׁן הִתְקָרֵב. עַל הַשַּׁעַר הָיָה כְּתוּב: "כָּל הַפּוֹתֵחַ אֶת הַדֶּלֶת הַזֹּאת, יִמָּצֵא בְּמָקוֹם שֶׁחָלַם עָלָיו."

שׁוֹשָׁן הִסְתַּכֵּל, וְאָנַח. הוּא לֹא רָאָה בּוֹרֵא לְעֶזְרָתוֹ. הוּא לֹא יָדַע כֵּצִיצָה לִפְתֹּחַ אֶת הַדֶּלֶת. הוּא הָיָה דָּג, וְהַדֶּלֶת הָיְתָה עֲצוּבָה. הוּא קִוָּה לַיָּשׁוּעָה, הִתְפַּלֵּל. הוּא חָשַׁב עַל חֲלוֹמוֹ, עַל גַּן הַפְּרָחִים שֶׁרָצָה לִרְאוֹת. הוּא רַק רָצָה לִפְתֹּחַ אֶת הַדֶּלֶת, וְלִרְאוֹת מָה יֵשׁ לְאַחֲרֶיהָ.

בְּנֶבֶג לְלָאוּ מְנִימָה, הוּא רָאָה אֶת הַשַּׁעַר נִפְתָּח! הָאוֹר זָרַח מִתּוֹכוֹ, וְשׁוֹשָׁן יָדַע, שֶׁזֶּה הַגָּן שֶׁחָלַם עָלָיו. הוּא כָּל כָּךְ שָׂמַח, שֶׁהוּא שָׂחָה דֶּרֶךְ הַשַּׁעַר.

הוּא הִגִּיעַ לְגָן מְפֻאַּר, מָלֵא בִּפְרָחִים יָפִים וּצְבָעוֹנִיִּים. הָיָה צְלִילִים מְשֻׁנִּים, הַפְּרָחִים זָרְחוּ בְּכָל צְבַע שֶׁאָהַב. בַּמֶּרְכָּז הָיָה אֲגַם קָטָן וְשׁוּבָב, וּבְתוֹכוֹ שׁוֹשָׁן רָאָה אֶת בְּהַשְׁקָפָתוֹ, מַבִּיט יִתּוֹ.

הוּא שָׂחָה בַּגָּן, הִתְרַחֵץ בֵּין הַפְּרָחִים, וְהָרִילוֹכְּלָה בְּהַנְגָּנָה. הוּא הִרְגִּישׁ שְׁבִיעִי אוֹשֶׁר. הוּא הֵבִין, שֶׁחֲלוֹמוֹ הִתְגַּשֵּׁם. הוּא קִבֵּל אֶת מַה שֶׁכָּל כָּל הַיָּם הַזֶּה הַלֹּא מוּשְׁלַם הָיָה מַעֲנִיק לוֹ - כִּכְלָיוֹת.

כְּשֶׁשׁוֹשָׁן חָזַר לַיָּם, הוּא סִפֵּר לַצִּפּוֹר עַל חוּוָּן. הַצִּפּוֹר שָׂמְחָה מְאוֹד לִשְׁמֹעַ שֶׁשׁוֹשָׁן הִצְלִיחַ לִמְצֹא אֶת הַגָּן. "אֲנִי יוֹדַעַת שֶּׁכָּל אֶחָד יָכוֹל לַעֲשׂוֹת כָּתוּב," אָמְרָה, "אֲבָל אַתָּה הוֹכַחְתָּ שֶׁכָּל דָּבָר אֶפְשָׁרִי, אִם מַאֲמִינִים."

מֵאוֹתָהּ הָעֵת, שׁוֹשָׁן הָיָה יוֹצֵא לְבַקֵּר בַּגָּן הַיָּפֶה כְּכָל שֶׁהוּא רָצָה. הוּא הֵבִין, שֶׁהַחֲלוֹמוֹת אֵינָם רַק דְּבָרִים שֶׁחוֹלְמִים עֲלֵיהֶם, אֶלָּא דְּבָרִים שֶׁצָּרִיך מִין לְהַשִּׁיג. הוּא לֹא וִתֵּר עַל חֲלוֹמוֹ, וְסוֹף סוֹף הִצְלִיחַ לִמְצֹא אוֹתוֹ. וְכָל דָּג תְּמִיד הִבִּיט עַל שָׁמַיִם, לִזְכֹּר שֶׁהַחֲלוֹמוֹת לֹא צָרִיךְ לָעַזֹּב, לֹא לְעוֹלָם.
