בַּיַּעַר הַיָּרֹק וְהַשָׁמֵחַ, בְּרֵאשִׁית הָהָר הַגָּדוֹל, חָיָה מִשְׁפָּחָה שֶׁל גַּמָּדִים. הַגַּמָּדִים הָאֵלֶּה, זֵהִים בְּגוּפָם הַקָּטָן, מֵימַד הַכּוֹבַעִים הָאֲדֻמִּים וְהַזְּקָנִים הַלְּבָנִים, הָיוּ שְׂמֵחִים וְטוֹבִים לֵב. אֲבָל בְּתוֹךְ הַמִּשְׁפָּחָה, הָיָה גַּמָּד אֶחָד שֶׁהָיָה שׁוֹנֶה קְצָת. שְׁמוֹ הָיָה שִׁמְשׁוֹן, וְהוּא אָהַב אֶת הַיִּקֶּרֶת.

שִׁמְשׁוֹן לֹא אָהַב לָעֲבוֹד עִם הָאֲחֵרִים בַּמַּכְּרָה, לֹא אָהַב לִשְׁתּוֹת מֵיץ מִפְּרָחִים, וְכִי מָה שֶׁהָרְגִיז אוֹתוֹ יוֹתֵר מִכֹּל הָיָה לִשְׁחֵק עִם הַגַּמָּדִים הָאֲחֵרִים בַּחֶצֶר. מַה שֶּׁהוּא אָהַב לַעֲשׂוֹת, כְּלֵיל מִכֹּל שְׁאָר הַדְּבָרִים, הָיָה לְבַקֵּר בַּמְּעָרָה הַמְּאוֹרָה.

הַמְּעָרָה הַמְּאוֹרָה הָיְתָה, שׁוּב לְצַרֵי הַשׁוּעַר, מַצְבֵּעַ גָּדוֹל בּוֹלֵט שֶׁל הָהָר. הַמְּעָרָה הָיְתָה מְלֵאָה בְּאַלְפֵי אֲבָנִים יְקָרוֹת, מְשֻׁנּוֹת בְּצוּרָתָן. הָיוּ שָׁם אֲבָנִים כְּמוֹ כּוֹכָבִים קְטַנִים, אֲבָנִים שֶׁצְּבָעָן יָרֹק כְּמוֹ עֲלֵי עֵץ, וַאֲבָנִים מֵאֲשֶׁר בָּלֶהָם בְּל-כָּל. הָאוֹר הַבָּא מִלִּפְנִים, מֵאוֹתָה הַקְּרִינָה שֶׁל הָאֲבָנִים עַצְמָן, הֵאִיר אֶת כָּל הַמְּעָרָה בַּצְּבָעִים שֶׁל הַדִּמְיוֹן.

כָּל יוֹם, כַּאֲשֶׁר הַשֶּׁמֶשׁ הִתְחִיל לִהְיוֹת קָטָן וְנָמוּךְ, שִׁמְשׁוֹן הָיָה לִחְצוֹן אֶת הַקֶּשֶׁר שֶׁלּוֹ, לְעַמֵּס אֶת סַלּוֹ, וְלִצְעוֹד לְעֵבֶר הַמְּעָרָה. שָׁם, הוּא הָיָה שׁוֹעֵל שָׁעוֹת בְּתוֹךְ הַמְּעָרָה הַמְּאוֹרָה. הוּא הָיָה הוֹפֵךְ אֶת הָאֲדָמָה, מְחַפֵּשׂ אֶת הָאֲבָנִים, וְאוֹסֵף אוֹתָן לְתוֹךְ הַסַּלָּה שֶׁלוֹ.

כַּאֲשֶׁר הַסַּלָּה שֶׁלוֹ הָיְתָה מִתְמַלֵּאת, שִׁמְשׁוֹן הָיָה חוֹזֵר לְבֵיתוֹ, כְּשֶׁלִּבּוֹ מָלֵא שִׂמְחָה. הוּא הָיָה מְרַעֲלָה אוֹתָן עַל שֻׁלְחַן הָאֵבָן שֶׁלּוֹ, מְחַיֵּךְ בְּהִתְפַּעֲלוּת, וּמְיַחֵד אֶת הָאֶבֶן הַיָּפָה בְּיוֹתֵר בַּמְּקוֹם הַמְּיֻחָד שֶׁלָּהּ בַּאוֹצָר שֶׁלּוֹ.

אֲבָל יוֹם אֶחָד, מַשֶּׁהוּ שָׁנָה. כַּאֲשֶׁר שִׁמְשׁוֹן נִכְנַס לַמְּעָרָה, הוּא לֹא מָצָא אֶת הָאוֹר הָרָגִיל. הַמְּעָרָה הָיְתָה חֲשׁוּכָה, וְהָאֲבָנִים לֹא נָתְנוּ אֶת הַצְּבָעִים הָיָפִים שֶׁלָּהֶם.

שִׁמְשׁוֹן הִתְעַצֵּב. "מָה קָרָה?" הוּא שָׁאַל בְּקוֹל רָעִיל. הוּא הִתְחִיל לְחַפֵּשׂ, לִידֵעַ מַה הִשְׁתַּנָּה. הוּא הִלֵּךְ בַּחֲשֵׁכָה, מִתְקָרֵב לְקִירוֹת הַמְּעָרָה, עוֹבֵר עַל הָאֲבָנִים הַחֲשׁוּכוֹת.

לְפֶתַע, הוּא שָׁמַע זִמְזוּם מוּזָר. הוּא הִתְקָרֵב לְחַפֵּשׂ אֶת הַקּוֹל, וְרָאָה שְׁתֵּי עֵינַיִם גְּדוֹלוֹת, זוֹרְחוֹת, שֶׁלָּהֶן מָלֵא בְּנֵי הַקֶּרֶן הָעֲנָקִים. הָעֵינַיִם הָיוּ שַׁיָּכוֹת לְעַכָּבִישׁ עֲנָק, שֶׁטּוֹוָה קַוִּי עַכִּישׁוּלִים עֲדִינִים בֵּין הָאֲבָנִים.

"מָה אַתָּה עוֹשֶׂה פֹּה?" שִׁמְשׁוֹן שָׁאַל בִּדְאֵגָה.

הָעַכָּבִישׁ עָנָה בְּקוֹל עָדוּן: "אֲנִי שָׁמַרְתִּי עַל הָאֲבָנִים. אוּלַי שַׂמְתָּ לֵב שֶׁהֵן צְרִיכוֹת מִנוּחָה, וְלֹא שִׁוֵּית נְזִין לְעֵין-צַד?".

שִׁמְשׁוֹן הִתְעַצֵּב. הוּא הֵבִין שֶׁהוּא לֹא הִתְייַחֵס לָאֲבָנִים בּכַּדּוּר. הוּא תָּמִיד רָאָה אוֹתָן רַק כְּאוֹצָר שֶׁלּוֹ, לֹא כְּדָבָרִים חַיִּים, וְאוֹפְנִיָּה עַל הָיַת שׁוּב, שֶׁגַּם לָהֶם יֵשׁ צְרָכִים.

"סְלִיחָה", שִׁמְשׁוֹן אָמַר בּבּוּשָׁה. "לֹא חָשַׁבְתִּי עַל זֶה. אֲנִי רַק רָצִיתִי לֶאֱסֹף אוֹתָן."

הָעַכָּבִישׁ הִצְטַחֵק. "אֲנִי יוֹדֵעַ, קָטָן. אֲבָל הָאֲבָנִים הָאֵלֶּה הֵן יְקָרוֹת, כֵּן, אֲבָל הַיֹּפִי שֶׁלָּהֶן הוּא כַּאֲשֶׁר הֵן בִּמְקוֹמָן. כַּאֲשֶׁר הֵן זוֹרְחוֹת זוֹ, מְעֵיל."

שִׁמְשׁוֹן הִבִּיט בָּעַכָּבִישׁ. הוּא הֵבִין שֶׁהוּא צָדַק.

"שִׁמְשׁוֹן", אָמַר הָעַכָּבִישׁ, "הַאִם תִּרְצֶה לְהַגִּישׁ לָאֲבָנִים בַּקָּשָׁה חֲדָשָׁה? תּוּכַל לְבַקֵּשׁ מֵהֶן לְהֵאוֹר מְעַט יוֹתֵר בְּכַדַּי שֶׁתּוּכַל לְהִתְפָּעֵל מֵהֶן."

שִׁמְשׁוֹן הִנְדַּר סָבָה. הוּא הִתְחִיל לְדַמְּיֵן אֶת הָאֲבָנִים כְּיְצוּרִים חַיִּים, לֹא רַק כְּאוֹצָר שֶׁלּוֹ. הוּא הֵרִים תְּפִלָּה: "אֲבָנִים יְקָרוֹת, סְלִיחָה שֶׁלִּי שֶׁלִּי לֹא יָדַעְתִּי. יַקֵּר הַיֹּפִי שֶׁלָּכֶם, תְּפַלָּלוּ לְיַלְדִּי הַיָּפָה, שְׁמַרְתֶּם לִי. אַתֶּם סְלִיחַתָתְכֶם, יוֹתֵר מִכַּדַּי".

בְּרֶגַע, הָאֲבָנִים הִתְחִילוּ לְהֵאוֹר. הַמְּעָרָה הִתְמַלְּאָה בְּצוּבָּעִים חֲשׁוּכִים וּבְאוֹר זוֹהֵר. שִׁמְשׁוֹן הִתְפַּעֵל. הוּא רָאָה אֶת הַיֹּפִי שֶׁל הָאֲבָנִים בְּצֻרַּת אוֹפֶן חָדָשׁ.

הוּא הִבִּיט בָּעַכָּבִישׁ. "תּוֹדָה", הוּא אָמַר. "עָזַרְתָּ לִי לִרְאוֹת דָּבָרִים בְּצֻרַּת אוֹר."

הָעַכָּבִישׁ חִיֵּךְ. "אֲנִי שָׂמֵחַ."

מֵאוֹתוֹ הַיּוֹם, שִׁמְשׁוֹן הִמְּשִׁיךְ לְבַקֵּר בַּמְּעָרָה. הוּא לֹא הִפְסִיק לֶאֱסֹף אֲבָנִים, כִּי הוּא מֵבִין שֶׁהוּא יִקַּח אוֹתָן מִמִּשְׁפַּחְתוֹ הַגְּדוֹלָה שֶׁלּוֹ. הוּא הִתְיַחֵס אֲלֵיהֶן בְּכַבּוּד, הִתְפַּעֵל מֵהֶן, וְשָׂמֵחַ לְהִשְׁתַּפֵּךְ יִתָּן.

הוּא הִתְחַל לְשַׁתֵּף אֶת הַאֲבָנִים שֶׁלּוֹ עִם הָאֲחֵרִים, מְרֵאָה לָהֶם אֶת הַיֹּפִי שֶׁלָּהֶן. הוּא הִתְחִיל לְשַׂחֵק עִם שְׁאָר הַגַּמָּדִים, לַעֲבוֹד עִמָּהֶם, וְלִשְׁתּוֹת מֵיץ עִמָּהֶם. הוּא הֵבִין שֶׁאוֹצָרוֹת יוֹתֵר יְקָרִים מֵאֲבָנִים הַכִּי יְקָרִים, הֵם הַיְּדִידוּת, הָאַהֲבָה וְהַחֵשֶׁק לְתוֹךְ הַיָּעָר.

וּמֵאוֹתוֹ הַיּוֹם, הַמְּעָרָה הַמְּאוֹרָה הָיְתָה מְלֵאָה שֶׁל אֲהָבָה, כְּמוֹ שֶׁהָאֲבָנִים הָיוּ מְלֵאוֹת. וְאִם תִּמְצְאוּ אֶת עַצְמְכֶם בַּיַּעַר, בְּטַח תִּשְׁמְעוּ אֶת הַכְּתִיבָה שֶׁל שִׁמְשׁוֹן וְהָעַכָּבִישׁ, מִתְגּוֹרְרִים, הוֹפֵךְ לְשִׁיר. סִיּוּם.
