לַשְׁמִיכָה שֵׁם קָרָאנוּ, "שְׁמִיכַת הַלַּיְלָה". לָמָּה? כִּי הִיא הָיְתָה מְגֵנָה עַל יְלָדִים קְטַנִּים כְּשֶׁהַשֶּׁמֶשׁ הָלְכָה לִישֹׁן וְהַלַּיְלָה יָרַד. הִיא הָיְתָה מַגֵּנָה מִכָּל הַפְּחָדִים הַנִּשְׁמָרִים בְּאֹפֶל, מִשְׁחָקִים וּמִטֶּבְעָת הַגַּעְגּוּעִים לַשְׁכִינוּת מִן הַפְּרָחִים וּשְׁזָרוֹתָם שֶׁבְּהָשְׁעֵרֵת הַשָּׁנָה הַבָּאָה.
אֲבָל הַדָּבָר הֲכִי חָשׁוּב בַּשְׁמִיכָה שֶׁלָּנוּ הָיָה שֶׁהִיא שָׁמְרָה עַל הַחֲלוֹמוֹת שֶׁל הַיְלָדִים. הִיא אָסְפָה אוֹתָם בִּכְבִישָׁן מַחְשָׁרָתִי וְשָׁמְרָה עֲלֵיהֶם כְּמוֹ בִּתִּי סְטִיכִית לְלֹא תְּחוּם עַד הַבֹּקֶר, כְּשֶׁהַיּוֹם הָיָה פּוֹקֵחַ עֵינָיו מֵחָדָשׁ.
בְּכָל לַיְלָה, כְּשֶׁהַיְלָדִים נִשְׁכְּבוּ לִישֹׁן, שְׁמִיכַת הַלַּיְלָה הָיְתָה מְכַסָּה אוֹתָם בְּרַכּוּתָהּ, וְכֹל הַחֲלוֹמוֹת שֶׁלָּהֶם הִתְחִילוּ לְצֹפִי.
לְיַד שְׁמִיכַת הַלַּיְלָה הִשְׁתַּרְרָה קְטַנָּה, שִׁירָה שֶׁקּוֹרְאִים לָה לִי לִי. לִי לִי הָיְתָה יַלְדָּה קְטַנָּה שֶׁאָהֲבָה לִרְאוֹת בְּכָל הַחֲלוֹמוֹת הָאֵלֶּה, וְגַם הִיא רָצְתָה לִשְׁמוֹר עַל הַיְּלָדִים וּלְדַאֵג שֶׁיִּהְיֶה לָהֶם חֲלוֹמוֹת יָפִים.
"כֵּיצַד אֲנִי יְכוֹלָה לְשַׁמֵּר עַל הַחֲלוֹמוֹת?" שְׁאֲלָה לִי לִי פַּעַם אֶת שְׁמִיכַת הַלַּיְלָה.
הַשְׁמִיכָה צָחֲקָה קְצָת, וְהִיא עָנְתָה בְּקוֹל מָלֵא אֲהָבָה: "לִי לִי יְקָרָה, אַתְּ כְּבָר שׁוֹמֶרֶת עַל הַחֲלוֹמוֹת. בִּזְכוּת לֵבֵך הַטּוֹב, וּבְעֶזְרַת הַחִיּוּךְ שֶׁלָּךְ."
לִי לִי שְׁמָעָה אֶת זֶה וְחִיְּכָה חֲזַר. "כִּי שָׂמַחְתִּי", חָשְׁבָה, "אָז יֵשׁ בִּי מַשְׁמַעּוּת לְלֹא לִכְאוֹב?"
מֵאוֹתוֹ הַיּוֹם, לִי לִי הִתְחִילָה לְנַסּוֹת דְּבָרִים שׁוֹנִים כְּדֵי לְשַׁמֵּר עַל הַחֲלוֹמוֹת שֶׁל הַיְלָדִים. הִיא הָיְתָה מְעַצֶּבֶת בַּצְלִילִים אֲלָם מִנְיַל שֶׁלְּטֶכֶּם נָגֵב אוֹ תְּנִימָה מְכַנֶנֶת בַּשְׁעַת הַחֵן. הִיא שְׁרַרְהָ שִׁירִים קְטַנִּים וְשַׁמְעֲתִים בְּשֵׁם הָאַגָּדָה, עִם יְלָדִים מִבְּנֵי הַיָּשְׁבִים.
בְּאֶחָד מֵהַיָּמִים, קָרָה מַשֶּׁהוּ מְיֻחָד. יֶלֶד שֶׁהָיָה צָרִיךְ לִנְסֹעַ מֵעִיר לְעִיר, בְּהַשְׁבָּחַת לֵב לְהַשְׁחָה. הוּא הָיָה בּוֹכֶה בִּשְׁעַת הַלֵּכֶת וְרָעָה לִישֹׁן, שֶׁהַכּוֹכָבִים שֶׁלּוֹ לֹא יִשְׁמְרוּ אוֹתוֹ.
לִי לִי רָאֲתָה אוֹתוֹ בְּעֵינֶיהָ הַגְּדוֹלוֹת וְהַמְּלֵאוֹת. הִיא חִיְּכָה אֵלָיו וְנִגְּשָׁה אֵלָיו בְּשֶׁקֶט, וְאָמְרָה לוֹ: "אַל תִּדְאַג, יֶלֶד יָקָר. שְׁמִיכַת הַלַּיְלָה תִּשְׁמֹר עָלֶיךָ וְעַל חֲלוֹמוֹתֶיךָ. הִיא תִּסַּע אִתְּךָ לְכָל מָקוֹם."
הַיֶּלֶד הִקְשִׁיב לָהּ בִּכְתִישָׁה, וְשַׁעַר דְּמָעוֹתָיו יָבַשׁ. הוּא חָיְּךָ וְעָנָה לָהּ: "אֲנִי מְקַוֶּה שֶׁזֶּה נָכוֹן."
כְּשֶׁהַלַּיְלָה יָרַד, הַיֶּלֶד נִכְנַס לַמִּטָּה, וְשְׁמִיכַת הַלַּיְלָה עֲטָפָה אוֹתוֹ. לִי לִי עָמְדָה לְיַד הַמִּטָּה וְחִיְּכָה. הִיא שָׁרָה לוֹ שִׁיר עָדִין עַל מַסָּע, עַל מִשְׁפָּחָה וְעַל דְּרָכִים חֲדָשׁוֹת, שֶׁיֵּעָלֵמוּ מִבָּא יָגוֹד שׁוֹנֶה.
הַיֶּלֶד שָׁמַע אֶת הַשִּׁיר וְעֵינָיו נֶעֶצְמוּ. הוּא נִרְדָּם שָׁלֵו וְחָשׁ כָּל סְטִיעַ עַל הַנִּשְׁמָה.
בַּחֲלוֹם, הוּא רָאָה אֶת עַצְמוֹ הוֹלֵךְ עַל הַכּוֹכָבִים. לִי לִי שֶׁלִּי לָה אִתּוֹ. הֵם צָחֲקוּ וְטִיְּלוּ יַחַד, וְהָיָה כְּפִלְיֶיךָ שֶׁלְּאוּלִמִּית שֶׁבִּפְגִישָׁתְהֶם כְּדֵי לִמְצֹא צִיר אֲפָקִים.
כְּשֶׁהַבֹּקֶר הִגִּיעַ, הַיֶּלֶד הִתְעוֹרֵר בְּחִיּוּךְ. הוּא הָיָה שָׂמֵחַ וְשַׁלְּוָם, וְהָיָה אִכַּפְתִּי לוֹ הַגָּעָה מֵאִיתָנוּ.
הוּא יָדַע שֶׁשְׁמִיכַת הַלַּיְלָה תִּשְׁמֹר עָלָיו בְּכָל מָקוֹם שֶׁיֵּלֵךְ, וְלֹא חָשַׁב יוֹתֵר עַל הַדּוּק מִסַּעֲף הַנֶּפֶשׁ.
לִי לִי רָאֲתָה אוֹתוֹ מִחְפְּצֵנִי, וְלִבָּהּ הָיָה מָלֵא שִׂמְחָה. הִיא יָדְעָה שֶׁהִיא יְכוֹלָה לְשַׁמֵּר עַל הַחֲלוֹמוֹת שֶׁל הַיְלָדִים, לֹא רַק עַל יְדֵי לִהְיוֹת שַׁיֶּשֶׁת לַכַּשְׁנָתָן, אֶלָּא גַּם עַל יְדֵי אֲהָבָה וְחִיּוּךְ.
מֵאוֹתוֹ הַיּוֹם, לִי לִי הָפְכָה לְשׁוֹמֶרֶת הַחֲלוֹמוֹת הַמֻּכְשֶׁרֶת שֶׁל כָּל הַיְּלָדִים. הִיא הָיְתָה שְׂמֵחָה וְתָמִיד רַק שָׂמֵחַ, וְתָמִיד כִּשְׂמָאתוֹ כַּדָּעוּ. הִיא עָזְרָה לַיְלָדִים לְהִתְגַּבֵּר עַל הַפְּחָדִים, וְלִשְׁמֹר עַל הַחֲלוֹמוֹת שֶׁלָּהֶם שׁוּב.
וְכָךְ, בִּשְׁמִיכַת הַלַּיְלָה וּלְיַד לִי לִי, כָּל הַיְּלָדִים יְכֹלוּ לָלֶכֶת לִישֹׁן בְּשַׁלְוָה, כְּשֵׁלַמְתִּי כַּוֹּפֵף הַלִּמּוּדִי לְקָרַת יָכֹל לִבְעֹל מַה שֶּׁקָּרָה. וְכָךְ כָּל הַלַּיְלוֹת, שֶׁל כָּל הַיְּלָדִים, הָיוּ מְלֵאִים בַּחֲלוֹמוֹת מְתֻקִּים, בְּסִיפּוּרִים מְתֻקִּים, וּבְרַגְשׁוֹת טוֹבִים.
וְהַשְּׁמִיכָה הַכְּחוּלָה תָּמִיד הָיְתָה שָׁם, לְשַׁמֵּר עֲלֵיהֶם, וּלְהַזְכִּיר לָהֶם שֶׁהֵם בְּטוּחִים וְאָהוּבִים, תָּמִיד וּלְעוֹלְמֵי עוֹלָמִים. וְכֵן, הַסֵּפֶר הַזֶּה הוֹדִיעַ, שֶׁגַּם לָנוּ, בָּרֹאשׁ עֲבֹדָה רָאשִׁית, בַּיָּמִים הַאֲרוּכִים, בְּסוֹף הַכֹּל - הָאָדָם הוּא אוֹהֵב, וְהָאוֹהֵב הוּא עָקְבִי.
