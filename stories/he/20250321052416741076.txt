<p>בַּיָּעַר הָרָחוֹק חָיוּ דֹּב שָׁמֵן וְחָבִיב וְדוּבְלִית קְטַנָּה וּמְתוּקָה. לְדֹב הָיָה שֵׁם - דּוּבּוּ, וְלַדּוּבְלִית הָיָה שֵׁם - דּוּבְּלִיתָא. הֵם אָהֲבוּ לִחְיוֹת יַחַד בְּבַיִת מְיֻחָד עָשׂוּי מִקּוֹר עֵץ גָּדוֹל וְיָשָׁן.</p>

<p>בְּכָל שָׁנָה, כְּשֶׁהַקַּר בְּעֵינָיוּ עֲבָר, הָיוּ דּוּבּוּ וְדּוּבְּלִיתָא נִרְדָּמִים שְׁנֵי הֶרְדֵּמִים אֲרוּכִּים בְּתוֹךְ הַבַּיִת שֶׁלָּהֶם. הֵם הָיוּ יְשֵׁנִים כָּל הַחוֹרֶף. כְּשֶׁהָאָבִיב הִתְחִיל לְהַגִּיעַ, הָיָה דּוּבּוּ מִתְעוֹרֵר בָּרִאשׁוֹנָה. הָיָה לוֹ תָּמִיד רָעָב גָּדוֹל אַחֲרֵי הַשֵּׁנָה הָאֲרוּכָּה.</p>

<p>בְּיוֹם אֶחָד, בְּאוֹתוֹ אָבִיב מְיֻחָד, דּוּבּוּ פָּקַח אֶת עֵינָיו. הַשֶּׁמֶשׁ זָרְחָה עַל הָיַּעַר, וְהָיָה חַם בַּחוּץ. "דוּבְּלִיתָא! דוּבְּלִיתָא! קוּמִי! הָאָבִיב הִגִּיעַ!" קָרָא דּוּבּוּ בְּקוֹל גָּדוֹל וּמְלֵא שִׂמְחָה.</p>

<p>דּוּבְּלִיתָא פָּתְחָה אֶת עֵינֶיהָ בַּלֵּאט. "אוֹ, שֶׁמֶשׁ, שֶׁמֶשׁ!" אָמְרָה בְּקוֹל נָעִים. הִיא הִסְתַּכְּלָה עַל דּוּבּוּ וְחִיְּכָה. "אֲנִי רְעֵבָה בִּמְיֻחָד הַיּוֹם."</p>

<p>הֵם יָצְאוּ מֵהַבַּיִת וְהִתְחִילוּ לְטַיֵּל בַּיַּעַר. כָּל הַצְּמָחִים הָיוּ יְרֻקִּים וְשֶׁפַע, וְהַצִּפֳּרִים שָׁרוּ בְּקוֹלוֹת יָפִים. דּוּבּוּ וְדּוּבְּלִיתָא הָלְכוּ וְהָלְכוּ, מְחַפְּשִׂים מַשֶׁהוּ לֶאֱכוֹל.</p>

<p>לְפֶתַע, דּוּבּוּ רָאָה מַשֶּׁהוּ מְעַנְיֵן בְּאֶחָד הַטּוּבִים שֶׁל הָעֵצִים. "הֵי, דּוּבְּלִיתָא, בּוֹאִי! רְאִי אֶת זֶה!" קָרָא דּוּבּוּ בִּדְבָרִים סִימְנָיִים.</p>

<p>דּוּבְּלִיתָא רָצָה אֶל דּוּבּוּ. בְּעֵץ הָיָה כַּוֶּרֶת דְּבוֹרִים עֲנָקִית. "אוֹ, יַי!" אָמְרָה דּוּבְּלִיתָא. "דְּבַשׁ!"</p>

<p>דּוּבּוּ לִיקֵּק אֶת שְׂפָתָיו. "דְּבַשׁ הָאוֹהֶב הַכִּי טוֹב בָּעוֹלָם! אֲבָל אֲנִי לֹא רוֹצֶה שֶׁהַדְּבוֹרִים יַעֲקוּץ אוֹתָנוּ."</p>

<p>"אֲנִי יוֹדַעַת!" אָמְרָה דּוּבְּלִיתָא בְּהִתְלַהֲבוּת. "בּוֹא נִמְצָא עֲלֵי שַׁיִשׁ מְרֻחַ שֶׁל תִּפּוּחִים. הַדְּבוֹרִים לֹא יַעֲמִידוּ לָנוּ מִתְּנַגְּדִים, וְנִזְכֶּה בַּדְּבַשׁ הַטָּעִים!"</p>

<p>דּוּבּוּ הִסְכִּים בִּמְיָדִיּוּת. הֵם הָלְכוּ לְהִסְתַּכֵּל בַּסְּבִיבָה שֶׁל הַכַּוֶּרֶת שֶׁלָּהֶם, מְחַפְּשִׂים תִּפּוּחִים. כְּשֶׁהֵם הִגִּיעוּ לְמַטֶּה גַּם שֶׁלָּהֶם הֵם מָצְאוּ גַּן יָפֶה, וְשָׁם גַּם עֲצֵי תִּפּוּחִים. מְכִיבִּים עַל הָאָדְמָה.</p>

<p>"וַאֲנַחְנוּ עֲכַשִׁיב בְּסֵדֶר." אָמַר דּוּבּוּ. הֵם גִּלְּחוּ שִׁבְעַת מַכּוֹנוֹת תִּפּוּחִים וְהֵבִיאוּ אוֹתָם לְיַד הַכַּוֶּרֶת.</p>

<p>דּוּבְּלִיתָא הִתְחִילָה לְמַשִּׁיחַ אֶת הַתִּפּוּחִים בַּמַּשְׁכִּיט. לְאֲשֶׁר הִתְפָּעַל, הַדְּבוֹרִים שַׂמְחוּ לֶאֱכֹל אֶת הַתִּפּוּחִים, וְהִתְחִילוּ לִהְיוֹת פְּחוּת עַקְשָׁנִים. כָּךְ, דּוּבּוּ וְדּוּבְּלִיתָא יְכָלוּ לָקַחַת אֶת הַדְּבַשׁ בְּשָׁלוֹם!</p>

<p>הֵם חָזְרוּ לְבַיִּתָם עִם כְּלִי מָלֵא דְּבַשׁ. "אֲנִי כְּלִי יָשִׁין!" אָמַר דּוּבּוּ. "זֶה הָיָה רָעָב גָּדוֹל!"</p>

<p>"גַּם אֲנִי!" אָמְרָה דּוּבְּלִיתָא. "בּוֹא נֶאֱכֹל אֶת הַדְּבַשׁ הַטָּעִים שֶׁלָּנוּ."</p>

<p>הֵם יָשְׁבוּ וְאֲכָלוּ אֶת הַדְּבַשׁ עַד שֶׁהָיוּ שְׂבֵעִים. הָיָה כָּל כָּךְ טָעִים! הַמְּיֻחָד בְּמִינוֹ!</p>

<p>אַחֲרֵי שֶׁאֲכָלוּ, הִתְחִילוּ לְשַׂחֵק בַּיַּעַר. דּוּבּוּ רָץ אַחֲרֵי דּוּבְּלִיתָא, וְהֵם צָחֲקוּ וְצָחֲקוּ. הֵם מָצְאוּ פִּרְחֵי שֵׂעָר וְקָבְרוּ אוֹתָם בִּשְׁעָרָם.</p>

<p>בְּאֶחָד הַיָּמִים, כַּאֲשֶׁר שַׂחֲקוּ בַּנָּהָר, שְׁמָעוּ רַעַשׁ מוּזָר. "מָה זֶה?" שָׁאֲלָה דּוּבְּלִיתָא.</p>

<p>דּוּבּוּ הִסְתַּכֵּל. "נִרְאֶה שֶׁיֵּשׁ הַרְבֵּה יְלָדִים שַׂחֲקָנִים!"</p>

<p>הֵם הִתְקָרְבוּ לְהָגִיד כִּי שֶׁרַק תְּקִיעוֹת. הִתְבָּרֵר שֶׁיֵּשׁ קְבוּצַת יְלָדִים שַׂמָּחָה שֶׁשָּׂחֲקָה עַל הַגִּבְעָה. דּוּבּוּ וְדּוּבְּלִיתָא הֵצִיעוּ לְהַכְנִיס לְמַשְׁחָק אֶת הַיְּלָדִים.</p>

<p>הַיְּלָדִים שְׂמָחוּ לִשְׁמוֹעַ. הֵם הִתְחִילוּ לְשַׂחֵק יַחַד. הָיוּ מַשְׁחָקִים כְּמוֹ מַחְבּוֹאִים, הֵקְפוֹת, וְעוֹד הַרְבֵּה! כּוּלָּם צָחֲקוּ וְהִתְרַגְּשׁוּ יַחַד. זֶה הָיָה כִּיף גָּדוֹל!</p>

<p>כַּאֲשֶׁר הַשֶּׁמֶשׁ הִתְחִילָה לִשְׁקֹעַ, הַיְּלָדִים הָלְכוּ לְבָתֵּיהֶם. דּוּבּוּ וְדּוּבְּלִיתָא חָזְרוּ לְבֵיתָם. הֵם הָיוּ עֲיֵפִים מֵהַכֵּיף, אֲבָל הָיוּ שְׂמֵחִים.</p>

<p>"הַיּוֹם הָיָה מוּשְׁלָם!" אָמַר דּוּבּוּ.</p>

<p>הֵם חִבְּקוּ אִישׁ אֶת חֲבֵרוֹ וְהָלְכוּ לִישֹׁן בְּבָטַחַת בַּבַּיִת שֶׁלָּהֶם. כָּל הַלֵּיל הָיוּ מְחַיְּכִים בַּחֲלוֹמוֹתֵיהֶם, נִזְכָּרִים בַּיּוֹם הַיָּפֶה שֶׁעָבַר עֲלֵיהֶם. הַיָּעַר שֶׁלָּהֶם תָּמִיד הָיָה מָלֵא בִּתְחוּשׁוֹת אוֹשֶׁר, וְהֵם הָיוּ מְצַפִּים לְאוֹתָם יְמֵי אָבִיב מְתוּקִים שׁוּב בַּשָּׁנָה הַבָּאָה.</p>

<p>לִפְנֵי שֶׁנִּרְדְּמוּ, דּוּבּוּ וְדּוּבְּלִיתָא שָׁמְעוּ מִחוּץ לְבֵיתָם צִפּוֹר שֶׁשָּׁרָה שִׁיר עֲרִיסָה מְתוּקָה עַל אָבִיב וְעַל חַיִּים מְלֵאִים כֵּיף וְשֶׁמֶשׁ. כַּאֲשֶׁר שָׁמְעוּ אֶת הַשִּׁיר, דָּבְשָׁם נָעוֹר בְּתוֹכָם, מְסַפֵּר לָהֶם שֶׁהָאָבִיב הִגִּיעַ, וְעִמּוֹ - הַרְבֵּה יוֹתֵר שִׂמְחָה.</p>

<p>בָּאִי יָמִים, דּוּבּוּ וְדּוּבְּלִיתָא הָיוּ גְּדֵלִים, אֲבָל הַשֶּׁקֶר בֵּין הַתִּשְׁבְּרוֹת מָשַׁךְ לָהֶם הַרְבֵּה רְצוֹן שֶׁל שַׁוִּי הָאוֹשֶׁר, מָכִּי מְעִירִים עַל הָאָבִיב הָאָפֶן. לְעוֹלָם הֵם לֹא שָׁכְחוּ אֶת הַיּוֹם הָרִאשׁוֹן שֶׁל הָאָבִיב הָזֶה. הָיָה בְּדִיּוּק יָפוֹת.</p>
