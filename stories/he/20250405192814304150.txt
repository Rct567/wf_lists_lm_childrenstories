לָמָּה הֵם הָיוּ צִבְעוֹנִיִים? כִּי הֵם הָיוּ צֶבַע! אֲדוּם, כָּחוֹל, יָרוֹק, צָהֹב, סָגֹל, תְּכֵלֶת, בָּחוּם – וְעוֹד הַרְבֵּה. כָּל צֶבַע וְצֶבַע הָיָה לוֹ אֹפִי מְיֻחָד. הָאָדֹם הָיָה חָזָק וּמְאֻשָּׁר, הַכָּחוֹל הָיָה שָׁקֵט וְטוֹב לֵב, הַיָּרֹק הָיָה טִפּוּסִי וּפִיכְחִי, הַצָּהֹב הָיָה שָׂמֵחַ וּמַעֲרִיץ שֶׁמֶשׁ, וְהַסְּגֹל הָיָה תַּעֲלוּמָאִי וְחָכָם. וְהַשְׁאָר? הַשְׁאָר כְּמוֹ כָּל הַצְּבָעִים הָאֲחֵרִים, צִבְעוֹנִיִים, מְרַתְּקִים, וְטוֹבִים.

בַּיּוֹם מִן הַיָּמִים, נִכְנְסָה לַבַּיִת תֵּינוֹקֶת קְטַנָּה. שְׁמָהּ הָיָה תָּמָר, וְהִיא הִיא הָיְתָה אוֹהֶבֶת לְצַיֵּר יוֹתֵר מִכֹּל דָּבָר בָּעוֹלָם. תָּמָר נָשְׂאָה דַּף לָבָן, וּלְפָנֶיהָ הָיְתָה הַמְּשִׂימָה הַגְּדוֹלָה מִכֻּלָּם: לְצַיֵּר תְּמוּנָה.

"אֲנִי צְרִיכָה תְּמוּנָה יָפָה," אָמְרָה תָּמָר בְּקוֹל רָם, "אֲבָל אֵיךְ אֲנִי יְכוֹלָה לְהַתְחִיל?"

"שׁוֹבָב," אָמַר הַכָּחוֹל, שֶׁיָּשַׁב לְיַד הָאֲדוּם. "הָאֲדֻמִּים תָּמִיד רוֹצִים לְהַגִּיד דְּבָרִים חֲשׁוּבִים." הוּא בָּרַח לָרֹאשָׁה שֶׁל תָּמָר. "אֲנִי אֲצַיֵּר אֶת הַיָּם! הַיָּם הוּא דָּבָר יָפֶה וְרָגוּעַ."

הַיָּרֹק, שֶׁהָיָה יָשׁוּב קַפְצָן מִכָּל אֲחֵרִים, קָפַץ גַּם הוּא וְקָרָא: "אֲנִי אֲצַיֵּר עֵץ עָנֵק! עֵץ מָלֵא עָלִים וְשִׁירִים!"

הַצָּהֹב צָהַל בְּשִׂמְחָה. "לֹא, לֹא! אֲנִי אֲצַיֵּר שֶׁמֶשׁ גְּדוֹלָה וְחַמָּה, שֶׁתָּאִיר אֶת הַכֹּל!"

הַסְּגֹל, בִּנְיוֹן חַכְמָתוֹ, אָמַר: "חֲכוּ רֶגַע. רַק אִם נִשְׁלֵם אֶת הַתְּמוּנָה כֻּלָּהּ, נִתֵּן לָהּ אֶת הַיֹּפִי שֶׁהִיא צְרִיכָה."

תָּמָר צָחֲקָה מִכָּל הַשְׁטֻיּוֹת שֶׁל הַצְּבָעִים. "טוֹב, חֲבִיבִּי, אֲנִי אֲתַן לְכָל אֶחָד מִכֶּם לִצְיֵּר קְצָת. אֲבָל בְּסוֹף, הַכֹּל יִצְטָרֵף יַחַד!"

הִתְחִיל הָאָדֹם לְצַיֵּר לֵב גָּדוֹל וְדַלְתָּנִי. הַכָּחוֹל הִתְחִיל לִבְנוֹת אֶת הַיָּם הָרָגוּעַ, מָלֵא בַּבְּרִיזֶה הָרַכָּה. הַיָּרֹק צִיֵּר אֶת גְּזַע הָעֵץ הָעָנֵק, וְאֵלָיו הִתְחִילִים לְהִתְאַסֵּף עָלִים. הַצָּהֹב בָּנָה אֶת הַשֶּׁמֶשׁ הָרַבָּה, שֶׁהוֹסִיפָה חֲוָיָּה נָעִים.

תָּמָר, שֶׁרָאֲתָה אֶת כָּל זֹאת, נֶהֱנֵית מִכָּל מִדְרָכִים. "לֹא לְהִתְוַכֵּחַ," אָמְרָה. "הָאַהֲבָה תּוֹרִיד וְתַכְנִיס לְכָל אֶחָד מִכֶּם אֶת הַמִּלּוֹת."

הִיא הִתְחִילָה לְצַיֵּר יַחַד אִתָּם. הִיא עָזְרָה לָאָדֹם לְהַשְׁלִים אֶת הַלֵּב שֶׁלּוֹ, כָּל הָעֵיקָרוֹן הָעוֹצְמָתִי שֶׁלּוֹ. הִיא עָזְרָה לַכָּחוֹל לְהַשְׁלִים אֶת הַיָּם, הַסּוּלְטָן שֶׁל הַבְּרִיזֶה וְשֶׁל הַנְּצִיבִּיּוּת. הִיא עָזְרָה לַיָּרֹק לְהוֹסִיף עוֹד עָלִים וּפֵרוֹת לָעֵץ שֶׁלּוֹ, בִּמְחוּר כִּוּוּי טוֹב. הִיא עָזְרָה לַצָּהֹב לְהַגְדִּיל אֶת הַשֶּׁמֶשׁ, לְהָאִיר וְלִשְׁמֹעַ שָׁרוּי בָּהּ.

בְּסוֹפוֹ שֶׁל דָּבָר, כֻּלָּם שִׁתְּפוּ פְּעֻלָּה, וְצִבְעוּ יַחַד. הַלֵּב הָפַךְ לְעוֹד יוֹתֵר יָפֶה, הַיָּם הִתְחִיל לִצְחֹק עַל יְדֵי הַשְׁקִיפוּת שֶׁלּוֹ, הָעֵץ נִהְיָה מָלֵא בְּיוֹתֵר מְנוֹת אֲכִילָה, וְהַשֶּׁמֶשׁ הֵאִיר עַל הַכֹּל.

לְפֶתַע, הֵחֵל הָרֵיק לְהִתְמַלֵּא. הַצְּבָעִים לֹא הֵבִינוּ מָה קוֹרֶה.

"מָה זֶה?" שָׁאֲלָה תָּמָר.

הַסְּגֹל חִיֵּךְ. "זֶה מָה שֶׁקּוֹרֶה כְּשֶׁמִּשְׁתַּפְּפִים יַחַד! זֶה הַיֹּפִי שֶׁל הַשְּׁתּוּף פְּעֻלָּה!"

הַתְּמוּנָה הִתְחִילָה לְהִתְמַלֵּא בִּכְאֵב, בְּמֵרוּצָה, בְּיֶשׁ תְּמוּנָה יָפָה, מְמֻיֶּנֶת. הַתְּמוּנָה הִתְמַלְּאָה בִּשְׁמֵי כְּחוּלִים, עֲנָנִים לְבָנִים, צִפֳּרִים מְעֹפְפוֹת, גִּבְעוֹלוֹת יְרֻקִּים, דֶּשֶׁא נָעִים, וּפְרָחִים בְּכָל הַצְּבָעִים.
וּבְמֶרְכָּז הַכֹּל, שָׁלֵם וְטוֹב, עָמַד לֵב גָּדוֹל, חַם וְאַהֲבָתִי.

תָּמָר הִסְתַּכְּלָה עַל הַתְּמוּנָה בְּשִׂמְחָה גְּדוֹלָה. "זֹאת הַתְּמוּנָה הֲכִי יָפָה שֶׁרָאִיתִי!"

וְכָךְ הַצְּבָעִים הַצִּבְעוֹנִיִים הֵבִינוּ שֶׁכַּאֲשֶׁר עוֹבְדִים יַחַד וּמִשְׁתַּפְּפִים, יְכוֹלִים לְצַיֵּר אֶת הַדְּבָרִים הֲכִי יָפִים בָּעוֹלָם. וּמֵאוֹתוֹ הַיּוֹם, כָּל הַצְּבָעִים הֵמְשִׁיכוּ לִחְיוֹת בְּבֵית הַצֶּבַע, לְצַיֵּר, לִצְחֹק, וְלִהְיוֹת שְׁמֵחִים יַחַד. וְאִם תִּשְׁאֲלוּ אוֹתִי, אֵין דָּבָר טוֹב מִזֶּה!
