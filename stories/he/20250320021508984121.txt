אֶרֶץ הַקֶּרַח הִרְשְׁתָה לָהֶם לְשַׁחֵק יַחַד עַד שֶׁהַשֶּׁמֶשׁ שָׁקְעָה, וְעִם יְרִידַת הָעֶרֶב, פֶּפֶּה שִׁיר לִבּוֹ כִּי הַחַיִּים יָפִים כְּשֶׁאֶפְשָׁר לִהְיוֹת יַחַד. הוּא יָדַע שֶׁהוּא בַּבַּיִת, וְשֶׁיֵּשׁ לוֹ חֲבֵרִים, וְשֶׁהוּא יָכוֹל לְהַתְחִיל חַיִּים חֲדָשִׁים.

מֵאוֹתָם הַיּוֹם, פֶּפֶּה חַי בְּאֶרֶץ הַקֶּרַח. הוּא לֹא שָׁכַח אֶת הַחֲבֵרִים שֶׁלּוֹ, וְכַאֲשֶׁר הוּא מָצָא מְעֵט עֵת הָיָה מַזְמִין אוֹתָם לְבַקֵּר אוֹתוֹ. הוּא אָהַב לִשְׁחֵק, לָשִׁיר, וְלִשְׁתוֹת דָּגִים עִם חֲבֵרָיו, שֶׁנִּגְלַוּ כַּחֲשִׁיבִים בְּמֶשֶׁךְ הַמַּסָּע הַזֶּה. הוּא יָדַע שֶׁאֵין דָּבָר יָפֶה יוֹתֵר מֵחַיִּים טוֹבִים בְּחֶבְרָה טוֹבָה.

כַּאֲשֶׁר הוּא הִסְתַּכֵּל בַּכּוֹכָבִים שֶׁהִתְחִילוּ לָעוּף עַל הַשָּׁמַיִם, הוּא יָדַע שֶׁהוּא מְאֻשָּׁר. הוּא לֹא חָשַׁב עוֹד עַל הַפַּחַד שֶׁל הַדֶּרֶךְ, אֶלָּא עַל הַטּוּב וְהַמִּשְׂחָקִים שֶׁמְּחַכִּים לוֹ. הוּא הָפַךְ לִמְשֻׁבָּח בְּהַמְשֵׁךְ הַנַּעֲרוּת שֶׁלּוֹ, הָיָה יָדוּעַ כְּפִינְגְּוִין הַשַּׂמֵּחַ, שֶׁתָּמִיד אוֹמֵר, "אֲנִי לֹא חַי בִּשְׁבִיל הַשְּׁחִיָּה, אֶלָּא לְהַרְגִּישׁ שֶׁאֲנִי חַי".

וְכָךְ, פֶּפֶּה הַפִּינְגְּוִין הַקָּטָן, חַי בְּאֶרֶץ הַקֶּרַח בְּשִׂמְחָה וּבְטַעַם. וְאִם תִּהְיוּ יוֹם אֶחָד שָׁם, תּוּכְלוּ לִרְאוֹת אוֹתוֹ מְשַׁחֵק בַּקֶּרַח, מִשְׁתּוֹת אֶת דַּרְכּוֹ לִכְיוָון הַהֲנָאָה.
