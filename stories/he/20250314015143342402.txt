<p>לְפָנֵי שֶׁהֵעֵזוּ לָרֶדֶת, שָׁאֲלוּ אֶת פִּירְחַת, הַיַּלְדָּה שֶׁיָּכְלָה לִשְׁמֹעַ אֶת לֵב הַטֶּבַע, "פִּירְחַת, מַה אַתְּ שׁוֹמַעַת?" פִּירְחַת הִקְשִׁיבָה בִּקְשִׁיבוּת רַבָּה וְאָמְרָה: "אֲנִי שׁוֹמַעַת שְׁקִיקָה רְגִישָׁה. שְׁקִיקָה שֶׁל צֹרֶךְ. צֹרֶךְ לְעֶזְרָה."</p>

<p>לֹא חָלַף זְמַן רַב וְהֵם יָרְדוּ בַּמַּעֲבָר הַמְּחֻשְׁמָל. הָאַוֵּיר שָׁם הָיָה קָר וְלָח, וְהָאוֹר הָיָה בִּשְׁלֵשָׁה מְאוּד מֻעַט. הֵם הִשְׁתַּתְּפוּ בְּמַעֲרֶכֶת מְעָרוֹת אֲרֻכָּה, וּלְבַסּוֹף הִגִּיעוּ לְאֶחָד הַחֲדָרִים הַגְּדוֹלִים בְּיוֹתֵר שֶׁרָאוּ מִיּוֹמָם. הַחֶדֶר הָיָה מָלֵא בְּכָל מִינֵי מִכְשִׁירִים, קְעָרוֹת וְכָל מִינֵי עֲצָמִים שֶׁלֹּא יָדְעוּ לְהַסְבִּיר אֶת פְּשִׁיטוּתָם. בְּאֶמְצַע הַחֶדֶר עָמַד עַצְרִי בּוֹהֵק, וְעָלָיו יוֹשֵׁב עָרִיץ עַתִּיק – גָּדוֹל עֲצוּם, עִם עֵינַיִם קוֹדְחוֹת מַמָּשׁ כְּבַרְזֶל, מְשֻׁכָּר בְּכֹחוֹ.</p>

<p>בְּאִישׁ הַמֶּרְכָּז שָׁאַל בְּקוֹל צַרְדָּנִי, "מִי אַתֶּם וּמַדּוּעַ בָּאתֶם לְהַפְרִיעַ?" עִידָן הֵשִׁיב: "אֲנַחְנוּ מֵעִיר הַפְּלָאוֹת. שָׁמַעְנוּ שֶׁאַתָּה צָרִיךְ עֶזְרָה."</p>

<p>"עֶזְרָה?!" צָחַק הַיָּשָׁן בְּשִׁנֵּי אַרְגָּז. "אֵינִי צָרִיךְ שׁוּם עֶזְרָה. אֲנִי שַׁלִּיט עַל כָּל זֶה! אֲנִי הַשְׁכּוּנָה."</p>

<p>כְּשֶׁשָּׁמְעוּ אֶת זֹאת, הִתְחִילוּ הַיְּלָדִים לִכְעֹס. פִּינִי צִיְּרָה כְּמוֹת הַשְׁרָעָה אַלְּפָה שֶׁלֹּא הוּשְׁלְמָה לִפְנֵי רֶגַע, וּמִמֶּנָּה שָׁמְעוּ הַיַּלְדִּים מוּסִיקָה. גַּם הַצִּפּוֹרִים הַלָּלוּ הִתְחִילוּ לְשַׁנּוֹת אֶת זִימַרְתָּן. שְׁלֹשִׁים שְׁנִיּוֹת וְחַדַל, עַל זֶה גַּם הֵם דַּרְשׁוּ צָרְכוּ בְּסִירוּב לְהִתְקַע וּלְעַבֵּר מִינִים בְּחִינָם לְהִתְחַלֵּק, וּפִינִי, עַל פִּי דַּעְתָּהּ שֶׁל פִּירְחַת שָׁמְעָה שְׁקִיקַת הַחֲדָר.</p>

<p>הַיָּשָׁן נֶעֱלַם בְּנִשְׁמַע הַזִּימוּר וְהָאֲנָחוֹת, עַד שֶׁהִפְסִיק כְּשֶׁהַקּוֹלוֹת הֵחֵלּוּ לְדַרְבֵּן אוֹתוֹ בְּנִגּוּן אַחֵר. הוּא הִתְחִיל לְשַׁנּוֹת אֶת הַיְּלָדִים. פִּירְחַת נִגְּנָה מוּסִיקָה אֲרֻכָּה וּמַרְשִׁימָה, וְשֶׁר לָהּ מִלִּים עֲמוּקוֹת עַל כֵּף הָאוֹשֶׁר הַסּוּפָנִית, שֶׁהֵשַּׁעֲנוּ עָלֶיהָ תָּמִיד. בִּזְמַן שֶׁהָעִידָן נִשְׁמַע שָׁר עִם הַצִּפּוֹרִים, וּפִּינִי צִייְרָה שֶׁמֶשׁ שְׁקִיעָה יְפֵהפִיָּה שֶׁגָּרְמָה לְשֶׁלְוָוה. בַּסּוֹף, הָאִישׁ הַיָּשָׁן עָצַם אֶת עֵינָיו מְצַחֵק, וְנָעַלְמוּ חֶדֶר וְאַשְׁכּוּנָה.</p>

<p>כְּשֶׁפָּקְחוּ אֶת עֵינֵיהֶם, הָיּוּ הַיְּלָדִים שׁוּב בַּגִּנָּה שֶׁלָּהֶם. הַשֶּׁמֶשׁ חִיְּכָה עֲלֵיהֶם בְּחוּם, וְהַצִּפּוֹרִים שָׁרוּ שִׁיר מְיֻחָד, שִׁיר שֶׁכָּל אֶחָד מֵהֶם הֵבִין. לֹא יָדְעוּ מָה קָרָה בְּדִיּוּק בַּמַּעֲמָקִים, אֲבָל הֵם יָדְעוּ שֶׁעָשׂוּ מַשֶּׁהוּ טוֹב. כְּשֶׁהִתְחִילוּ הַיְּלָדִים לְחַיֵּךְ, אֶפְשָׁר הָיָה לִשְׁמֹעַ עַל פְּנֵי כָּל עִיר הַפְּלָאוֹת מַמָּשׁ אֶת הַזִּמּוּר שֶׁל הַתְּשׁוּבָה.</p>

<p>בְּאוֹתוֹ יוֹם הֵבִינוּ הַיְּלָדִים אֶת הַכֹּחַ הָאֲמִתִּי שֶׁלָּהֶם. לֹא רַק הַכִּשְׁרוֹנוֹת הַמְיֻחָדִים, אֶלָּא הַיַּחַד שֶׁלָּהֶם, הַיְּכֹלֶת לַעֲזֹר לְאֲחֵרִים, לִשְׁמֹעַ אֶת הַלֵּב וְלִבְנוֹת עוֹלָם שֶׁל שָׁלוֹם. מִשְׁקָם זֶה לָאוֹשֶׁר הֵבִיאָם לְדַרְכָּם.</p>

<p>מֵאוֹתוֹ הַיּוֹם, עִיר הַפְּלָאוֹת הָיְתָה מְפֻתָּחַת עֲדַיִן, גִּשְׁרוֹת בְּאֵיפָה נָכּוּ עַל נִפְלָאוֹת, בְּבִטָּחוֹן. יְלָדִים כֻּלָּם יָדְעוּ שֶׁאִם צָרִיךְ עֶזְרָה, תָּמִיד יֵשׁ מִי שֶׁיַּגִּישׁ אֶת יָדוֹ, תָּמִיד יֵשׁ אִמּוּנִים כִּי יֶשְׁנוּ הֶעָשְׁוָה שֶׁבָּהֶם. כְּשֶׁהַשֶּׁמֶשׁ הָיְתָה שׁוֹקַעַת בָּעֶרֶב, הֵם הָיוּ נִצְבָּעִים עַל הַגִּבְעָה וּשְׁמִיעִים צְחוֹק צָפוּף לְהַגָּנָה עֲלֵיהֶם, וְשָׁרוּ בְּיַחַד עַד שֶׁעָפַרָם נָח בְּאִסּוּף לָרֶדֶת לְמִטּוֹתֵיהֶם. הַיַּחַד שֶׁלָּהֶם, הָאַהֲבָה וְהָאִמּוּן הִצְמִיחוּ אֶת עִיר הַפְּלָאוֹת לִמְקוֹם מֻפְלָא, שָׁמֵינִים מְמַשְׁמָעוֹת.</p>

<p>בְּכָל עֶרֶב, לִפְנֵי שֶׁיָּשְׁנוּ, הַיְּלָדִים שָׁאֲלוּ: "מַה בָּרִיאוֹת תֵּלֵךְ?" וְהַתְּשׁוּבָה הָיְתָה תָּמִיד: "הָאוֹשֶׁר! הָאַהֲבָה! הָאֱמוּנָה! הַמֶּשֶׁךְ!" וְכָךְ שׁוֹמְרִים שֶׁל עִיר הַפְּלָאוֹת חָיוּ בְּאֹשֶׁר תָּמִיד.</p>
