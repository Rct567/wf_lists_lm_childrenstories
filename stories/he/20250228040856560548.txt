<p>"אַבָּא, מָה קָרָה לַיְּעָר שֶׁלָּנוּ? לָמָּה כֻּלָּם כָּל כָּךְ עֲצוּבִים?"</p>

<p>אָיָלוֹת הִתְיַשְׁבָה עַל הַסַּפְסֵל בְּגַן הַמֶּלֶךְ, מְעוֹרֶבֶת בְּמַחְשָׁבוֹת. הִיא זָכְרָה אֶת הַאֲגָדָה שֶׁאָמְרוּ עַל צְחוֹקָהּ שֶׁלָּהּ, שֶׁהוּא מְכַלֶּה אֶת הַשְּׁלִילוּת וּמֵבִיא אוֹשֶׁר וְשִׂמְחָה. אִם צְחוֹקָהּ הָיָה קָסוּם, אֲפִיּלוּ לְמַדַּי, אָז אוּלַי הִיא יְכוֹלָה לַעֲזֹר.</p>

<p>אָיָלוֹת עָמְדָה וְאָמְרָה בְּקוֹל רָם וְאֶלֶ"ף:</p>

<p>אָיָלוֹת יָצְאָה לַיְּעָר הָעָצוּב. הָעֵצִים כְּבָר לֹא הָיוּ יְרוּקִים, הָאֲוִיר הָיָה כָּבֵד וְדָמוּם, וְלֹא הָיוּ צִפּוֹרִים שֶׁלָּהּ לְשִׁיר. הָעֵצִים עִמְמוּ בְּשֶׁקֶט, וְהַחַיּוֹת סָגְרוּ אֶת עֵינֵיהֶם מֵעֲצִיבוּת.</p>

<p>אָיָלוֹת הִתְחִילָה לְהִתְהַלֵּךְ בַּיְּעָר, וְהִיא הִתְחִילָה לָשִׁיר. הִיא שָׁרָה עַל הַיָּמִים הַיָּפִים שֶׁעָבְרוּ עָלֶיהָ, עַל הַפְּרָחִים הַצְּבוּעִים וְהַשֶּׁמֶשׁ הַזּוֹרַחַת. אֲבָל, הַיֶּלֶד הֵבִינָה, שִׁירָה לֹא הָיָה מְסֻגָּל לְהַשְׁפִּיעַ עַל הַמַּצָּב.</p>

<p>לְפֶתַע, הִיא עָצְרָה. הִיא נִזְכְּרָה בְּהַבְטָחָתָהּ. הִיא נִשְׁמְעָה יוֹתֵר, וְאֲזַי – הִיא הִתְחִילָה לִצְחֹק. לֹא צְחוֹק קָטָן, אֶלָּא צְחוֹק יָפֶה, עָצוּם וּמַדְבִּיק. הִיא צָחֲקָה עַל הַשְּׁמַיִם הַכְּחוּלִים, עַל הַפְּרָחִים הָרַכִּים, עַל הַפִּיּוּטִים הָרַכִּים שֶׁבַּכְּנָפַיִם שֶׁל הַצִּפּוֹרִים!</p>

<p>בְּהַתְחָלָה, הַצְּחוֹק שֶׁלָּהּ נִשְׁמַע מוּזָר בַּשֶּׁקֶט הַכָּבֵד שֶׁל הַיְּעָר. אֲבָל, כְּכָל שֶׁאָיָלוֹת הִמְשִׁיכָה לִצְחוֹק, קָרָה דָּבָר מְאוֹד מְעַנְיֵן. הָעֵצִים הִתְחִילוּ לְהִתְיַיֵּשׁ וְהַפְּרָחִים הִתְחִילוּ לְהֵיעוֹר. הָאֹכֶל בָּאָרֶץ שָׁב לִהְיוֹת יָרוֹק, וְהַשָּׁמַיִם הִתְחִילוּ לִזְרֹחַ מֵחָדָשׁ.</p>

<p>לְאָט לְאָט, כָּל הַחַיּוֹת הִתְחִילוּ לְהַקְשִׁיב לְצְחוֹקָהּ שֶׁלָּהּ. פַּרְפָּרִים עָפוּ מִסְּבִיבָהּ, וְצִפּוֹרִים הִתְחִילוּ לָשִׁיר שׁוּב. הָעֵינַיִם שֶׁל הַחַיּוֹת הִתְחִילוּ לִנְצוֹץ מֵאוֹשֶׁר, וְהֵן הִתְחִילוּ לִצְחֹק גַּם כֵּן.</p>

<p>הַקִּלְלָה שֶׁל מִירָב הִתְחִילָה לְהִתְפַּיֵּר. הָעֶצֶב נָמֵס וּרָחַק, וְהָעֲצוּבוּת עָזְבָה אֶת הַיְּעָר. כָּל רַחֲשֵׁי הַצְּחוֹק שֶׁל אָיָלוֹת הִתְחִילוּ לְהֵיעוֹר מֵחָדָשׁ אֶת צוֹחֵק פִּיּוּט.</p>

<p>מִירָב, הַמְּכַשֵּׁפָה הָרָעָה, הִרְגִּישָׁה בְּכָךְ. הִיא הִכִּירָה אֶת הַקֶּסֶם שֶׁל צְחוֹקָהּ שֶׁל אָיָלוֹת, וְהִיא מִיָּד יָדְעָה שֶׁהַבַּיְּעָר שֶׁלָּהּ בְּסַכָּנָה. הִיא רָצָה לְעַכֵּז תְּחִלָּה, לִפְנֵי שֶׁאָיָלוֹת תַּשְׁפִּיעַ עַל הַכֹּל הַרְבֵּה יוֹתֵר.</p>

<p>הִיא טָסָה עַל מַטְאֲטֵא שֶׁלָּהּ לַיְּעָר וְנִסְּתָה לְעָצְרָהּ צְחוֹקָהּ שֶׁל אָיָלוֹת. הִיא יָצְאָה מִתּוֹךְ הַצְּלָלִים וְצָרְחָה בְּקוֹל תַּכְלוּלִי:</p>

<p>"תִּשְׁתֹּקִי! תַּפְסִיקִי לִצְחֹק!"</p>

<p>אֲבָל אָיָלוֹת לֹא פָּחֲדָה. הִיא הִמְשִׁיכָה לִצְחֹק, וּצְחוֹקָהּ הָפַךְ לְרַם יוֹתֵר מֵאֵשׁ. הִיא צָחֲקָה עַל לֵב הַמְּכַשֵּׁפָה הָרָעָה, עַל הַמַּטְאֲטֵא שֶׁלָּהּ וְעַל הַכֹּל הַרָע שֶׁלָּהּ.</p>

<p>הַקֶּסֶם שֶׁל הַצְּחוֹק שֶׁל אָיָלוֹת הָיָה כָּל כָּךְ חָזָק, עַד כִּי הַקִּלְלָה שֶׁל מִירָב נִשְׁבְּרָה לְגַמְרֵי. הַמְּכַשֵּׁפָה הָרָעָה הִתְחִילָה לְהִתְחַיֵּךְ, וּבְסוֹף הִצְטָרְפָה לִצְחוֹק שֶׁל אָיָלוֹת. מִיָּד, הַמְּכַשֵּׁפָה הִתְחִילָה לְהִשְׁתַּנּוֹת. לִבָּהּ הַקָּר נָמֵס, וְהִיא הָפְכָה לְאִשָּׁה טוֹבָה וּמְאוּשֶּׁרֶת.</p>

<p>הַיְּעָר חָזַר לִהְיוֹת לְיָפִיתוֹ הָרִאשׁוֹנָה. הָעֵצִים הָיוּ יְרוּקִים, הַפְּרָחִים פָּרְחוּ וְהַחַיּוֹת שִׂמְּחוּ. כֻּלָּם הֶעֱרִיצוּ אֶת אָיָלוֹת עַל הָאוֹמֶץ וְעַל שִׂמְחַת הַחַיִּים שֶׁלָּהּ. הַחַיּוֹת הִתְחִילוּ לָשִׁיר יַחַד עִם הַנְּסִיכָה, וְעִם כָּל צְחוֹק, הַיְּעָר הָפַךְ לְמָקוֹם יוֹתֵר טוֹב.</p>

<p>מְאוּתָם הַיּוֹם וְהָלְאָה, הַנְּסִיכָה אָיָלוֹת חָיְתָה בְּאוֹשֶׁר בַּמַּמְלֶכֶת. הִיא לֹא שִׁכְּחָה לְעוֹלָם אֶת הַשִּׂמְחָה שֶׁהִשְׁתַּמְּשָׁה בָּהּ, וְתָמִיד שִׁמְחָה עִם כָּל מִי שֶׁהִסְתַּכְּלוּ עָלֶיהָ. כָּל דַּרְכָּה, הַגּוֹרָל שֶׁלָּהּ הָיָה לְהַצִּיל יְעָרוֹת וּלְעַזֵּר לַאֲחֵרִים לְהִתְמַלֵּא בְּאוֹשֶׁר עַל יְדֵי הַצְּחוֹק הַקָּסוּם שֶׁלָּהּ.</p>

<p>וְאִם תַּקְשִׁיבוּ הֵיטֵב בַּיָּעָר, אֲפִיּלוּ הַיּוֹם, אַתֶּם תּוּכְלוּ לִשְׁמֹעַ הֲדִי שֶׁל צְחוֹק קָטָן, שֶׁאֵינֶנּוּ שֶׁל אָיָלוֹת כַּיּוֹם, אֲבָל אֲפִישׁוּת תַּזְכִּיר לָכֶם אֶת הַיֹּפִי וְאֶת הַשִּׂמְחָה, שֶׁהֵם כָּל כָּךְ חֲשׁוּבִים.</p>
