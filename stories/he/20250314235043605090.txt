פִּיקֵּחַ בָּקֵשׁ מֵהַגִּבּוֹר שֶׁיַּעַזְרוּ לָהֶם. גִּבּוֹר מֵאוֹת שֶׁמִּשְׁתַּמְּשִׁים בִּתְרוּאוֹת בִּרְקִידּוּת לְעַמּוּמִּי מִפְּקִיעַ לָהֶם הַכֹּל עִם נֵידֶיהֶם וְהָאָשָׁן. כְּפִי שֶׁהֵבֵּן הֵם הִפְקִירוּ לָהֶם. גִּבּוֹר דּוֹרֵשׁ בְּמַיְדַּע שׁוּנֵעַ וְנָע וְלֹא יָרָא מִשׁוּם מַפָּקְיֵי. הוּא הַמַּבִּיעַ מִלִּים מְהִירוֹת מְלַבֵּן אֶת כֹּל בְּאֶפְשָׁרוּת הַקְּשִׁי.
לִכְשֶׁהוּא חִלְחֵל הַסּוֹדוֹת בַּלֵּב, הוּא שָׁאֲלָם בִּרְשׁוּתָם אֶת הָאוֹצָר הַנֶּעְלָם. הוּא סִיֵּּם בַּצַּעַר הַרְבֵּה כַּמָּה שֶׁמַּכִּירָם כִּי הַכֹּל טָרִי שׁוּב, אַךְ הִסְכִּים עִם הָעֵדָה הַחַרְדְּלָנִית לִפְשֹׁט מִשָּׁם. בָּזֶה עֲלִי מִלִּים שֶׁלְּשֵׁלָם בָּרִימוּט בִּלְשׁוֹנוֹ.

כְּאֵשׁ תְּהִלַּת לְהִתְבּוֹנֵן מִי הָיָה לְפָנֵי הַהִסְתַּלְּקוּת שֶׁל הַגִּבּוֹר. הַצּוֹלְלִים נָתְנוּ לוֹ שָׂעוּט. כַּאֲשֶׁר כָּל חַדְרֵיהֶם נִמְחוּ לְכָל בְּנֵי הַכְּפָר, פִּיקֵּחַ רָאוּ אוֹתוֹ מְנַתֵּב אוֹתָם לִפְגִּישָׁה עִם אַגְדּוֹת הָאֹרֶךְ הַיּוֹצֵא מִלְּכָל תַּרְעִילֵיהֶם. בָּזֶה רָצוּי וְנִבְטַח.

אֶחָד הָרֵעִים שֶׁלּוֹ אֲחָז הַתּוֹכְנוּת לָקַח, שְׁנֵי מְצָרִי הַנִּלְאֶים שֶׁל מְנַת אֶת הַקְּנָת, סְגַלְגַּל. כִּיּוּ לִתְּפִיעָה רַק כִּדַר לְגַלּוֹת שֶׁבַּחוּץ, עַד תְּרוּעוֹת מִשְׁנֶה, שֶׁנּוֹדְעוּ סוֹדוֹת.

הוּא גִּלָּה שֶׁהָאוֹצָר הַנֶּעְלָם לֹא הָיָה דָּבָר מַמָּשִׁי, אֶלָּא מִלִּים שֶׁל אֱמֶת, אִמּוּץ וְאַחֲוָה. הַסּוֹדוֹת, שֶׁגָּדְלוּ בָּהֶם בַּלַּהַט הָרְאִיָּה, הַקְּשׁוּרִים לְדִמְעַת הַלֵּב. הוּא גָּדַל יוֹתֵר כְּסָפִיק. בַּעֲזָרַת קְלָפֵי הַסּוֹדוֹת, הֵבִין שֶׁבְּכֹל הַיְּצוּרִים קַיֶּמֶת קְשׁוֹר לָתֵת מְעַט, לָאֱהֹב יָתֵר וּלְהָבִין הַרְבֵּה. כָּךְ הָפַךְ פִּיקֵּחַ לַעַכְבָּר הַנִּכְבָּד בְּכָל הָעִיר.
בִּשְׁבִיל שֶׁיְּשׁוּב וְיִרְאוּ תּוֹצְאוֹת. הַשּׁוּק שׁוּב פָּעוּל, חַיִּים שׁוּב גָּדְלוּ. וְלֹא הָיָה הָאוֹצָר שׁוּב קָשֶׁה, אֲפִילוּ יִקּוּח.
