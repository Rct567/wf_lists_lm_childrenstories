בִּרְכוּ לָכֶם, יַלְדֵי יְרוּשָׁלַיִם וּבָנֵי צִיּוֹן! אֲסַפֵּר לָכֶם עַל סִפּוּר שֶׁאֵרַע בְּיָם שֶׁל קֶסֶם, יָם שֶׁמְּתַיְּגוֹת בּוֹ אוֹת הַפְּלֵיאָה: "יָם פְּלֵיאָה פַנְטַסְטִי". וְהַסִפּוּר הוּא עַל דָּג יָפֶה, יָפֶה עַד מְאֹד, וְעַל דְּבַשׁ מְתוּק, שֶׁבְּדֶרֶךְ פִּלְאִית הִתְחַבֵּר אֶל חַיָּיו שֶׁל הַדָּג.

בְּיָם פְּלֵיאָה פַּנְטַסְטִי, הַמַּיִם כָּחֹלִים כִּשְׁמֵי הַשַּׁחַר, וּבְשֶׁקֶט מְחַיְּכִים אֶל כָּל אוֹר שֶׁבָּא לְבַקְּרָם. עַל קַרְקַע הַיָּם, מְרַקְּדִים אַלְמוֹגִים בְּכָל צֶבַע שֶׁתִּרְצוּ – אֲדֻמִּים כִּלְשׁוֹן אֵשׁ, צְהֻבִּים כְּשֶׂמֶשׁ בַּצָּהֳרַיִם, כְּחֻלִּים כְּאֶבֶן הַיָּם, וְיְרֻקִּים כִּדְשֵׁא שָׂדוֹת מִשְׁתּוֹקְקִים לִגְדֹּל. וּבֵין כָּל הַצְּבָעִים הָאֵלּוּ, שָׂחָה לוֹ דָּג – דָּג יָפֶה עַד אֵין סוֹף.

שְׁמוֹ שֶׁל הַדָּג הַזֶּה הָיָה אָרִי. אָרִי הָיָה שָׂחָה לוֹ בְּהִתְלַהֲבוּת בַּיָּם, פָּסִים שֶׁל זָהָב עַל גּוּפוֹ הַכָּחֹל הָעָמֹק, וְעֵינָיו שְׁנַיִם שֶׁל חָכְמָה וְסַקְרָנוּת. הוּא אָהַב לִבְדּוֹק אֶת הַכֹּל – אֶת מִי שֶׁעוֹבֵר לְיָדוֹ, אֶת הַצְּמָחִים הַנִּבְטוּ, וְאֶת מִינֵי הַדָּגִים הָאֲחֵרִים שֶׁמְּחַיְּכִים מִכָּל מִין וָמִין. הוּא הָיָה יְדִידוּתִי וְטוֹב לֵב, וְכָל תּוֹשָׁב שֶׁל הַיָּם אָהַב אוֹתוֹ.

בְּאֶחָד מֵהַיָּמִים, שֶׁחָה אָרִי לְיַד שְׂפַת הַיָּם. הוּא רָאָה בֵּית עֵץ קָטָן, כָּזֶה שֶׁאוֹהֲבִים לְבָנוֹת בְּלֹא כַּוְנַת מַטָּרָה, עִם גַּג מִכְּדֵי צַדָּפוֹת מַתִּיקוֹת וְתוֹסִיסָנִית. עַל הַשְּׁעָר שֶׁל הַבַּיִת הָיְתָה שְׁלֶטֶת: "בֵּית הַדְּבַשׁ שֶׁל אֵלִי". אָרִי הֵצִיץ בְּסַקְרָנוּת וְרָאָה דְּבַשׁ זָהוּב וְמָלֵא שֶׁבֶל עַל שֻׁלְחָן מִשְׁמֶשׁ הַמֻּנָּח בְּתּוֹך הַבַּיִת. הַדְּבַשׁ הַזֶּה הֵרִיחַ בְּצֶדֶק מִתְּקוּף וּבְכֹחַ בְּשֶׁפַע.

שְׁמוֹ שֶׁל הַדְּבַשׁ הַזֶּה הָיָה אֵלִי, וְהוּא הָיָה דְּבַשׁ מְיֻחָד בִּמְיֻחָד. הוּא לֹא נֶאֱסַף מִפִּיו שֶׁל הַדְּבוֹר, כִּי אִם נֶאֱפֶה עַל יְדֵי הַיָּם עַצְמוֹ. הַיָּם, בְּאֶמְצָעוּת כּוֹחוֹתָיו הַקְּסוּמִים, הִכְנִיס אֶל הַדְּבַשׁ טְעָמִים וְנֵיחוֹת שֶׁל כָּל הַיְצוּרִים שֶׁחַיִּים בּוֹ – שֶׁל הַדָּגִים, שֶׁל הָאַלְמוֹגִים, שֶׁל כָּל הַצְּמָחִים וְכָל הַיְצוּרִים. כַּאֲשֶׁר טוֹעֲמִים מִמֶּנּוּ, אִם כָּךְ, מַרְגִּישִׁים שְׁאִיפַת עָצְמָה לִכְדֵי הִתְאֲחֵד עִם הַיָּם עַצְמוֹ.

לְאָרִי הָיָה מִישָׁהוּ בִּלְתִּי נִרְאֶה בַּחַיִּים – תַּאֲוַת הַרְפָּיָה וּלְהִשְׁתּוֹבֵב. הוּא נִכְנַס בִּזְהִירוּת אֶל בֵּיתוֹ שֶׁל אֵלִי הַדְּבַשׁ. בַּיִת דְּבַשׁ רִצְפָתוֹ זֶה הָיָה זָהוּב, וּמִלְבַד הַדְּבַשׁ הָיְתָה שָׁם אִשָּׁה שׁוּבִיל יְרַקְרַקִּית, עוֹמֶדֶת מֵאֲחוֹרֵי הַשֻּׁלְחָן וְתוֹדִיעַ אֶת שְׁמָהּ – "אֲנִי הִיא מִי שֶׁאָחֵר לֹא חִכָּה" אָמְרָה. אָרִי הָיָה קְצָת מֻפְתָּע, אֲבָל כְּשֶׁרָאָה שֶׁהִיא מְחַיֶּכֶת וְשַׁאֲנָנוּת, הוּא הִרְגִּישׁ נוֹחַ יוֹתֵר.

"שָׁלוֹם," אָמַר אָרִי בִּסְקרָנוּת. "אֲנִי אָרִי. רָאִיתִי אֶת הַבַּיִת שֶׁלְּךָ, וְרָצִיתִי לִרְאוֹת אֶת הַדְּבַשׁ".

"הַבַּיִת שֶׁלִּי?" שָׁאֲלָה הָאִשָּׁה הַשּׁוּבִיל. "זֶה לֹא הַבַּיִת שֶׁלִּי. זֶה בֵּיתוֹ שֶׁל אֵלִי. אֲנִי כְּלָל לֹא מִתּוֹך מִשְׁפַּחְתּוֹ, אֲנִי הָעֶזְרָה הַקְּסוּמָה הַזּוֹ, וְרַק אַתָּה יוֹדֵעַ שֶׁאֲנִי תָּמִיד קוֹרֵאת אֶת לֵב הָאוֹהֲבִים".

"וְאֵיךְ אוּכַל לָגֶשֶׁת לִדְבַשׁ הַזֶּה?" שָׁאַל אָרִי.

"פָּשׁוּט," אָמְרָה הָאִשָּׁה הַשּׁוּבִיל. "צָרִיךְ רַק לָטְעֹם".

אֲרִי הִתְקָרֵב אֶל הַשֻּׁלְחָן. הַדְּבַשׁ הַזּוּרְנִי הֵאִיר בַּאֲשֶׁר הַשֶּׁמֶשׁ לִדְבַשׁ קִידֵם, וְרֵיחוֹ הֵבִיא אֶל שִׁכְּרוּתוֹ שֶׁל כְּשֶׁיּוֹדֵעַ שֶׁמְּכַוֵּן טוֹב. הוּא הַכְנִיס גַּם אֶת פִּיו, וְטָעַם.

כַּרְצוֹנִי הִתְפּוֹצְצוּ בְּפִיו צִנּוֹת שֶׁל טַעֲמִים שֶׁל דְּבַשׁ - שֶׁל דָּגִים וְאֶלְמוֹגִים, שֶׁל יָרָק וְצֶבַע. בְּרֶגַע, הִרְגִּישׁ אָרִי אֶת לִבּוֹ פּוֹעֵם בִּתְנוּעָה, הַרְגָּשָׁה שֶׁהוּא חוֹפֵשׂ בַּיָּם אֶל פְּלָאֵי טַעֲמֵי הַדְּבַשׁ. הוּא הִתְחִיל לִרְאוֹת אֶת כָּל הַיְּצוּרִים שֶׁבַּיָּם בְּעֵינַיִם חֲדָשׁוֹת. הוּא הֵבִין אֶת שְׁפַת הָאַלְמוֹגִים הָרוֹקְדִים, הוּא הִקְשִׁיב לְשִׁירָם שֶׁל הַדָּגִים.

אֵלִי הַדְּבַשׁ, הוּא רָאָה, כִּי קוֹשֵׁר אוֹתוֹ אֶל הַיָּם כְּפֶתֶל שֶׁל אַהֲבָה וְהֲבָנָה. אֶלָּא שֶׁיֵּשׁ כְּבָר יַחַס חָדָשׁ בְּחַיָּיו – הוּא רוֹאֶה אֶת כָּל דָּג מְעִידָן שֶׁהוּא יוֹדֵעַ מִמֶּנּוּ. מִשְּׁתֵּי עֵינָיו שֶׁל אָרִי נָפַל נִיצוֹץ שֶׁל אוֹר.

"אֲנִי מַרְגִּישׁ…" אָמַר אָרִי בְּקוֹל שֶׁהוּא שִׁמְעָה לַעֲצִימַת שֶׁקֶט, "אֲנִי מַרְגִּישׁ כִּי הַדְּבַשׁ הַזֶּה קוֹשֵׁר אוֹתִי אֶל הַיָּם בִּכְלָל אַחֵר! כְּאִלּוּ אֲנִי חֵלֶק מִכֹּל".

"וְזֶה הַדְּבָשׁ שֶׁלּוֹ שֶׁל אֵלִי," אָמְרָה הָאִשָּׁה הַשּׁוּבִיל בְּחִיּוּךְ חָכָם. "הוּא מְשַׁנֶּה אֶת הַכֹּל. הוּא מַרְאֶה לְכָל אֶחָד מִי הוּא בֶּאֱמֶת, וּמַה שֶׁהוּא יָכוֹל לִהְיוֹת".

וְשֶׁבְּרֶגַע הַזֶּה הִרְגִּישׁ אָרִי רְצוֹנוֹת שֶׁל רַק רְאִיָה.

אָרִי חָשַׁב. הִתְקָרֵב אֵלָיו הַיּוֹם הָעֵרָנִי שֶׁל הָאַהֲבָה. הַיָּם נִתְגַּלָּה לוֹ עַתָּה כְּגַן עֵדֶן – כָּל יְצוּר הַנּוֹשֵׁם בּוֹ חֲשׁוּב וּמֻעֲרָךְ. הוּא רָצָה לִדְאֹג לַיָּם, לִשְׁמֹר עָלָיו. הוּא יָצָא מִבֵּיתוֹ שֶׁל אֵלִי, לֵב עָרֵב וְעֵינַיִם תְּבוּנוֹת.

מֵאוֹתוֹ הַיּוֹם, הִתְחִיל אָרִי לְהִתְעַקֵּשׁ עַל מְעִידָנוּת הַיָּם. הוּא דִּבֵּר עִם כָּל הַיְּצוּרִים, הוּא שָׁאַל שְׁאֵלוֹת עַל הַסְּבִיבָה וּלְמַד, לִמֵּד. הוּא עָזַר לְכָל מִי שֶׁנִּזְקַק לְעֶזְרָה, לְכָל מִי שֶׁהָיָה בְּצָרָה. הוּא סִיֵּעַ לְסַלֵּק אֶת הַזֻּבְּרִיּוֹת מֵהַיָּם. הוּא לִמֵּד אֶת הַדָּגִים הַצְּעִירִים לְשׁוּט וּלְשַׁמֵּר אֶת הַיָּם. הַתּוֹשָׁבִים שֶׁל יָם פְּלֵיאָה פַּנְטַסְטִי הוֹדִיעוּ כִּי הֵם אוֹהֲבִים אוֹתוֹ.

בְּכָל יוֹם וְיוֹם הִשְׁתַּנָּה הַיָּם לְטוֹבָה. שִׂיא הַדָּגִים פַּרְחוּ בִּשְׁאֵלַת אַהֲבָה. הָאַלְמוֹגִים שׁוּב חִיְּכוּ בְּמָשָׁר. הַיָּם כְּאִלּוּ זָהָה כְּלוֹ אֶת אָרִי, הָרָגִישׁ שֶׁכָּל פַּעַם שֶׁהוּא מַבִּיט בּוֹ הוּא הוֹפֵךְ יוֹתֵר יָפֶה, מַרְשִׁים.

וְכָךְ הָפַךְ אָרִי מִדָּג יָפֶה לְשׁוֹמֵר הַיָּם. מִי שֶׁטָּעַם אֶת הַדְּבַשׁ הַפִּלְאִי שֶׁל אֵלִי, נָתַן לִבּוֹ לְשֶׁמֶשׁ וְעֵינָיו לַיָּם. הַדְּבַשׁ הָפַךְ אוֹתוֹ לְיֶלֶד אֶחָד בָּאֲפִיס.

בְּסוֹפוֹ שֶׁל דָּבָר, שָׂחָה אָרִי בְּעוֹד תְּרֵימִיל בְּמֵי יָם פְּלֵיאָה פַּנְטַסְטִי, מַאֲשִׁים וּבָטוּחַ בְּחַיָּיו. הוּא הֵבִין שֶׁהַיֹּפִי אֵינוֹ רַק בַּמַּרְאֶה, אֶלָּא בְּטִפּוּחַ, בְּאַהֲבָה, וּבְיָכֹלֶת לְהָרִיחַ אֶת כָּל אוֹתָם שֶׁתַּכְנִיסוּ לְיָם שֶׁל שַׁלְוָה.
וּלְכָל הַנּוֹדְדִים בָּאֲפִיס, שֶׁיְּהֵא לָכֶם סִפּוּר מְעוֹרֵר.
