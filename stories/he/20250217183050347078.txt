בְּכִּפְרִי קְטַן וּמַרְתֵּק עִם בָּתִּים צְבָעוּנִים כְּמוֹ מַרְיָה שֶׁל גְּלִידָה, שָׁכְנָה יְלָדָה שֶׁשְּׁמָהּ אֲדָמָה. לַאֲדָמָה הָיוּ עֵינַיִם כָּחוּלוֹת כְּמוֹ שָׁמַיִם שֶׁל יְוֹם קַיִץ וּשְׂעָר שָׁחוּם כְּמוֹ עֲפַר הָאֲדָמָה הַפּוֹרִיָּה. הִיא אָהֲבָה לִטֵּל בְּגָנָּה שֶׁל בֵּיתָהּ, לְהַשְׁקוֹת אֶת הַפְּרָחִים וְלָשֵׂאת שִׂיחוֹת עִם הַצִּפּוֹרִים הַמְּצַיְּצוֹת.

יוֹם אֶחָד, כַּאֲשֶׁר אֲדָמָה הָיְתָה מְשַׁחֶקֶת בֵּין הַכַּלְכָּלָנוּת, הִיא שָׁמְעָה קוֹל עָדִין כְּמוֹ צִלְצוּל פעמונים. הִיא הִתְקָרְבָה אֶל הַמָּקוֹם שֶׁמִּמֶּנּוּ בָּא הַקּוֹל וְרָאֲתָה דֶּרֶךְ מַעֲבָר קָטָן בֵּין לִבְנֵי הַגְּדֵרָה, גַּן שֶׁלֹּא רָאֲתָה כָּמוֹהוּ מֵעוֹלָם.

הַגַּן הָיָה מָלֵא בִּפְרָחִים שֶׁבְּצִבְעֵיהֶם נֶעֶלְמוּ כָּל הַצְּבָעִים שֶׁהִכִּירָה. וְעַל כָּל פֶּרַח, הָיוּ מְרַחְפִים פַּרְפָּרִים, מְעוּצָּבִים בִּצְבָעִים עַזִּים וְשׁוֹנִים, כְּאִלּוּ צִיֵּר אוֹתָם הָאָמָּן הֲכִי טוֹב בָּעוֹלָם.

"וְלָמָּה כָּל כָּךְ יָפֶה?" שְׁאֲלָה אֲדָמָה.

אֲדָמָה הָיְתָה מְאוּשָׁרָה לִשְׁמֹעַ אֶת זֹאת. הִיא הִתְחִילָה לְהִתְבּוֹנֵן עַל הַפַּרְפָּרִים הָאֲחֵרִים. הָיוּ שָׁם פַּרְפָּרִים אֲדֻמִּים כְּמוֹ דּוֹדָאִים, כְּחֻלִּים כְּמוֹ הַיָּם, צְהֻבִּים כְּמוֹ הַשֶּׁמֶשׁ, וְיְרֻקִּים כְּמוֹ הֶעָלִים. כָּל אֶחָד הָיָה מְיֻחָד בְּמִינוֹ.

הַפַּרְפַּר הַסְּגֻלָּה הִצִּיעַ לַאֲדָמָה לְהִצְטָרֵף אֲלֵיהֶם לְמַסָּע בְּתוֹךְ הַגַּן. הִיא שִׂמְּחָה לְהַסְכִּים. הֵם טָסוּ בֵּין הַפְּרָחִים, מְרַחְפִים עַל פְּנֵי הַטַּל שֶׁעַל הֶעָלִים. הֵם שָׂחוּ עַל גַּבֵּי בְּרֵכָה שֶׁל מַיִם צְלוּלִים כְּמוֹ בַּדִּיק, וְטִיְּלוּ בְּמִנְהָרוֹת שֶׁל צְמָחִים.

בְּמַהֲלַךְ הַמַּסָּע, הִיא שָׁמְעָה שְׁמוּעוֹת שֶׁעַלָּם בַּכְּפָר הָיְתָה תְּחוּשָׁה שֶׁל עַצְבוּת וְדִכָּאוֹן. הָעֵצִים הִתְחִילוּ לִנְשֹׁר אֶת הֶעָלִים שֶׁלָּהֶם, הַפְּרָחִים הִתְחִילוּ לִנְבֹּל, וְהַצִּפּוֹרִים הִפְסִיקוּ לְשִׁיר.

"לְמָה זֶה קוֹרֶה?" שְׁאֲלָה אֲדָמָה בְּצַעַר.

"זֶה בְּגלַל שֶׁהָאֲנָשִׁים שָׁכְחוּ אֶת הַיֹּפִי שֶׁבַּחַיִּים," עָנָה הַפַּרְפַּר הַסְּגֻלָּה. "הֵם שָׁכְחוּ לִצְחֹק, לְחַיֵּךְ, וְלִשְׁמֹחַ בִּדְבָרִים קְטַנִּים."

אֲדָמָה הִרְגִּישָׁה שֶׁהִיא צְרִיכָה לַעֲשׂוֹת מַשֶּׁהוּ. "אֲנִי אֲנַסֶּה לַעֲזֹר!" הִיא אָמְרָה בְּהַחְלָטָה.

הִיא חָזְרָה לַכְּפָר. הִיא הָלְכָה לְכָל בַּיִת וְחִיְּכָה לְכֻלָּם. הִיא שָׂחֲקָה עִם הַיְּלָדִים בַּכְּפָר, שָׁרָה שִׁירִים, וְסִפְּרָה סִפּוּרִים שְׂמֵחִים. הִיא הִזְכִּירָה לָאֲנָשִׁים לְהִשְׁתַּעְשֵׁעַ בַּדְּבָרִים הַקְּטַנִּים, לְהַבִּיט בַּשָּׁמַיִם, לָשֵׂאת שִׂיחוֹת עִם הַצִּפּוֹרִים, וְלִשְׁמֹחַ בַּשֶּׁמֶשׁ.

כַּאֲשֶׁר אֲדָמָה הָלְכָה, הִיא שָׁמְעָה צִלְצוּל שֶׁל פַּעֲמוֹן קָטָן. הִיא הִתְחִילָה לִרְאוֹת עֵצִים שֶׁמַּחְזִירִים לְהִשְׁתַּקֵּף בְּצִבְעֵיהֶם הַפְּלָאִיִים, וּפְרָחִים שֶׁפּוֹרְחִים בִּצְבָעִים עַזִּים. הַצִּפּוֹרִים שָׁבוּ לְשִׁיר שִׁירִים שְׂמֵחִים.

כַּאֲשֶׁר אֲדָמָה חָזְרָה לְגַן הָאָבִיב הַפְּלָאִי, הִיא רָאֲתָה שֶׁהַפַּרְפָּרִים הִתְחִילוּ לִהְיוֹת יוֹתֵר צִבְעוֹנִיִּים מִפַּעַם. הַגַּן הָפַךְ לִיפֶה מִתָּמִיד. הַפַּרְפַּר הַסְּגֻלָּה חִיֵּךְ אֵלֶיהָ וְאָמַר: "אַתְּ רוֹאָה, אֲדָמָה? כְּשֶׁאָדָם מִתְמַלֵּא שִׂמְחָה, כָּל הָעוֹלָם מִתְמַלֵּא שִׂמְחָה יַחַד אִתּוֹ."

אֲדָמָה הִרְגִּישָׁה שֶׁהִיא עָשְׂתָה דָּבָר גָּדוֹל. הִיא הֵבִינָה שֶׁכָּל אֶחָד יָכוֹל לִשְׁנוֹת אֶת הָעוֹלָם, עַל יְדֵי שִׂמְחָה וְטִיּוּב לִבָּם. הִיא הִמְשִׁיכָה לְבַקֵּר בְּגַן הָאָבִיב הַפְּלָאִי כָּל יוֹם, לְטַיֵּל עִם הַפַּרְפָּרִים, וּלְזַכּוֹת אֶת כָּל הַיְּלָדִים שֶׁל הַכְּפָר בְּשִׁמְחָה.

וְכָךְ, הַפַּרְפָּרִים הַצְּבָעוּנִים וְגַן הָאָבִיב הַפְּלָאִי הָיוּ מְקוֹר שֶׁל אוֹשֶׁר וְיֹפִי לַכְּפָר הַקָּטָן, וַאֲדָמָה, הַיַּלְדָּה עִם הָעֵינַיִם הַכָּחוּלוֹת וְהַשֵּׂעָר הַשָּׁחוּם, שָׁמְרָה עַל הַיֹּפִי הַזֶּה, שֶׁיִּתְפַּשֵּׁט בְּכָל הָעוֹלָם. כִּי הַשִּׂמְחָה, כְּמוֹ הַצְּבָעִים, הִיא כְּמוֹ מַגְעִית, מִתְפַּשֶּׁטֶת וּמַשְׁפִּיעָה עַל הַכֹּל. סוֹף.
