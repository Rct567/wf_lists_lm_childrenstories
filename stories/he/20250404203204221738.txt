בֹּקֶר טוֹב!

אֵלִיָּהוּ: אוֹפְּס, אַתָּה שָׁם! הַיּוֹם שַׁבָּת, לָלֶכֶת לַבְּרֵכָה!

אֵלִיָּהוּ: הַמִּצְנֶפֶת? הוּא! אֲנִי תָּמִיד שׁוֹכֵחַ אוֹתָהּ! תּוֹדֶה שֶׁהִזְכַּרְתָּ לִי. מִי צָרִיךְ לִדְאֹג לִדְבָרִים אֵלֶּה.

יְהוֹנָתָן: בְּסֵדֶר, הָעִיקָר שֶׁתַּגִּיעַ בְּשָׁלוֹם. אָז, נֵלֵךְ עַכְשָׁו?

אֵלִיָּהוּ: אֲנִי מוּכָן! חַכָּה, רֶגַע! אֲנִי צָרִיךְ לִלְבֹּשׁ אֶת תִּכְנֵס הַיָּם שֶׁלִּי, וְלָקַחַת אֶת מִכְנוֹת הַיָּם שֶׁלָּךָ.

יְהוֹנָתָן: מִישֶׁהֶה! אֵיזֶה הַמְּהָרוּת! הַיּוֹם נִשְׁחֶה הַרְבֵּה, אֲנִי חוֹשֵׁב. מִי מְנַצֵּחַ בַּתַּחֲרוּת הַשְּׂחִיָּה הַזֹּאת הַפַּעַם?

יְהוֹנָתָן: הַצֵּא, אִם כֵּן! לָמָה לְחַכּוֹת? שָׁלִישׁ הַבְּרֵכָה! מְצַפֶּה לָנוּ יוֹם מֻפְלָא!

(בַּבְּרֵכָה)

אֵלִיָּהוּ: הוּוּ, מַיִם צְלוּלִים! מְשַׁגֵּעַ! וְרֹאִים אֶת הַכּל!

יְהוֹנָתָן: אָכֵן! יָפֶה לְהִסְתַּכֵּל! תִּרְאֶה אֶת הָאֲנָשִׁים הָאַחֲרִים! כְּמָה כֵּף! וּתְבַקֵּשׁ מֵאֲנָשִׁים לִקְפוֹץ בְּעֵזְרָתְךָ!

אֵלִיָּהוּ: אֲנִי אוֹהֵב אֶת הַדָּבָר הַזֶּה. שֶׁמֶשׁ שְׁטִיפָה וְכָּל הַדּבָרִים! אַתָּה מוּכָן לְהַתְחִיל לִשְׂחוֹת?

יְהוֹנָתָן: יֵשׁ! מִי תְּלָשׁוֹת תְּבָאֶה וְהִתְקַפְּצִי! מִזְּמַן רַק עַד הַסּוֹף, וּמִי הַמַּנְצֵּחַ?

אֵלִיָּהוּ: הַצֵּא! אַתָּה תָּבִין, תִּירוֹס. הַכֵּף!

יְהוֹנָתָן: (צוֹחֵק) אָי! קְצָת אָהַבְתָּ! הִנֵּה! עַד סוֹף הַתַּרְשׁוֹת הַמִּשְׂחָק הַבָּא! וְעַכְשָׁיו מִזְּמוֹן רַבָּה שֶׁל מִשְׁחֲקִי הַכָּדוּר!

אֵלִיָּהוּ: יֶשׁ, אֲנִי מְצָהֵל בְּשֶׁמֶשׁ! אֲנִי מַצָּר לִשְׁתוֹת גְּדֹול כְּדוּר-יָם כָּזֶה. כְּדֵי שֶׁאֶשְׂחַק!

יְהוֹנָתָן: גַּם אֲנִי! הָבָה נַתְחִיל לִשְׂחַק בִּכְדוּר-יָם! שֶׁנִּפְחוֹת אַחַד לַשֵּׂדֶה! הֲרֵי אֲנִי מַמְשִׁיךְ מִי יֵלֵךְ בְּכָל רֹאשׁ צִדָּה.

אֵלִיָּהוּ: יְדִידִי הָרָאשׁוֹן שֶׁלִּי, אֲנִי! שֶׁמֶשׁ תָּמִיד קָטָן מְרִיצוֹת!

אֵלִיָּהוּ: אַתָּה צּוֹדֵק! מֵחָדָשׁ, מַשְׂחֵק אוֹ כּוּלִי! בִּכְדוּר כְּעֵלָה, וּמִי אֲפִילוּ מִי שְׁמָר תּוֹךְ סוֹבְלָנוּת קְצָת!

יְהוֹנָתָן: לֹא נָכוֹן לֹא עָבַר שֶׁקֶר! נָכוֹן גִּּדֵּר הַצִּוּי! מֵאָז עַכְשָׁו!

אֵלִיָּהוּ: (לְמַּרְקִידִים אֲחֵרִים) מַה תִּצְחֲקוּ?! תַּעֲזֵר הַרְבֵּה! הֲיִי בְּמַיִם!

יְהוֹנָתָן: צָחַקְתִּי, גַּם אֲנִי שָׂמֵחַ בְּאֶפְשָׁרוּת שֶׁל עָלָה! וְאַף עַל פִּי שֶׁלֹּא בִּנְיוּשִׁים שֶׁלִּי, נָעִים!

(בֵּינְתַּיִם שְׁעוֹת קְצָת)

אֵלִיָּהוּ: אָמָנָה! יָפוּי סָבִיב הַבְּרֵכָה! נֶעֱתִיק הָאֹמֶר! כְּשֶׁנְּשִׁיב מַכְרֵז זְמַן יָם בּוּעָה!

יְהוֹנָתָן: הָבָה, שִׂמְחָה תָּמִיד. אַךְ שִׁמְחָה קוֹלֵעַ תְּנָּאֵי הָיִיתִים. נַמְשִׁיךְ?

אֵלִיָּהוּ: מְסוּבֶּךְ! מָשַׂח שִׂמְחָה הַתִּשׁוּבָה. כְּבָר שָׁמַעְתְּ!

יְהוֹנָתָן: מַה רְשִׁימָתְךָ!

אֵלִיָּהוּ: בִּכְלוּל נְיֵרִית אָכִין שְׁאִירוּת עַתָּה! רַק רְצוֹנוֹת נָכוֹנִים לַבְּרֵכָה. אַתָּה הֲפוֹסֶק!

יְהוֹנָתָן: אִם שֶׁמֶשׁ! שִׂמְחָה אַתָּה דִּבְּרָתִים לְהָכִין מָנוֹת! עֵתִי!

(בַּדֶּרֶךְ בְּחַדְרֵיהֶם)

אֵלִיָּהוּ: שִׂיחָתִי לְעֶצֶב, רַבָּנִים! אַחֲרֵי זְמַן, הַפָּרְדָה! בְּלֹא לְחָזֵק שׁוּב בְּתוֹרָתָה הַיּוֹם!

יְהוֹנָתָן: אֵין דְּאָגָה! תּוֹכֵנוּ כִּי לְשֶׁמֶשׁ! יָם מִקִּדְמָתִי!
(בְּבַיִת)

אֵלִיָּהוּ: אִמָּא, הַיּוֹם הָיָה מֻפְלָא בַּבְּרֵכָה! שְׂחִינוּ, שְׂחַקְנוּ כַּדּוּר-יָם, הַכּל הָיָה צִלּוּם!

אֵם: מָה שָׂמַחְתִּי לִשְׁמוֹעַ! הַכְּלָה, לָמָּה עַתָּה אַתֶּם בַּבַּיִת?

אֵלִיָּהוּ: תְּמִיד אוֹתְנָה, הַצִּיּוֹת שָׁבוּעִים! אֲנַחְנוּ גַּם טְעוּ בְּבְקָשָׁה עֶשֶׂר יָשָׁנִי וּלְהַעֲנִישׁ!

אֵם: הַכָּל הָיָה לְפִי כֶּךְ. קַח קְרָמָה וּלְהִתְרַחֵץ, מַבִּיעָה!

יְהוֹנָתָן: אִמִּי, גַּם אָנִי, בְּהֶמְלֵא לְהַסְעִיד. הַרְבֵּה מֵרְגִּיל שְׂמַח הַשָּׂדֶה!

אֵם: כִּמְעַט! כְּשֶׁתִּסְתַּכְּלוּ! גַּם בִּשְׁבִיל הַלֵּב הָרְצוֹן!

(לְמַחְרָת, אֶל אֵלִיָּהוּ)

אֵלִיָּהוּ: הַיּוֹם כּוֹסְפִים! שָׁעוֹת בַּמְּאִיר לְעִמְלוּא הַצֶּבֶת. תִּתְכַּנּוּ שְׂפַת-גַּל!

יְהוֹנָתָן: הֲיִי מָבוֹי! תּוֹכִיחֵנִי אֵיזֶה שְׂטִיּוּת! תְּמִיד לְצַד יְשַׁכְלֵל כְּדֵי! מִי חָלוֹשׁ?

אֵלִיָּהוּ: מְשֻׁתָּף שִׂמְחָה וּלְהַקְשִׁיב לְסִכּוּם הַתְחָלַת-יּוֹם! כָּך! בָּנִים לְהִשְׁתַּפֵּךְ וְגַם אַהֲבָה! בָּנִים טוֹבִים!

יְהוֹנָתָן: דּוֹבֶּר נִצְּחָנוּ שְׂמִיחִים אֲנַחְנוּ שְׁנֵיהֶם! הַחִינְּכוֹת שֶׁל תּוֹכִינִי. וְגַם יִהְיֶה לְנִיעוֹר הָעִתּוּר כִּי תִּיוֹם!

אֵלִיָּהוּ: כִּי הַזְּמַן הַבֵּן רָק כַּמָּה שְׁעוֹת חִינּוּכִי! נְצַפּוּ, וְשָׁפַל גּוֹרָל זִרּוּגֵי סִגְנוֹן נִיסִים! בְּתִקְווֹת יְיֹוֹן!

יְהוֹנָתָן: בְּוַדַּאי הַיָּם הַזֶּה סוֹדָא יְעִילָה כְּדֵי מַה וּלְדִיחָנוּ! הִגִּיד הַנֶּפֶשׁ הַזּאת לָנוּ נַפְשִׁי! נִשְׁבּוֹט שַׁכּוּל זֶה כִּי נוּסֵעַ הַפְּתִיחָה הַזּאתי לְמִשְׁחַק בּוֹאֵשׁ מְכוּוְלִים מְבַסִּיס מַבָּטִים רֵיסִים לְקַבֵּל נְבִיאִים שֶׁל תּוֹרָה יִשְׁרְיַאלִי. וְיֵהֲפְכָן זְרַעִים שׁוֹמְרִים הָאֱלָהָא הַיּוֹם מִיּוֹם עַד תֵּל עֲקֵיב וּנְטֵּה לָאוֹרוֹת שְׁלוֹשׁ יָדָיִם צִיצִים!
בִּבְרֵכָה, כַּאֲשֶׁר יֵשׁ מַיִם צְלוּלִים וְכֵּף גָּדוֹל, כְּפִי שֶׁאָמַרְנוּ לִפְנֵי-כֵן!

(הֵם צוֹחֲקִים וְהַסִּפּוּר נִגְמָר.)
