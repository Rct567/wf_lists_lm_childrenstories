אֲגַדַת סֵפֶר הַסּוֹד, שֶׁנִּכְתְּבָה עַל־יְדֵי הַחֲכָמִים הַקְּדוּמִים בִּירוּשָׁלַיִם.

אֲשֶׁר לֹא תָּמוּת עֵצָה, עֲבֹר וְשׁוּב: שִׁמְעוּ נָא, קוֹרְאִים יְקָרִים, אֶת סִפּוּר הַסּוֹד, שֶׁבְּלִי עֲבוּר לְפָנָיו תּוֹעֶלֶת. לְפָנִים, בַּיָּמִים שֶׁהָיוּ רַק שֶׁמֶשׁ וְיָרֵחַ, הָיָה סוֹד גָּדוֹל שֶׁלֹּא שֻׁמַּר.

שָׁלִי: אַבָּא, אֲנִי רוֹצֶה לִשְׁמוֹעַ סִפּוּר!

אַבָּא (סַבָּא שֶׁל שָׁלִי): בְּוַדַּאי, יַלְדִּי שֶׁלִּי. אֲנִי אֲסַפֵּר לְךָ עַל הַסּוֹד.

שָׁלִי: מַה הוּא הַסּוֹד?

אַבָּא: הַסּוֹד הָיָה תָּמִיד סְבִיבֵנוּ, אֲבָל אַף אֶחָד לֹא יָדַע לְהָבִין אוֹתוֹ. הוּא הָיָה בְּכָל מָקוֹם, בַּאֲוִיר שֶׁאָנוּ נוֹשְׁמִים, בַּשֶּׁמֶשׁ הַיּוֹקֶדֶת, וּבַלַּיְלָה הַמָּלֵא כּוֹכָבִים.

שָׁלִי: וּמָה קָרָה לַסּוֹד הַזֶּה?

אַבָּא: הַסּוֹד שֻׁמַּר. הוּא בָּרַח מֵאֲנָשִׁים שֶׁלֹּא הָיוּ מוּכָנִים לִשְׁמֹעַ אוֹתוֹ, וְהִסְתַּתֵּר בַּמְּקוֹם שֶׁיֵּשׁ בּוֹ רַק דְּמָמָה.

פֶּרֶק ב': מְסִירַת הַסּוֹד

שָׁלִי: הַיּוֹם שַׁבָּת. הָיִינוּ בְּבֵית הַכְּנֶסֶת. מָה נַעֲשֶׂה עַכְשָׁו?

אַבָּא: עַכְשָׁו נַעֲשֶׂה אֶת הַחֵלֶק הַיּוֹתֵר טוֹב: נַחְקֹר אֶת הַסּוֹד! הָיְתָה אִשְׁתָּא אַחַת, בְּתוּלָה טוֹבָה, שֶׁבִּשְׁמָהּ אַסְפּוֹלְמִירָה. הִיא הִתְיַעֲצָה בִּכְלִי מִשְׁקָל בָּרוּחַ, בָּרִיסוֹת אֲבָנִים שֶׁלָּהּ, עַד שֶׁקִּבְּלָה הַחְלָטָה.

שָׁלִי: מָה הִיא עָשְׂתָה?

אַבָּא: הִיא הִתְחִילָה לְדַבֵּר אֶל הַלֵּב שֶׁל עַצְמָהּ. הִיא שָׁאֲלָה שְׁאֵלוֹת. "לָמָּה אֲנִי קַיֶּמֶת? מִי אֲנִי?" וְכֵן הָלְאָה.

שָׁלִי: וְאִם כָּךְ?

אַבָּא: הַשֵּׁם, סוֹפְסוֹף, הֵשִׁיב לָהּ בַּעֲנָנִים שֶׁבָּאִים וְחוֹלְפִים. הוּא שָׁלַח אֶת סוֹד הַחֲכָמִים, אֲשֶׁר קַרְנוֹתָיו זָרְחוּ. הַסּוֹד הַזֶּה בְּצֵאתוֹ, בְּעֶזְרַת הַצֵּירוּף הַכַּבִּיר, שֶׁל צְפוֹן דָּרוֹם, מִזְרָח מַעֲרָב. הָפַךְ לְצוּרָה שֶׁאָמְרָה: "כַּאֲשֶׁר תַּבִּינוּ אֶת הַלֵּב שֶׁלְּכֶם, תָּבִינוּ אוֹתִי".

שָׁלִי: זֶה נִשְׁמָע מַפְחִיד! מַה קָּרָה לְאַסְפּוֹלְמִירָה?

אַבָּא: אַסְפּוֹלְמִירָה, בְּאוֹתוֹ רֶגַע, חָשָׁה אֶת מְהוּתוֹ שֶׁל הַסּוֹד. הִיא הֵבִינָה שֶׁהַסּוֹד אֵינוֹ רַק מִלִּים. זֶה דֶּרֶךְ חַיִּים, שֶׁהִיא צְרִיכָה לִלְמֹד, לְהַכִּיר, וּלְהַצְדִּיק.

פֶּרֶק ג': הַמִּסְתּוֹרִין הַגָּדוֹל

שָׁלִי: אֲבָל אֲנִי לֹא מֵבִין. מַה זֶּה "הַמִּסְתּוֹרִין הַגָּדוֹל"?

אַבָּא: הַמִּסְתּוֹרִין הַגָּדוֹל, יַלְדִּי, הוּא הַחִידָה שֶׁבְּתוֹכֵנוּ. זֶה הַכֹּל שֶׁלֹּא הֵבַנוּ עֲדַיִן, אֶת הַיְּקוּם, אֶת חַיֵּינוּ, אֶת עַצְמֵנוּ.

שָׁלִי: אֲנִי אוֹהֵב חִידוֹת!

אַבָּא: כֵּן. וְהַסּוֹד הִבְטִיחַ לַעֲזֹר לְכָל מִי שֶׁיְּנַסֶּה לְהָבִין אוֹתוֹ. הוּא לֹא יִתְחַבֵּא, לְעוֹלָם לֹא. הוּא שֶׁמֵּבִין שֶׁצְּרִיכִים לְהַשְׁקִיעַ תְּנוּעָה, כְּדֵי לְהַשְׁקִיעַ חָכְמָה.

שָׁלִי: אֲנִי רוֹצֶה גַּם לְהָבִין! מָה אֲנִי צָרִיךְ לַעֲשׂוֹת?

אַבָּא: דְּבָרִים פְּשׁוּטִים. לְפִי עֵצַת הַסּוֹד: שִׁמַּע אֶת לִבְּךָ, שְׁאַל שְׁאֵלוֹת, וּלְעוֹלָם אַל תְּפַחֵד מֵהַדְּבָרִים שֶׁאַתָּה לֹא מֵבִין. תַּמְתִּיק אֶת עֶשְׁרִיךָ עִם אֲנָשִׁים. דַּע שֶׁהָאֶמֶת תַּגִּיעַ לְךָ בְּרֶגַע הָעִנְיָן.

שָׁלִי: זֶה נִשְׁמָע קַל!

אַבָּא: בֶּאֱמֶת? שֶׁלָּךְ! הוּא כָּל כָּךְ קַל, שֶׁכֵּן כָּל אֶחָד יָכוֹל לְהָבִין אוֹתוֹ. הָאַתְגָּר הָעִיקָּרִי הוּא לְהַשְׁלִים אֶת הַדֶּרֶךְ.

פֶּרֶק ד': הָעֲצִיצוֹת שֶׁל הַסּוֹד

שָׁלִי: הַבַּיִת שֶׁלָּנוּ יָפֶה עִם הַכֵּלִים שֶׁנָּשָׂא בּוֹ.

אַבָּא: כֵּן, וּבְכָל זֹאת בֵּית הָעֲצִיץ יָשִׁיר עָלֶיךָ. אַסְפּוֹלְמִירָה הָיְתָה קוֹבַעַת קוֹמַת סְנֵי מִסְפָּר, הִיא הָפְכָה אוֹתָם לִכְלֵי אֻמָּנוּת עֲדִינוּת. זֶה הָיָה הַיִּעוּד שֶׁלָּהּ. הַיַּחַד הַגָּדוֹל, הַצִּדּוּק שֶׁבַּעֲמִידַתָם בִּסְבִיבוֹת שֶׁהִתְאִימוּ, הִתְקַבֵּל מְאֹד בִּמְנוּחַת לִבָּהּ.

שָׁלִי: וְאִם כֵּן, מַה אֵיפְשָׁר לְהָבִין מִכֵּלִים אֵלּוּ?

אַבָּא: מִכְּלֵי הַזֶּה יָכוֹל לְהָבִין עַל הָאַהֲבָה, עַל הָאַחְוָוה. עֲצִיצֵי הַזְּמַן אֲשֶׁר הֵכִינָה. בָּהֶם שׁוֹכְנוּ זִכְרוֹנוֹת. כָּל אֶחָד הָיָה סִימָן, קוֹנֶה מְשַׁמֵּשׁ שֶׁל סִפּוּר, שֶׁהַסּוֹד דִּבֵּר עַל זֵה. וּבִתְנוּעַת כְּפִיפַת הָרֹאשׁ וּלְמֹסִיפוּת עַכְשָׁו, מַרְגִּישִׁים אֶת מְעוּפָם.

פֶּרֶק ה': הַסּוֹד וְהָעוֹלָם

שָׁלִי: הַרְבֵּה שָׁנִים עוֹבְרִים...

אַבָּא: נָכוֹן. וּבְכָל שָׁנָה הַסּוֹד הִמְשִׁיךְ לַעֲזוֹר לַאֲנָשִׁים. הַרְבֵּה מִכַּתְבִים וּמִשְׁעִישִׁים, וְכָל מִי שֶׁחָשַׁב שֶׁהוּא כְּלוּם, הָפַךְ לְשַׁיָּךְ עִם עֲזֵרַת הַסּוֹד. הוּא עָזַר לָהֶם לְהִתְגַּבֵּר עַל פַּחֲדֵיהֶם, לְהַאֲמִין בְּעַצְמָם, וְלִמְצֹא שִׂמְחָה בְּכָל דָּבָר.

שָׁלִי: אֲבָל לָמָּה הוּא לֹא עָזַר לְכֻלָּם?

אַבָּא: כִּי לֹא כֻּלָּם הָיוּ מוּכָנִים לְהַקְשִׁיב. יֵשׁ אֲנָשִׁים שֶׁמְּחַפְּשִׂים דְּבָרִים פְּשׁוּטִים, אֲשֶׁר הַכֹּל צָרִיךְ לְהָבִין, וְאֵינָם מוּכָנִים לַעֲמֹד. הַסּוֹד יָדַע שֶׁהוּא צָרִיךְ לִהְיוֹת מִשֶּׁלָּהֶם בַּכּוּל, עַד שֶׁיֵּרָצוּ. הוּא לֹא יָכֹל לִכְפוֹת עַל אִישׁ אֶת הַיְדִיעָה.

שָׁלִי: זֶה הַרְבֵּה.

אַבָּא: כֵּן. הַסּוֹד הֵבִין שֶׁזֶּה חֲשׁוּב לְאֶחְדוּת, לְהָבִין אֶת הַשּׁוֹנוּת. הוּא הִרְחִיק אֲנָשִׁים שֶׁאִיְּרוּ כְּדֵי לְלַמֵּד בִּבְטִיחוּת שֶׁיֵּשׁ מִסְפַּר אֱמֶת, עֲנִיָּה אוֹ עֲשִׁירָה.

פֶּרֶק ו': הַמִּסְתּוֹרִין הַיּוֹם

שָׁלִי: אֲנִי רוֹצֶה שֶׁתַּסְפִּיק שׁוּב!

אַבָּא: כֵּן, וְעַכְשָׁו הָיִינוּ חָשׁוּבִים. הַסּוֹד לֹא פָּסַק לִהְיוֹת פָּעִיל. אֲנָשִׁים שׁוּב וָשׁוּב גִּלּוּ אוֹתוֹ, וּבַתּוֹסְפָתָה תְּכוּנָה. אֲנָשִׁים קִבְּלוּ שִׁכְלָם חָשׁוּב לַחַיִּים, לְמִי שֶׁקָּרָא, וּבְרֹאשׁ הַסִּירוּס.

שָׁלִי: וּמָה אִתָּנוּ?

אַבָּא: גַּם אֲנַחְנוּ. בְּכָל יוֹם אֲנַחְנוּ מְנַסִּים לְהָבִין יוֹתֵר, לְהַקְשִׁיב לַסּוֹד בַּלֵּב שֶׁלָּנוּ. אֲנַחְנוּ עוֹסְקִים בַּלִּימוּד שֶׁל לֵב שְׁנִי. מִלִּימוּד זוֹ, אֲנִי הַשְׁלָמָה. הַדֶּרֶךְ בָּרֶגַע זֶה.

שָׁלִי: אֲנִי אוֹהֵב אֶת הַסּוֹד!

אַבָּא: וְהַסּוֹד אוֹהֵב אוֹתְךָ. הוּא תָּמִיד אִתְּךָ. הַסּוֹד הַגָּדוֹל שֶׁהוּא כְּדַי לַעֲבֹד כְּדֵי לָשׁוּב וְלָשׁוּב, וְלִהְיוֹת מְסוּגָּלִים לְהִתְמוֹדֵד עִם הַמְּצִיאוּת הַגְּדוֹלָה הַזֹּאת. וְלְעוֹלָם לֹא לִשְׁכֹּחַ שֶׁהַיַּלְדּוּת הִיא שֶׁלָּנוּ וְרַק בָּהּ אֶפְשָׁר לְבַקֵּשׁ. שֶׁהָאֱמֶת בָּאֲדָמָה כָּרָה, בַּיְּרִיחִים תֵּלֵךְ. וְלֹא תִּשְׁכַּח שֶׁהַסּוֹד הַגָּדוֹל הֵינוּ שֶׁלָּךְ, וּמִי שֶׁיּוֹדֵעַ הַכֹּל הֵינוּ הַסּוֹד. וּתְנַסֶּה, יֶלֶד שֶׁלִּי, לְהַכִּיר אוֹתוֹ כַּאֲשֶׁר תִּגְדַּל.

שָׁלִי: הָיָה מַרְהִיב! תּוֹדָה, אַבָּא!

אַבָּא: בִּבְקָשָׁה, יַלְדִּי. הַכָּל בְּסֵדֶר, אֲנַחְנוּ עֲדַיִן הַיּוֹם, וְהַסּוֹד הַגָּדוֹל תָּמִיד שֶׁלָּנוּ. וּכְשֶׁתְּבַקֵּשׁ - תְּגַלֶּה. עַד לַפַּעַם הַבָּאָה.
