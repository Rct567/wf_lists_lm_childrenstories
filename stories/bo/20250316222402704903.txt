བུ་ཆུང་གཉིས་ཀྱིས་ཀྱང་ཕ་ལགས་ཀྱི་གསུང་ལ་ཉན་ཞིང་། རང་གི་སེམས་ནང་དུ་བཅངས་སོ། ཁོ་ཚོ་རང་ཁྱིམ་དུ་ལོག་སྐབས། བསྟན་འཛིན་གྱིས་ “ཨ་ཕ། ངས་འབྱུང་འགྱུར་དུ་གནའ་རྫས་སྲུང་སྐྱོབ་བྱེད་མཁན་ཞིག་བྱེད་འདོད།” བདེ་སྐྱིད་ཀྱིས་ཀྱང་ “ང་ཡང་ལོ་རྒྱུས་སློབ་མཁན་ཞིག་བྱེད་འདོད།” ཅེས་བཤད། ཕ་ལགས་དགའ་པོའི་ངང་གནས་ཤིང་། རང་གི་བུ་ཕྲུག་རྣམས་ལ་བརྩེ་བས་བལྟས།

མཚན་མོར་བསྟན་འཛིན་གྱིས་རྨི་ལམ་དུ་རིག་འཛིན་ཆེན་པོ་མཐོང་། ཁོང་གིས་བསྟན་འཛིན་ལ་གདེང་འཇོག་བྱེད་ཀྱི་ཡོད། བསྟན་འཛིན་གྱིས་རྨི་ལམ་དུ་ཤེས་བྱ་མང་པོ་ཞིག་རྙེད་བྱུང་། བདེ་སྐྱིད་ཀྱིས་ཀྱང་རྨི་ལམ་དུ་ལོ་རྒྱུས་དེབ་གཞུང་མང་པོ་མཐོང་། དེབ་དེ་དག་ནི་མཛེས་སྡུག་དང་རྟོགས་དཀའ་བའི་གནས་ཚུལ་མང་པོ་ཡོད་པ་མཐོང་བྱུང་། དེའི་ཉིན་ནས་བཟུང་། བུ་ཆུང་གཉིས་ཀྱིས་སློབ་སྦྱོང་ལ་ཧུར་ཐག་བྱས་ཤིང་། ཕ་ལགས་ཀྱིས་ཀྱང་དེ་དག་ལ་རོགས་རམ་གནང་།

ལོ་མང་པོའི་རྗེས་སུ། བསྟན་འཛིན་གནའ་རྫས་རིག་པའི་མཁས་དབང་ཆེན་པོར་གྱུར་། ཁོང་གིས་རང་གི་ཤེས་བྱས་གནའ་རྫས་མང་པོ་ཞིག་བཙལ་རྙེད་བྱུང་བ་མ་ཟད། གནའ་རྫས་ལ་སྲུང་སྐྱོབ་བྱེད་པའི་ལས་དོན་ཡང་བསྒྲུབས། བདེ་སྐྱིད་ཀྱང་ལོ་རྒྱུས་མཁས་དབང་ཆེན་པོར་གྱུར་ཏེ། ཁོང་གིས་གནའ་བོའི་ལོ་རྒྱུས་དེབ་བྲིས་པ་དང་། སློབ་མ་མང་པོར་ལོ་རྒྱུས་ཀྱི་གོ་རྟོགས་སྤེལ། དེ་ལྟར་བུ་ཆུང་གཉིས་ཀྱིས་རི་ཁྲོད་ཀྱི་གནའ་རྫས་དང་ལོ་རྒྱུས་གསང་བར་བརྟེན་ནས། རང་ཉིད་ཀྱི་འཚོ་བ་གསར་བ་ཞིག་གཏོད་ཐུབ་པ་བྱུང་།

གཏམ་རྒྱུད་འདིར་མི་རྣམས་ལ་གནའ་རྫས་དང་ལོ་རྒྱུས་ལ་དུང་བ་དང་སྲུང་སྐྱོབ་བྱེད་དགོས་པའི་བསླབ་བྱ་གནང་ཡོད། དེ་བཞིན་དུ་རང་ཉིད་ཀྱི་བསམ་བློ་བརྟན་པོ་བཅངས་ཏེ་འབད་པ་བྱས་ན། འདོད་དོན་འགྲུབ་པའི་བརྗོད་དོན་ཡང་ཡོད། བུ་ཆུང་གཉིས་ནི་མི་རབས་རྗེས་མ་དག་ལ་སེམས་ཤུགས་སྦྱིན་པའི་མིག་དཔེ་ལྟ་བུར་གྱུར་ཡོད།
