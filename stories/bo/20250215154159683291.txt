དགའ་ལྡན་ཧ་ཅང་ཧོན་འཐོར་སོང་། ཁོས་སེམས་ལ་བསམ་བློ་བཏང་། དྲི་བ་ག་རེ་འདྲི་དགོས་སམ། མཐའ་མར། ཁོས་སེམས་ཐག་བཅད་དེ། “ལགས་སོ། སངས་རྒྱས་ལགས། འཇིག་རྟེན་འདིར་བདེ་སྐྱིད་ཇི་ལྟར་སྒྲུབ་དགོས་སམ།” ཞེས་དྲིས།

སངས་རྒྱས་ཀྱིས་འཛུམ་དམུལ་ངང་། “བུ་ཆུང་། བདེ་སྐྱིད་ནི་ཕྱི་རོལ་དུ་འཚོལ་དགོས་པ་ཞིག་མིན། དེ་ནི་ཁྱོད་ཀྱི་སེམས་ནང་ལ་ཡོད། ཁྱོད་ཀྱིས་སེམས་པ་བཟང་པོ་ཞིག་བསྐྱེད་ནས་གཞན་ལ་བྱམས་བརྩེ་དང་སྙིང་རྗེས་བལྟོས་ན། བདེ་སྐྱིད་དེ་རང་བཞིན་གྱིས་ཡོང་།” གསུངས།

དེ་ནས་སངས་རྒྱས་ཀྱིས་མུ་མཐུད་དུ། “ཁྱོད་ཀྱིས་གཞན་ལ་རོགས་རམ་བྱེད་དགོས། དཔེར་ན། ཁྱོད་ཀྱི་གྲོགས་པོ་ཚོར་དཀའ་ངལ་ཡོད་ཚེ། ཁྱོད་ཀྱིས་ཁོང་ཚོར་རོགས་རམ་བྱེད་དགོས། སེམས་ཅན་གཞན་དག་ལ་སྡུག་བསྔལ་ཡོད་ཚེ། ཁྱོད་ཀྱིས་དེ་དག་ལ་བརྩེ་བ་སྟོན་དགོས། ཁྱོད་ཀྱིས་གཞན་ལ་སྡང་སེམས་དང་འཁོན་འཛིན་བཅོལ་མི་རུང་། བཟོད་སྒོམ་དང་བརྩེ་བའི་སེམས་བསྐྱེད་ན་བདེ་སྐྱིད་དེ་རང་བཞིན་གྱིས་ཡོང་།” གསུངས།

དགའ་ལྡན་གྱིས་སངས་རྒྱས་ཀྱི་གསུང་དེ་དག་ཚང་མ་ཉན་རྗེས། ཁོ་རང་བསམ་བློ་གཏོང་བཞིན་བསྡད། ཁོ་རང་གིས་མཚམས་རེར་གྲོགས་པོ་ཚོར་ཁོང་ཁྲོ་ཟ་བ་དང་། གཞན་ལ་རོགས་རམ་མ་བྱས་པ་དྲན་སོང་། སངས་རྒྱས་ཀྱིས་མུ་མཐུད་དུ་གསུངས་དོན། “བུ་ཆུང་། དགོས་པ་ནི་རང་ཉིད་ཀྱི་སྐྱོན་མཐོང་ནས་དེ་དག་ལ་བཅོས་བསྒྱུར་བྱེད་དགོས། ཐོག་མར་ཁྱོད་རང་ཉིད་ལ་བཟང་ཕྱོགས་ལ་བསྒྱུར་དགོས། དེ་ནས་གཞན་ལ་བཟང་ཕྱོགས་ལ་སློབ་གསོ་སྤྲོད་དགོས།”

དེ་ནས་རླུང་བསྐྱོད་པ་ལྟར་རྨི་ལམ་ནང་གི་ཡུལ་ལྗོངས་ཡོངས་རྫོགས་འགྱུར་ལྡོག་བྱུང་སོང་། དགའ་ལྡན་ནི་གངས་རི་ལས་བབས་ཏེ། ཁོ་རང་གི་གྲོང་པ་བརྒྱུད་དེ་ཕྱིན་སོང་། ཁོས་མི་རྣམས་ཀྱི་འཚོ་བ་ལ་བལྟས་པ་དང་། མི་རྣམས་ལ་དཀའ་ངལ་མང་པོ་ཡོད་པར་མཐོང་། སྐྱེ་བོ་འགའ་ཞིག་དབུལ་ཕོངས་ཀྱིས་མནར་བ། འགའ་ཞིག་ལ་ནད་གཞིས་མནར་བ། འགའ་ཤས་ལ་ཁོང་ཁྲོ་ཡོད་པ་མཐོང་། དགའ་ལྡན་གྱིས་སེམས་ལ་ཕངས་སེམས་སྐྱེས་ཏེ། ཁོ་རང་གིས་ཅི་ཞིག་བྱེད་ཐུབ་བམ།

ཁོ་རང་ཁྱིམ་དུ་ལོག་སྟེ། ཕ་མ་གཉིས་ལ་གཏམ་རྒྱུད་དེ་བཤད་པ་དང་སངས་རྒྱས་ཀྱི་གསུང་དེ་དག་བཤད་པས། ཕ་མ་གཉིས་ཧ་ཅང་དགའ་པོ་བྱུང་། ཕ་ཡིས་དགའ་ལྡན་ལ་བཤད་དོན། “བུ་ལགས། ཁྱོད་ཀྱིས་སེམས་པ་བཟང་པོ་ཞིག་བསྐྱེད་ནས་གཞན་ལ་རོགས་རམ་བྱེད་དགོས། ང་ཚོས་དབུལ་ཕོངས་ཚོར་ཟས་རོགས་དང་། ནད་པ་ཚོར་སྨན་རོགས། ཁོང་ཁྲོ་ཡོད་པ་ཚོར་བཟོད་སྒོམ་གྱི་རྣམ་འགྱུར་སྟོན་དགོས།”

དགའ་ལྡན་གྱིས་དེར་ཁས་བླངས་ཏེ། ཕ་མ་གཉིས་དང་མཉམ་དུ་སེམས་ཅན་ཚང་མར་རོགས་རམ་བྱེད་པའི་ལས་ལ་ཞུགས་སོང་། ཁོང་ཚོས་ལམ་འགྲམ་དུ་ཟས་སྐོམ་འཇོག་པ་དང་། ནད་པ་ཚོར་སྨན་སྤྲད་པ། ཁོང་ཁྲོ་ཡོད་པ་ཚོར་བཟོད་སྒོམ་གྱི་སྐད་ཆ་བཤད། དེ་ལྟར་བྱས་པས་མི་རྣམས་ཀྱི་སེམས་ལ་བདེ་སྐྱིད་དང་དགའ་སྤྲོ་སྐྱེས་ཤིང་། ཁྱིམ་ཚང་ཡང་མཐུན་སྒྲིལ་གྱིས་ཁེངས་སོང་།

དུས་ཚོད་རིང་པོའི་རྗེས་སུ། དགའ་ལྡན་གྱི་སེམས་ནང་དུ་ཁ་དོག་སྣ་ཚོགས་ཀྱི་མེ་ཏོག་བཞད་དེ་། ཁོ་རང་ལ་བདེ་སྐྱིད་ཀྱི་ཚོར་བ་ཟབ་མོ་ཞིག་བྱུང་། ཁོའི་རྨི་ལམ་དུ་ཡང་བསྐྱར་སངས་རྒྱས་བྱོན་ནས། “བུ་ཆུང་། ཁྱོད་ཀྱིས་ལས་ཀ་ཧ་ཅང་ཡག་པོ་བསྒྲུབས་སོང་། ཁྱོད་ཀྱིས་བདེ་སྐྱིད་ནི་གཞན་ལ་རོགས་རམ་བྱེད་པ་དང་། རང་གི་སེམས་པ་བཟང་པོ་ཞིག་བསྐྱེད་པ་ལས་བྱུང་བ་ཤེས་ཡོད།” གསུངས།

དགའ་ལྡན་གྱིས་འཛུམ་དམུལ་བྱས་ཏེ། “ལགས་སོ། ངས་ཤེས་སོང་། སངས་རྒྱས་ལགས། ངས་སེམས་ཅན་ཚང་མར་བརྩེ་བ་དང་སྙིང་རྗེ་བསྐྱེད་རྒྱུ་ཡིན།” ཞེས་བཤད།

རྨི་ལམ་མཇུག་བསྡུས་ཤིང་། དགའ་ལྡན་གཉིད་ལས་སད་དོ། ། ཉི་མ་ཤར་ཟིན་པས། ཁོའི་སེམས་ནང་དུ་བདེ་སྐྱིད་ཀྱི་འོད་ཟེར་འཕྲོ་བཞིན་འདུག ། ཁོས་ཕ་མ་དང་རྒན་པོ་བཀྲ་ཤིས་ལ་གཏམ་རྒྱུད་དེ་བཤད་པས། ཁོང་ཚོ་ཚང་མ་ཧ་ཅང་དགའ་པོ་བྱུང་། དེ་ནས་ཁོང་ཚོས་མཉམ་དུ་གཞན་ལ་རོགས་རམ་བྱེད་པའི་ལས་ལ་བརྩོན་པར་བརྩམས། དགའ་ལྡན་ནི་སེམས་པ་བཟང་པོ་དང་། བརྩེ་བ་ལྡན་པའི་བུ་ཆུང་ཞིག་ཏུ་གྱུར་སོང་། ཁོས་གནས་ཚུལ་དཀའ་ངལ་ཅན་མང་པོ་ཞིག་ལ་གདོང་ལེན་བྱས་པ་དང་། རང་གི་ཕ་ཡུལ་གྱི་མི་དམངས་ལ་བདེ་སྐྱིད་དང་ཞི་བདེ་འཚོལ་བར་རོགས་རམ་བྱས། ཁོས་ཚེ་གང་པོར་དཀའ་སྤྱད་བརྒྱུད་དེ་མིའི་རིགས་ལ་བདེ་སྐྱིད་སྦྱིན་པའི་ལས་ལ་གཞོལ་བཞིན་ཡོད་དོ།།
