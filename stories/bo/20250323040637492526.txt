ཡིན་ན་ཡང་། བདེ་སྐྱིད་ནི་ད་དུང་རེ་བ་མི་འཆོར་བར་བོང་བུར་བསླབ་བྱ་སྣ་ཚོགས་བཤད་དོ། ཁོས་“འཚམས་འདྲི་བྱེད་པ་མཁྱེན་ནམ། ཁྱེད་ཀྱི་མིང་ལ་ཅི་ཟེར། ཁྱེད་ཀྱིས་དགའ་བའི་གླུ་དེ་ཅི་རེད།” ཅེས་དྲིས་སོ། སྨྲ་བ་མེད་པའི་རྣམ་པ་དེས་མུ་མཐུད་གནས་ནས། བདེ་སྐྱིད་ནི་ཞེད་སྣང་ཆེན་པོ་སྐྱེས། སེམས་ཀྱི་འཆར་སྒོ་བཅད་དེ། “བོང་བུ། ཁྱོད་རང་སེམས་སྐྱིད་པོ་མི་འདུག ང་ཚོས་ཁྱེད་རང་ལ་རོགས་རམ་བྱེད་དགོས།” ཞེས་བཤད། སྐད་ཆ་འདི་ཐོས་མ་ཐག་ཏུ། བོང་བུས་ཡ་མཚན་པའི་ངང་ནས་འགུལ་སྐྱོད་བྱས། མིག་གཉིས་ཡངས་པོར་ཕྱེས་ཤིང་། ནང་གི་སྒྲ་དེ་མངོན་དུ་འགྱུར། ཁོའི་སྐད་ཆ་དེ་ནི་རྒན་པོ་ཞིག་གི་སྐད་ཆ་ཡིན་པར་བསྡད།

“ང་ནི་ཚེ་རིང་རེད།” བོང་བུས་ལན་བཏབ། “ངས་གླུ་ལེན་པར་དགའ་པོ་ཡོད། ཡིན་ནའང་མི་ཞིག་གིས་ང་རྡོ་རྗེའི་མདུན་དུ་བཏབ་རྗེས། ང་འདིར་བཀག་སོང་། ཁྱེད་ཚོས་ང་རང་ཐར་པར་རོགས་རམ་བྱེད་ཐུབ་བམ།” རྡོ་རྗེ་དང་བདེ་སྐྱིད་ནི་ཧ་ལས་དགོད་སྤྲོ་སྐྱེས། བོང་བུ་ནི་རྣམ་ཤེས་ཡིན་པར་མི་སེམས་མོད། ཡིན་ན་ཡང་ཁོས་སྡུག་བསྔལ་མྱངས་པར་མཐོང་སྐབས། ཁོ་ཚོའི་སྙིང་རྗེ་སྐྱེས་སོ། བདེ་སྐྱིད་ཀྱིས་ “གལ་ཏེ་ང་ཚོས་རོགས་རམ་བྱས་ན་ག་འདྲ་ཡིན། ང་ཚོས་ག་རེ་བྱེད་དགོས།” ཞེས་དྲིས། ཚེ་རིང་གིས་བརྗོད་དོ། “ཁྱེད་ཚོས་ཐོག་མར་རྡོ་རྗེའི་མདུན་ལ་སོང་དགོས་པ་རེད། ཁོས་བུད་ཤིང་རྙིང་པ་དེ་ཡོང་ནས་ང་རང་ཐར་དུ་འཇུག་ཐུབ།”

རྡོ་རྗེ་དང་བདེ་སྐྱིད་གཉིས་ཀྱིས་འཆར་གཞི་འདི་ལ་ཁས་ལེན་བྱས། ཁོ་གཉིས་རྡོ་རྗེའི་མདུན་ལ་རྒྱུགས་པ་རེད། ཡིན་ན་ཡང་། རྡོ་རྗེའི་ཁྱིམ་དུ་སླེབས་སྐབས། རྡོ་རྗེ་མེད་པར་གྱུར་ཡོད། ཁོ་ཚོའི་ཕ་མ་ནི་ཁྱིམ་གྱི་ལས་ཀ་ལ་བྲེལ་ཟིང་ཆེ་བས་ན་ཐོས་མ་ཐག ཕྲུ་གུ་གཉིས་ལ་ངོ་མཚར་ལྡན་པའི་བདེ་སྐྱིད་ཀྱི་མདུན་ལམ་ཞིག་ཡོང་སྲིད། ཁོ་གཉིས་ཀྱིས་རྡོ་རྗེ་འཚོལ་བར་སོང་། འཚོལ་ཞིང་འཚོལ་ཞིང་། གནས་མཚམས་ཤིག་ལ་སླེབས་དུས། རྡོ་རྗེ་ནི་དགོན་པའི་རྩྭ་ཐང་ཐོག་ནས་སྨན་སྦྱོར་བྱེད་བཞིན་འདུག་ལ། ཁོ་ལ་ཤེས་བྱ་མང་པོ་ཡོད་པར་སྣང་། བདེ་སྐྱིད་ཀྱིས “རྡོ་རྗེ། ང་ཚོ་ལ་རོགས་རམ་བྱེད་དགོས།” ཞེས་བརྗོད་དོ། རྡོ་རྗེས་གནས་ཚུལ་གོ་མ་ཐག་ཁོང་གིས་རོགས་རམ་བྱེད་པར་ཁས་ལེན། ཁོ་གསུམ་པོས་མཉམ་དུ་བོང་བུའི་གམ་དུ་ཕྱིར་ལོག རྡོ་རྗེས་བུད་ཤིང་རྙིང་པ་དེ་བསླངས་ཏེ། ཁོས་བོང་བུར་རལ་གྲི་བརྒྱབ་སོ། ཧ་ལས་པའི་ངང་ནས་བལྟས་ན། བོང་བུ་དེ་དུམ་བུར་འགྱུར་ནས་མི་སྣ་ཞིག་ཡོང་བཞིན་འདུག མི་དེ་ནི་རྒས་པ་རྒན་པོ་ཞིག་རེད།

མི་རྒན་པོ་དེས“ངས་ཁྱེད་ཚོར་ཐུགས་རྗེ་ཆེ་ཞུ་ཡོད། ཁྱེད་ཚོས་ང་ལ་རོགས་རམ་བྱས་པས་ངས་ད་དུང་དངོས་ཡོད་འཇིག་རྟེན་འདིར་མུ་མཐུད་ནས་གནས་ཐུབ།” ཅེས་བཤད། ཁོ་ནི་ཚེ་རིང་རེད། ཁོས་རྡོ་རྗེ་དང་། བདེ་སྐྱིད། ཕ་མ་ཡོད་ཚད་ལ་གཏམ་བཤད་དོ། ཁོ་ནི་དོན་དངོས་སུ་འདས་པའི་དུས་ཤིག་གི་མི་རྒན་པོ་ཞིག་ཡིན་ཞིང་། ཁོས་མཛའ་བརྩེ་དང་བདེ་སྐྱིད་ཀྱི་སྐོར་ལ་གླུ་བླངས་ནས། མི་ཚང་ལ་དགའ་སྐྱིད་བསྩལ། ཁོས་གླུ་དེ་བླངས་ཚར་རྗེས། རང་གི་ངོ་བོར་ལོག་ནས་མཁའ་ལ་འཕུར་སོང་། འོད་སྣང་དང་འོད་ཟེར་གྱིས་ཁྱིམ་གང་བར་གྱུར། ཁོ་ཚོས་རྣམ་ཤེས་ཀྱི་རྩེད་མཚར་དེ་ཤེས་རྟོགས་བྱུང་།

ཁྱིམ་ཚང་དེ་ཡང་བསྐྱར་དགའ་སྤྲོ་དང་བདེ་སྐྱིད་ཀྱིས་ཁེངས་སོ། བདེ་སྐྱིད་ཀྱིས་གདོང་ལ་འཛུམ་མདངས་ཀྱིས་ཁེངས་ཤིང་། ཁོས་གྲོང་སྡེའི་མི་མང་ལ་རྣམ་ཤེས་ཀྱི་གླུ་དབྱངས་དང་དེའི་རྒྱུ་རྐྱེན་སྐོར་བཤད་པ་རེད། ཁོས་རྡོ་རྗེ་ལའང་སྐྱེས་སྦྱངས་ཀྱི་མི་མངོན་པའི་སྤྱང་གྲུང་སྐྱེས་པར་བྱས་ཏེ། མི་ཚང་ཚང་མས་མཉམ་དུ་བདེ་སྐྱིད་ཀྱི་དགའ་སྐྱེམས་ཀྱི་སྒྲུང་ལ་ཉན་དུ་བཅུག རྣམ་ཤེས་ཀྱི་གླུ་དབྱངས་དེས། གྲོང་སྡེ་ཡོངས་རྫོགས་ཀྱི་སེམས་ལ་བདེ་སྐྱིད་དང་རེ་བ་སྦྱིན་པ་རེད། ཕ་མས་ཀྱང་རང་གི་ཕྲུ་གུའི་རྣམ་དཔྱོད་དང་། གསར་གཏོད་ལ་སྤྲོ་བ་ཆེན་པོ་སྐྱེས། དེ་ནས་བཟུང་ཁོ་ཚོའི་ཁྱིམ་ཚང་ལ་ཡང་བསྐྱར་རྣམ་ཤེས་ཀྱི་གླུ་ལེན་པ་དེར་བརྟེན་ནས། འཚོ་བ་ཧ་ཅང་སྙན་འཇེབས་ལྡན་ཞིང་། སེམས་ཀྱི་ཞི་བདེ་དང་སྙིང་ཉེ་ཡོད་པ་ཞིག་ཏུ་གྱུར་སོ།། །
