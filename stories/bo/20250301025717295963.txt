རྫོང་ཁ་ཆེན་པོ་ཞིག་གི་མཐའ་མཚམས་སུ།  ཡངས་ཤིང་རྒྱ་ཆེ་བའི་རྩྭ་ཐང་ཞིག་ཡོད་དེ། དེའི་ནང་དུ་རྩི་ཤིང་སྣ་ཚོགས་དང་། རི་དྭགས་རྣམ་པར་བཀྲ་ཤེས་གསལ་བའི་བཀོད་པ་ལྟ་བུ་རྣམས་ཀྱིས་མཛེས་པའི་རྒྱན་གྱིས་བརྒྱན་ཡོད།  དེའི་ནང་དུ་གཞོན་ནུ་ཞིག་ཡོད་དེ།  སེམས་ཤིན་ཏུ་རྣོ་ཞིང་དྲང་བའི་ཡིད་ཀྱིས་དམིགས་བསལ་གྱི་སྤོབས་པ་ཡོད་པས་ གཡང་ཊེ་ཟེར་བའི་མིང་གིས་གྲགས་ཡོད།  གཡང་ཊེ་ནི་ཕོ་ཉ་ལྟ་བུའི་སེམས་དང་ལྡན་ཞིང་།  ཁོང་གི་སེམས་ལ་རང་ཉིད་ལ་ཡིད་ཆེས་གཞི་ཚུགས་ཤིང་།  སྙིང་རྗེ་ཆེན་པོ་ཡི་གཞི་རྩའི་སྟེང་ནས་འཇིག་རྟེན་ལ་གཟིགས་པའི་སྒྲིག་ལམ་གསལ་བ་ཞིག་ཡོད།


གཡང་ཊེ་ནི་ཉིན་ཞིག་རྩྭ་ཐང་དུ་འགྲོ་བའི་རིང་དུ་འུ་ཐུག་པའི་དོགས་པའི་སྟོན་པ་ཞིག་དང་འཕྲད་སོང་།  དོགས་པ་དེ་ནི་དྲི་མ་མེད་པའི་ཉི་མའི་དཀྱིལ་འཁོར་ལྟ་བུ་དྭངས་ཤིང་གཙང་བ་ཞིག་ཡིན་ནའང་།  སེམས་ལ་སྡུག་བསྔལ་ཆེན་པོ་ཞིག་གིས་བརྒྱན་ཡོད་པ་མངོན་གསལ་དུ་རེད།  དོགས་པས་གཡང་ཊེ་ལ་ཁོང་གི་རི་མོའི་རྨི་ལམ་ལ་སྐྱོ་བའི་ཁྱད་པར་གྱིས་ཉིན་རེ་བཞིན་མི་དར་སོང་བའི་སྡུག་བསྔལ་གྱི་སྒྲུབ་ཐབས་དང་།  སྤྱི་ཚོགས་ཀྱི་རྐྱེན་ངན་གྱིས་ཁྱིམ་ཚང་དུ་སྡུག་བསྔལ་གྱི་མིག་ཆུ་རྒྱུག་གི་ཡོད་པར་ངག་འབྲི་བྱས་པ་ན།  གཡང་ཊེ་ཡི་སྙིང་ལ་སྐྱོ་བའི་དྲོད་པས་ཁེངས་ཏེ་སེམས་ཤིང་སྐྱིད་ཀྱིས་ལྷག་པར་སེམས་པ་ཟད་གྲི་བྱུང་བ་རེད།


“དོགས་པ་བློ་ལས་ཕྱིར་དེང་སང་ཅི་ཞིག་བྱུང་ཡོད་དམ།”གཡང་ཊེས་དྲིས་པ་ན། “ང་ནི་ཆེ་ཐབས་ཀྱི་སྲོལ་རྒྱུན་གྱི་རླབས་ཆེན་བསླབ་བྱ་ཞིག་སློབ་མཁན་ཞིག་ཡིན།  འོན་ཀྱང་གྲོགས་པོ་གྲངས་མང་པོ་ཡིན་ནའང་མི་སེམས་ཀྱི་སྤྲོ་སེམས་རྩ་ནས་རྫོགས་ཟིན་སོང་།  ཁོང་ཚོར་རི་མོའི་རྨི་ལམ་གྱིས་སྡུག་བསྔལ་གྱིས་བཟུང་བའི་དཀྱིལ་དུ་མྱོང་གྲགས་ཡོད། ང་རང་སྡུག་སྔུན་ཡིན་ནའང་གནམ་ལ་ཡིད་ཆེས་མི་བྱེད།  ང་ཡིས་སྡུག་སྐྱོ་བ་མྱོང་མྱོང་ཡིན།” དོགས་པས་སྨྲས་སོང་།


གཡང་ཊེ་ཡི་སེམས་པ་སྔར་ལས་རབ་ཏུ་བསྒུལ་སོང་། ཁོང་གིས་དོགས་པའི་སྐད་ཆ་དེ་རྣམས་ལ་མཉན་ཞིང་སྙིང་སྟོབས་ཆེ་རུ་སོང་།  ཁོང་གིས་དོགས་པའི་སྡུག་བསྔལ་ལ་དམིགས་ཏེ་ ཁོང་གི་སེམས་ནས་བསམ་གཞིག་མང་པོ་གསལ་བཏོན་བྱས། ཁོང་གི་སྙིང་སྟོབས་དར་ཡོང་བ་དང་ཆབས་གྲུ་དང་བསམ་བློའི་གནད་ལ་ཐུག་སྟེ་གཡང་ཊེ་ཡི་སེམས་ལ་འཚེ་མེད་ཞི་བའི་ལམ་ལྟ་བུའི་རླབས་ཆེན་སྙིང་རྗེ་ཞིག་འབྱུང་བའི་རེ་ལྟོས་དེ་ཀྱིས།  ཁོང་གིས་དོགས་པའི་སྡུག་བསྔལ་སེལ་ཆེད་ལ་ཐབས་ཤེས་གསར་བ་ཞིག་བསམས་བྱུང་།


“སྡུག་བསྔལ་གྱི་མཐའ་མེད་པའི་རྒྱུད་ནས། རྩེ་ཕུན་མི་སེམས་ཀྱི་དཀྱིལ་གཞི་དེའི་ནང་ནས་ཞི་བའི་མེ་ཏོག་གིས་ཁེངས་ཡོད།  དེས་ན་ང་ཚོས་འཚེ་མེད་ཞི་བའི་ལམ་ལ་བརྩོན་འགྲུབ་བྱེད་ནུས་པ་རེད།  ཁྱོད་ཀྱི་སྡུག་བསྔལ་ལ་ཐག་གཅོད་ཤོར་ནས། དེའི་མཐའ་མེད་པའི་སྐྱོ་གསུམ་ལ་ཐུག་པར་བསླབ་བྱ་ཤིན་ཏུ་གལ་ཆེན་ཞིག་རེད།”


གཡང་ཊེས་སློབ་གསོའི་སྒྲོན་མའི་ཁུར་བ་བསྒུལ་སྟེ།  འཚེ་མེད་ཞི་བའི་ལམ་ལ་བསླབ་པའི་བླང་བྱ་ཡིས་ཁོང་གིས་འཇིག་རྟེན་གྱི་ཁྱོན་ལ་ཐམས་ཅད་ལ་སྙིང་རྗེ་ཆེན་པོའི་བསྐྱེད་རིམ་སྤེལ་ཏེ།  གཡང་ཊེ་དང་དོགས་པའི་སེམས་པ་སྟབས་བསྟུན་ནས་དཀའ་ངལ་ཡོངས་སུ་བསལ་སོང་།


དེའི་ཕྱོགས་མང་པོ་ཞིག་དཀའ་ངལ་གྲང་ལྷད་ཅན་ལ་མ་ཡེངས་སྟེ་རྟག་ཏུ་འཚེ་མེད་ཞི་བའི་ལམ་སློབ་གསོ་ལ་ཞུགས་ཡོད་དོ།
