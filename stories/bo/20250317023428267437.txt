བློ་གྲོས་འདྲེ་དང་བཅར་འདྲིར་ཕྱིན་ཏེ། འདྲེས་ཁོང་ལ་འདོད་དོན་འགའ་ཞིག་བསྒྲུབ་དགོས་པའི་བཀའ་བཏང་། འདྲེ་ཡིས་བཤད་དོན། "ང་ལ་ཁྱོད་ཀྱིས་བསྒྲུབ་དགོས་པའི་ལས་ཀ་ཞིག་ཡོད། ངའི་ཆེད་དུ་རིན་ཐང་ལྡན་པའི་ནོར་བུ་ཞིག་བཙལ་དགོས། དེ་ལས་ཀ་དེ་བསྒྲུབ་ཚར་ན་ཁྱོད་ཕྱིར་ལོག་ཆོག" བློ་གྲོས་ཀྱིས་ངུ་ཞོར་བཤད་དོན། "ང་ལ་དེ་ལྟ་བུའི་ནུས་པ་མི་འདུག" འདྲེས་བཤད་དོན། "ངས་ཁྱོད་ལ་འཐུས་སྒོ་ཚང་བའི་ནུས་པ་སྤྲད་ཆོག་། འོན་ཀྱང་ཁྱོད་ཀྱིས་དགོངས་པ་མ་འདྲེན་པར་ངའི་བཀའ་བསྒྲུབ་དགོས།"

བློ་གྲོས་ལ་བྱ་ཐབས་གཞན་མེད་པས། ཁོས་འདྲེའི་ཆ་རྐྱེན་ཁས་བླངས་ནས་འཚོལ་བའི་ལམ་དུ་བསྐྱོད། ཁོས་འདྲེ་ཡིས་བཤད་པའི་ནོར་བུ་རིན་ཐང་ཅན་དེ་རྙེད་དགོས་པའི་འཆར་གཞི་བཏིང་། ཁོའི་འགྲོ་ལམ་དུ་མཚན་མོ་དང་། དཀའ་སྡུག གྲང་རླུང་བཅས་མང་དུ་འཕྲད། ཡིན་ནའང་ཁོས་གོ་སྐབས་གཅིག་ཀྱང་མ་བརླག དུས་རྒྱུན་དུ་བརྩོན་འགྲུས་བྱས་ཏེ་མདུན་དུ་བསྐྱོད་དོ། མཐར་ཁོས་རི་མོའི་ཞིང་ཁམས་ཀྱི་མཚེའུ་ཆེན་པོ་ཞིག་གི་འགྲམ་དུ་སླེབས་པ་དེར་བསླེབས། མཚེའུ་དེའི་ནང་དུ་ནོར་བུ་རིན་ཐང་ཅན་ཞིག་འོད་འཚེར་བ་མཐོང་།

བློ་གྲོས་ཀྱིས་ནོར་བུ་དེ་བླངས་ཏེ་འདྲེའི་དྲུང་དུ་ཕྱིར་ལོག འདྲེས་དགའ་ཚོར་ཆེན་པོས་ནོར་བུ་དེ་ལེན་པ་དང་བློ་གྲོས་ལ་བཤད་དོན། "བཟང་། ད་ཁྱོད་རང་གི་ཕྱིར་ལོག་ཆོག" འོན་ཀྱང་བློ་གྲོས་ཕྱིར་ལོག་སྐབས་དྲན་པ་ཞིག་འཆར་བྱུང་། འདྲེས་སེམས་ཅན་དག་ལ་སྡུག་བསྔལ་སྦྱིན་བཞིན་ཡོད་པར་གོ་ནས་ཡིད་མ་བདེན་པ་ཞིག་བྱུང་། ཁོས་འདྲེའི་མདུན་དུ་སླེབས་ནས་དཔའ་ངར་ཆེན་པོས་བཤད་དོན། "ང་ཁྱོད་ཀྱིས་དགོས་ངེས་ཀྱི་ནོར་བུ་སྤྲད་ཡོད། འོན་ཀྱང་སེམས་ཅན་གྱི་སྡུག་བསྔལ་ལ་བསམ་བློ་བཏང་མེད། དེས་ན་ངས་ཁྱོད་ཀྱི་ནོར་བུ་སྤྲོད་ཀྱི་མིན།"

འདྲེ་ཧ་ཅང་ཁྲོ་ནས་བཤད་དོན། "ཁྱོད་རང་གིས་ང་ལ་སྐད་ཆ་འདི་ལྟ་བུ་ཞིག་བཤད་པར་དགའ། བློ་གྲོས། ཁྱོད་ང་ལ་ངོ་ལོག་བྱེད་དམ།" བློ་གྲོས་ཀྱིས་འདྲེའི་བཤད་ལ་འདི་ལྟར་བརྗོད་དོན། "ངོ་ལོག་བྱེད་ཀྱི་ཡིན། ངས་སེམས་ཅན་ལ་སྙིང་རྗེ་བྱེད་ཀྱི་ཡིན། ཁྱོད་ཀྱིས་བཙན་དབང་སྤྱད་ནས་སྡུག་བསྔལ་བསྐྱེད་ཀྱི་ཡོད། འདི་དོན་གནས་མིན།" དེའི་རྗེས་བློ་གྲོས་དང་འདྲེ་གཉིས་འཐབ་འཁྲུག་བྱེད་འགོ་བཙུགས། འཐབ་འཁྲུག་ཧ་ཅང་ཛ་དྲག་ཆེ་བ་ཞིག་ཡིན་མོད། བློ་གྲོས་ལ་གདེངས་ཚོད་ཡོད་པ་མ་ཟད། བདེན་དོན་དང་ཞི་བའི་ཆེད་དུ་འབད་བརྩོན་བྱས་པས་འདྲེ་ཕམ་པར་བཏང་། འདྲེ་ཡིས་འོ་དོད་སྒྲ་བཏོན་ནས་མེ་ཏོག་ཏུ་བསྒྱུར་སོང་།

འདྲེས་ཕམ་རྗེས། བློ་གྲོས་ནོར་བུ་དེ་སྤྱད་ནས་རི་མོའི་ཞིང་ཁམས་སུ་ཞི་བདེ་བསྐྱར་འཛུགས་བྱས། རི་མོའི་ནང་གི་སེམས་ཅན་ཚང་མར་སྤྲོ་སེམས་སྐྱེས། ཁོང་ཚོས་བློ་གྲོས་ལ་བསྔགས་བརྗོད་དང་བཀའ་དྲིན་ཆེ་ཞུས། དེ་རྗེས་བློ་གྲོས་འཕྲུལ་དེབ་ནང་ཕྱིར་ལོག་སྐབས། གྲུབ་ཆེན་རྒན་པོས་བཤད་དོན། "བུ་ཁྱོད་དེ་ལྟར་འགྱུར་བ་བྱུང་བར་ང་དགའ། ཁྱོད་ཀྱིས་བློ་བརྟན་བྱེད་པར་ནུས་པ་ཐོབ་སོང་། ཁྱོད་ཀྱི་སྙིང་སྟོབས་དེ་ལས་ཀྱང་ལེགས་པར་གྱུར་སོང་།" བློ་གྲོས་ཀྱིས་འཛུམ་དགོད་དང་བཅས་བཤད་དོན། "ངས་ཀྱང་སློབ་སྦྱོང་མང་པོ་བྱས་སོང་། དེ་ནི་སྙིང་རྗེའི་ཐོག་ནས་བྱུང་བ་ཞིག་ཡིན།"

དེའི་རྗེས་བློ་གྲོས་ཀྱིས་འཕྲུལ་དེབ་བེད་སྤྱོད་བྱས་ནས་རི་མོའི་ནང་དུ་འགྲོ་བ་དང་། མི་རྣམས་ལ་རོགས་རམ་བྱེད་པ་སོགས་ཀྱི་བྱ་བ་བྱེད་པར་བརྩོན་པར་བྱས་སོ། ཁོས་འདྲུལ་འདྲ་དང་གཏམ་རྒྱུད་འདྲ་མིན་མང་པོ་བྲིས། འཕྲུལ་དེབ་དེས་ནམ་ཡང་ཁོ་རང་གི་ཕྱོགས་སུ་བསྐྱོད་ཅིང་། བློ་གྲོས་ནི་རི་མོའི་ཞིང་ཁམས་ཀྱི་དཔའ་བོ་ཆེན་པོ་ཞིག་ཏུ་གྱུར་སོང་། ཁོང་གི་མིང་སྙན་པོ་དེ་དུས་ཡུན་རིང་པོར་འདས་སོ། །
