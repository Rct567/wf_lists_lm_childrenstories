ཚེ་རིང་ལ་བློ་སྤོབས་ཆེན་པོ་ཡོད་པའི་རྒྱུ་མཚན་གཙོ་བོ་ནི་ཁོང་གིས་དཀའ་ལས་ལ་འཛེམས་སྤྲང་མེད་པ་དང་། གཞན་ལ་རོགས་རམ་བྱེད་པར་དགའ་པོ་བྱེད་པ། བདེན་པ་ལ་བརྩེ་བ། ཡ་རབས་ཚུལ་མཐུན་གྱི་སྤྱོད་པ་སྤེལ་བ། མི་ཚེ་ལ་ཡིད་ཆེས་བྱེད་པ། དེ་མིན་སློབ་སྦྱོང་ལ་ཧུར་ཐག་བྱེད་པ་བཅས་ཡིན། ཁོང་ནི་ཕ་མས་གནང་བའི་སློབ་གསོར་བརྩི་བཀུར་དང་། ཡུལ་མི་ཚང་མར་བརྩེ་བས་སྐྱོང་མཁན་ཞིག་ཡིན།

ཉིན་ཞིག་ལ། ཚེ་རིང་གིས་རྒས་པའི་དགེ་རྒན་ཞིག་དང་མཇལ་འཕྲད་བྱས། དགེ་རྒན་གྱིས་ཚེ་རིང་ལ། ཁྱོད་རང་འདི་ལྟར་མི་ཡ་རབས་ཤིག་ཏུ་གྱུར་པར་ང་དགའ་པོ་བྱུང་། མི་ཚེ་ནི་རླབས་ཆེན་གྱི་རྒྱ་མཚོ་ཞིག་དང་འདྲ། མཚོ་ནང་གི་རླབས་རིས་ལ་ཞེད་སྣང་མེད་པར་མདུན་དུ་སྐྱོད་དགོས། བདེན་དོན་ལ་འབད་པ་དང་། དཀའ་ངལ་ལ་འཇིགས་སྣང་མ་བྱེད་། ཁྱོད་ཀྱི་མི་ཚེ་ལེགས་པར་བསྐྱེལ་ཐུབ། ཅེས་གསུངས།

ཚེ་རིང་གིས་དགེ་རྒན་གྱི་གསུང་ལ་ཉན་རྗེས། སེམས་ལ་གོ་བ་གསར་པ་ཞིག་རྙེད། ཁོས་བསམ་བློ་བཏང་སྟེ། རྒན་ལགས་ཀྱིས་གསུང་པ་ནང་བཞིན་དུ། མི་ཚེ་ནི་མུ་མཐའ་མེད་པའི་འགྲུལ་བཞུད་ཅིག་དང་འདྲ་བས། མདུན་དུ་སྐྱོད་དགོས་པ་དང་། དཀའ་ལས་ཁྱད་དུ་བསད་ནས་རང་གི་ཕུགས་བསམ་དོན་དུ་གཉེར་དགོས།

ཚེ་རིང་གིས་དགེ་རྒན་གྱི་གསུང་བསྒྲུབ་པའི་ཐོག་ནས་འཚོ་བ་རོལ་བ་དང་། གཞན་གྱི་བདེ་སྡུག་ལ་བསམ་བློ་གཏོང་བ། རང་ཉིད་ཡར་རྒྱས་གཏོང་བ་བཅས་བྱས། ཁོང་གིས་དཀའ་ལས་ཁག་པོ་ཇི་འདྲ་ཞིག་ཡིན་རུང་། བློ་སྤོབས་དང་སྙིང་སྟོབས་དང་བཅས་མདུན་དུ་བསྐྱོད་པ་དང་། དོན་དག་ཡག་པོ་སྒྲུབ་ཐུབ་ཀྱི་ཡོད། ཁོང་གིས་མི་ཚེའི་མཇུག་ཏུ། མི་རྣམས་ཀྱི་བརྩི་བཀུར་དང་བསྟོད་བསྔགས་བཅས་ཐོབ་པ་མ་ཟད། མི་ཚེ་ལེགས་པོ་ཞིག་བསྐྱལ་ཐུབ་པ་རེད། ཚེ་རིང་གི་མི་ཚེའི་ལོ་རྒྱུས་ནི་ཕྲན་བུ་དག་ལ་དཀའ་སྤྱད་ཁྱད་དུ་བསད་ནས་མདུན་དུ་སྐྱོད་དགོས་པའི་བློ་སྤོབས་ཀྱི་སྐོར་ཡིན།
