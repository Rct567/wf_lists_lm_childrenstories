རི་སྟག་གི་ཕྲུ་གུ་དྲག་བསྐྱེད་བྱུང་རྗེས་རི་སྟག་གིས་སྟག་འཚེར་ལ་ཐུགས་རྗེ་ཆེ་ཞུས་པ་མ་ཟད། ད་དུང་གྲོང་སྡེའི་མི་དམངས་ལ་ཐུགས་རྗེ་ཆེ་ཞུས། རི་སྟག་གི་མིག་ནང་དུ་གནོད་འཚེའི་ཚབ་ཏུ་བརྩེ་བ་དང་བདེ་སྐྱིད་ཀྱི་འོད་ཟེར་འཕྲོས། གངས་ལྗོངས་གྲོང་སྡེ་ཡང་ཡ་མཚན་ཆེན་པོའི་ངང་ནས་དེར་བསྡད་པ་རེད། དེ་ནི་ཕན་ཚུན་ལ་བརྩེ་བ་དང་བྱམས་སེམས་སྟོན་པ་ལ་བརྟེན་ནས་ཡོང་བའི་འབྲས་བུ་ཡིན་པ་ཚང་མས་ཤེས་ཡོད། གྲོང་སྡེའི་མི་ཚོས་སྟག་འཚེར་ལ་སྤྲོ་སེམས་འཁོལ་བའི་ངང་བསྟོད་བསྔགས་བྱས།

དེ་རྗེས་སུ། སྟག་འཚེར་གྱིས་གྲོང་སྡེའི་མི་རྣམས་ལ་སློབ་གསོ་སྤྲད། ཁོས་ནམ་རྒྱུན་བརྗོད་དོན། ང་ཚོ་མི་ཚང་མ་ཕན་ཚུན་ལ་བྱམས་སེམས་དང་བརྩེ་བ་སྟོན་པར་འབད་བརྩོན་བྱེད་དགོས། འཚེ་བ་ནི་མི་ཚེ་འདིའི་ཆ་ཤས་མིན་པ་མ་ཟད། ཞི་བའི་འཚོ་བ་སྐྱེལ་ཐབས་བྱེད་དགོས། སྟག་འཚེར་གྱིས་བཤད་པའི་གཏམ་དེ་དག་གྲོང་སྡེ་ཡོངས་ལ་ཁྱབ་སོང་། མི་རྣམས་ཀྱིས་དེར་ཉན་ཅིང་རང་ཉིད་ཀྱི་འཚོ་བར་དེའི་ཤུགས་རྐྱེན་ཐེབས། གྲོང་སྡེ་དེར་བརྩེ་བ་དང་ཞི་བའི་རྣམ་པ་ཞིག་འབྱུང་བར་གྱུར་ཏོ། གྲོང་སྡེ་དེར་འཚེ་བ་དང་རྩོད་གླེང་གང་ཡང་བྱུང་མ་སོང་།

སྟག་འཚེར་ཡང་གངས་ལྗོངས་ཀྱི་མི་ཚང་མའི་སེམས་ལ་བརྩེ་བས་ཁེངས་པ་དང་སྤྲོ་སྐྱིད་ཀྱིས་ཁེངས་ཏེ་གནས་སོང་། དེའི་བྱས་རྗེས་དེ་ཡང་མི་རབས་ནས་མི་རབས་བར་དུ་བརྒྱུད་སོང་། སྟག་འཚེར་ལ་དྲན་གསོ་སློབ་གསོ་སྤྲད་པའི་ཕ་མ་རྣམས་ནས་བཤད་དོན། ཁྱེད་ཚོས་ནམ་ཡང་འཇིག་རྟེན་འདིར་བརྩེ་བ་སྟོན་དགོས། བྱམས་སེམས་ཀྱིས་ཁྱེད་ཚོར་བདེ་སྐྱིད་དང་སྐྱིད་པོ་ཡོང་བར་བྱེད་ཅེས་བཤད་ཡོད་དོ།

ཡུན་རིང་ཞིག་གི་རྗེས་སུ། གངས་ལྗོངས་ཀྱི་བྱིས་པ་རྣམས་ཀྱིས་སྟག་འཚེར་དྲན་གསོ་བྱེད་པའི་ཚུལ་དུ་རང་ཉིད་ཀྱི་མིང་ལ་ལྷང་དཀར་ཞེས་བཏགས། ལྷང་དཀར་ནི་མཚན་མོ་སྐར་ཚོགས་འོད་ཆེམ་པ་ལྟར་ཡིན། དེའི་གཏམ་རྒྱུད་དེ་ནི་མི་རྣམས་ཀྱི་སེམས་ནང་དུ་བརྩེ་བའི་སེམས་པ་སློང་བའི་རྒྱུད་སྐུལ་ཞིག་ཏུ་གྱུར་སོང་། ལྷང་དཀར་གྱི་གཏམ་རྒྱུད་དེས། ང་ཚོ་ཚང་མར་རང་ཉིད་ཀྱི་ནུས་པ་གང་ལེགས་སྤྱད་དེ་འཇིག་རྟེན་འདི་ཡུལ་བཟང་ཞིག་ཏུ་བསྒྱུར་ཐུབ་པའི་བློ་སྤོབས་སྐྱེས།
