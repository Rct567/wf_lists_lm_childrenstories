རྟ་ཕྱུགས་སྐྱིད་པོའི་སེམས་ལ་ཡིད་ཚིམ་པ་ཆེན་པོ་སྐྱེས། ཁོས་འདི་ལྟར་བཤད།

ལུག་གིས་དེའི་རྗེས་སུ་འདི་ལྟར་སྨྲས། “ངའི་བསམ་པར་ཁོང་ནི་ང་ཚོར་གདམས་ངག་གནང་བར་ཡོང་གི་ཡོད། དེ་བས་ང་ཚོས་ད་ལྟ་ཁོང་དང་མཇལ་འཕྲད་དུ་འགྲོ་དགོས།”

སྐྱིད་པོས་འདི་ལྟར་བཤད། “ངེད་ཅག་འགྲོ། དེ་ལྟར་ན་ད་ལྟ་ཆུ་མཚམས་སུ་འགྲོ་རྒྱུ་ནི་ཕྱི་བསྐྱུངས་ན་ལེགས་པོ་འདུག ཁོང་ནི་དགུང་ལོ་མཐོན་པོ་ཡིན་པས་བསྒུགས་ཡོད།”

ཁོ་ཚོ་མྱུར་དུ་འབབ་ཚུགས་ཁང་གི་ཕྱོགས་སུ་བསྐྱོད་ཅིང་། གང་དུ་ནོར་འཛིན་བླ་མ་བཞུགས་ཡོད། སྐབས་དེར་བླ་མ་ནི་སྤང་ཐང་གི་སྟེང་དུ་བཞུགས་ནས་སྨོན་ལམ་འདེབས་ཀྱི་ཡོད། རྟ་ཕྱུགས་སྐྱིད་པོ་དང་ཁོའི་གྲོགས་པོ་རྣམས་འབྱོར་བ་དང་བླ་མས་ཁོ་ཚོར་འཛུམ་དཀར་བསྟན།

སྐྱིད་པོས་གུས་ཕྱག་ཞུས་རྗེས། “བླ་མ་ལགས། ང་ཚོས་ཁྱེད་རང་ལ་མཇལ་འཕྲད་དུ་ཡོང་བ་ཡིན། ཁྱེད་རང་གིས་ང་ཚོར་ཐུགས་རྗེས་ཆེན་པོས་བསླབ་བྱ་གནང་གི་ཡོད་རེད། དེ་བས་ང་ཚོས་རྟག་ཏུ་ཁྱེད་རང་གི་བཀའ་སློབ་ཉན་འདོད་ཡོད།”

སྐྱིད་པོས་དེ་གོ་བས། ཁོ་ཧ་ཅང་སེམས་ཁྲལ་བྱུང་། ཁོས་འདི་ལྟར་བཤད།

གླང་ཆེན་གྱིས་དེར་ཉན་ཞིང་ཁ་སེམས་མ་བཟོད་པར་འདི་ལྟར་བཤད།

གླང་ཆེན། “བདེན་པ་རེད། ང་ཚོས་ཚང་མར་བྱམས་སེམས་འཛིན་དགོས། ང་ཚོས་འགྲོ་བ་མིའི་རིགས་ལ་བརྩེ་བ་དང་བསམ་ཤེས་བྱེད་དགོས།”

ལུག་གིས་དེར་རོགས་རམ་བྱེད་བཞིན་འདི་ལྟར་སྨྲས།

ལུག “ང་ཚོས་གཞན་ལ་རོགས་རམ་བྱེད་དགོས། མི་དམངས་སྡུག་བསྔལ་གྱི་ནང་དུ་འཚོ་བ་སྐྱེལ་དུས་། ང་ཚོས་ཁོ་ཚོར་རོགས་བྱེད་དགོས། དེ་ནི་ཧ་ཅང་གལ་ཆེན་རེད།”

བླ་མས་ཁོ་ཚོར་སེམས་ཤུགས་སྤྲད་དེ་འདི་ལྟར་གསུངས།

བླ་མ། “ལེགས་སོ། ངའི་བུ་ཚོ། ངས་ཁྱེད་ཚོར་ཐུགས་རྗེ་ཆེ་ཞུ་བ་ཡིན། ཁྱེད་ཚོས་དྲང་མོའི་སྐད་ཆ་ཤོད་ཀྱི་འདུག ང་ཚོའི་འཚོ་བའི་ནང་ལ་གནས་ཚུལ་གཞན་ཞིག་ཀྱང་ཡོད་པ་ནི་ང་ཚོས་སེམས་ཤུགས་བསྐྱེད་དེ་མདུན་དུ་སྐྱོད་དགོས། གལ་ཏེ་ང་ཚོ་འཇིགས་སྣང་དང་དཀའ་ངལ་དང་འཕྲད་ན། ང་ཚོས་གཞི་ནས་བློ་རྩེ་གཅིག་སྒྲིལ་གྱིས་ཐག་གཅོད་བྱེད་དགོས། དེ་ནི་ང་ཚོའི་མི་ཚེ་སྐྱིད་པོ་ཡོང་བར་ཧ་ཅང་གལ་ཆེན་རེད།”

རྟ་ཕྱུགས་སྐྱིད་པོའི་མིག་ནང་གནམ་གྱི་སྐར་ཚོགས་འདྲ་བའི་འོད་ཟེར་ཞིག་འཕྲོས། ཁོས་སེམས་གསལ་པོར་བྱེད་དགོས་པ་ཤེས། ཁོས་བླ་མ་ལ་འདི་ལྟར་བཤད།

སྐྱིད་པོ། “བླ་མ་ལགས། ཁྱེད་ཀྱི་གསུང་དེ་དག་ངས་ནམ་ཡང་བརྗེད་མི་སྲིད། ང་ཚོས་གང་འདྲ་བྱེད་དགོས་པ་ངས་ཤེས།”

བླ་མས་ཁོ་ཚོར་འདི་ལྟར་བརྗོད།

བླ་མ། “ལེགས་སོ། ངའི་བུ་ཚོ། ཡང་ན་གཞན་ཡང་ཤོད་རྒྱུ་ཡོད། བློ་སྤོབས་དང་སྙིང་རྗེ། རང་ཉིད་ལ་ཡིད་ཆེས། དེ་དག་རྟག་ཏུ་བསྲུངས་དགོས། ཁྱེད་ཚོས་ཉིན་རྟག་ཏུ་བརྩོན་འགྲུས་བྱེད་དགོས། ང་དེ་རིང་ཁྱེད་ཚོར་དགའ།”

དེ་ནས་ཉི་མ་ནུབ་ལ་ཡར་མར་འགུལ་བཞིན་པའི་དུས་ཚོད། བླ་མས་ཁོ་ཚོར་བཀྲ་ཤིས་བདེ་ལེགས་ཞུས་ཏེ་རང་གནས་སུ་ཕེབས། རྟ་ཕྱུགས་སྐྱིད་པོ་དང་ཁོའི་གྲོགས་པོ་རྣམས་ཀྱང་སེམས་ལ་སྤྲོ་སྣང་གིས་ཁེངས་ནས་ཕྱིར་ལོག ཁོ་ཚོས་ནོར་འཛིན་བླ་མའི་གསུང་ཇི་ལྟར་བཞིན་སྤྱོད་ལམ་བཟང་པོ་བསྟན་ཅིང་གཞན་ལ་བརྩེ་བ་བྱེད་རྒྱུའི་ཁས་བླངས།

སྐྱིད་པོ། “ང་ཚོས་ད་ཆུ་མཚམས་ལ་མ་ཕྱིན་ཀྱང་རེད། བླ་མའི་གསུང་གིས་ངའི་སེམས་ཁོག་བདེ་བར་བྱས་སོང་།”

གླང་ཆེན། “ང་ཡང་སྐྱིད་པོ་འདུག ངེད་ཅག་ཚང་མ་གྲོགས་པོ་ཡིན་པས་། དེ་བས་མཉམ་དུ་གནས་ན་སྤྲོ་བ་ཡོང་གི་རེད།”

ལུག “ང་རང་ཉིད་མུ་མཐུད་དུ་གཞན་ལ་བརྩེ་བ་བྱེད་པར་ཡིད་ཆེས་ཡོད། དེ་ནི་ཧ་ཅང་ཡག་པོ་རེད།”

དེ་ནས་ཁོ་ཚོས་མཉམ་དུ་སྤང་ཐང་དུ་རྩེད་མོ་མུ་མཐུད་དེ་རྩེས་པ་རེད། ཁོ་ཚོས་འདི་ལྟར་ཤེས་བྱུང་། སྤྲོ་སྐྱིད་ལྡན་པའི་འཚོ་བ་སྐྱེལ་བ་ནི་ལས་སླ་པོ་ཞིག་རེད། འོན་ཀྱང་མི་ཚེ་སྐྱིད་པོ་ཞིག་ཡོང་བར་བྱམས་སེམས་དང་བློ་སྤོབས། ཡིད་ཆེས་བཅས་ཡོད་དགོས་པ་ངེས་ཤེས་བྱུང་བ་རེད།

བླ་མས་རྟ་ཕྱུགས་སྐྱིད་པོ་དང་ཁོའི་གྲོགས་པོ་རྣམས་ལ་རྟག་ཏུ་བྱམས་བརྩེ་དང་སྙིང་རྗེ་བསླབས། དེ་དག་ནི་འཚོ་བ་སྐྱིད་པོ་ཡོང་བར་གལ་ཆེན་པོ་ཡིན་པ་ཡང་ཡང་བཤད་ཡོད།

ཉི་མ་ནུབ་རྗེས་སྐར་ཚོགས་འོད་དང་ལྡན་པའི་ནམ་མཁའ་འོག་ཏུ་རྟ་ཕྱུགས་སྐྱིད་པོ་དང་ཁོའི་གྲོགས་པོ་རྣམས་མཉམ་དུ་དགོད་སྒྲས་ཁེངས། ཁོ་ཚོའི་སེམས་ནང་ལ་ནམ་ཡང་འགྱུར་བ་མེད་པའི་སྤྲོ་སྣང་དང་བདེ་སྐྱིད་ཡོད་ཅིང་། གཏམ་རྒྱུད་འདི་ཡང་དེ་ལྟར་མཇུག་རྫོགས་སོ། །
