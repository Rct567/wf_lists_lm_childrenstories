ཁོ་ཚོས་ཁོ་ལ་ཡང་ཁ་ཡོད་ལག་ཡོད་ངང་དྲི་བ་འདྲི་འགོ་བརྩམས་བྱུང་། འབྲུག་སྒྲས“ཁྱེད་ཚོའི་ལུས་པོ་བདེ་པོ་ཡིན་ནམ། སེམས་ལ་དཀའ་ངལ་འདུག་གམ།”ཞེས་དྲིས་པར། ཁོ་ཚོས་ཀྱང་རང་ཉིད་ཀྱི་དཀའ་ངལ་མང་པོ་བཤད། འོན་ཀྱང་འབྲུག་སྒྲའི་གཏམ་གྱིས་ཁོ་ཚོའི་སེམས་གསོས་པ་དང་། ཁོ་ཚོར་བདེ་སྐྱིད་ཀྱི་ཚོར་བ་བསྐྲུན། འབྲུག་སྒྲ་ལ་མཚོན་ན་མི་ལ་གནོད་སྐྱེལ་མི་སྲིད། སྤྱིར་བཏང་དུ་ཁོ་ནི་ཡ་རབས་ཤེས་ཡོན་ཅན་གྱི་བྱིས་པ་ཞིག་ཡིན། དེས་ན་དེང་གི་དུས་སུའང་ཕ་མས་ཁོ་ལ་རྒྱབ་སྐྱོར་བྱེད་པ་མ་ཟད། བྱིས་པ་ཚོས་ཀྱང་ཁོ་ལ་དགའ་པོ་བྱེད།

ཉིན་གཅིག་། འབྲུག་སྒྲ་རང་ཉིད་ཀྱི་ཨ་ཕའི་ལས་ཁུངས་སུ་སོང་། དེར་འབྲུག་སྒྲ་ཡིས་ཁོའི་ཨ་ཕས་རི་མོ་མང་པོ་བྲིས་པ་མཐོང་། དེའི་ཁྲོད་དུ་རི་མོ་ཞིག་ཡོད་པ་ནི་སྤྲིན་དཀར་པོའི་ཁྲོད་ནས་ཐོན་པའི་འབྲུག་ཅིག་རེད། དེ་ནི་འབྲུག་སྒྲའི་མཚན་མ་དང་གཏམ་རྒྱུད་ཀྱི་འགོ་འཛུགས་ཡིན། ཨ་ཕས་འབྲུག་སྒྲ་ལ་བཤད་རྒྱུར། “འབྲུག་ནི་བོད་ཀྱི་རིག་གཞུང་ནང་གི་མཚོན་རྟགས་གལ་ཆེན་ཞིག་རེད། འབྲུག་ནི་སྟོབས་ཤུགས་དང་། ཤེས་རབ། བདེ་སྐྱིད་བཅས་ཀྱི་མཚོན་དོན་རེད། ཁྱོད་རང་འབྲུག་གི་རྗེས་འབྲངས་པ་ཞིག་ཡིན་པས། ཁྱོད་ཀྱིས་འབྲུག་གི་བསམ་བློ་དང་ལམ་ལུགས་མུ་མཐུད་དུ་སྦྱང་དགོས། འབྲུག་གིས་བདེན་པ་དང་། བླ་ན་མེད་པའི་བརྩེ་བ་ལ་དགའ་པོ་བྱེད།” ཅེས་བཤད།

འབྲུག་སྒྲས་ཨ་ཕའི་བཀའ་མངའ་བཞིན་གནང་བ་རེད། ཁོས་འབྲུག་སྐོར་ལ་སློབ་སྦྱོང་དང་། འབྲུག་གི་རྩེད་མོ་རྩེས། གནམ་གཤིས་དང་རང་བྱུང་ཁམས་ལ་སེམས་ཁུར་བྱེད། རིམ་བཞིན་འབྲུག་སྒྲས་རང་ཉིད་ལ་མི་ཚེའི་དམིགས་ཡུལ་ཞིག་རྙེད་པ་རེད། དེ་ནི་རང་གི་འཇོན་ཐང་གིས་མི་ཚང་མར་རོགས་བྱེད་རྒྱུ་དེ་རེད།

གནམ་ལོ་གསར་པ་འཆར་ལ་ཉེ། འབྲུག་སྒྲ་དགའ་སྤྲོ་ཆེན་པོས་ལམ་བར་དུ་རོགས་རམ་བྱེད་བཞིན་པའི་བྱིས་པ་ཚོ་མཐོང་བར་བསྐྱོད་དེ། ཁོང་ཚོའི་དྲན་ཤེས་ལ་ཕན་པ་དང་གནོད་མེད་ཀྱི་སྒོ་ནས་སེམས་གསོ་བྱེད། འདི་ནི་ཁོའི་འགན་འཁྲི་གསར་པ་རེད། འབྲུག་སྒྲས་རང་ཉིད་ཀྱི་གནས་ཚུལ་ཚང་མ་དང་གནས་སྐབས་ཀྱི་འཁྱེར་སོ་སྦྱར་བས། འབྲུག་སྒྲ་ཡིས་སྤྱིར་བཏང་བོད་ལ་མཐོང་སྣང་གསར་པ་ཞིག་བསྐྲུན་ཐུབ་ཡོད། འབྲུག་སྒྲ་འཛུམ་དང “ངས་མི་ཚང་མར་རོགས་བྱེད་དགོས། གནམ་ལོ་གསར་པར་ང་ཚོས་སེམས་ཀྱི་བདེ་སྐྱིད་བསྐྲུན་ཆོག” ཅེས་བརྗོད།

འབྲུག་སྒྲའི་གཏམ་རྒྱུད་ནི་འདི་ལྟར་མཇུག་རྫོགས་སོ། །ཡིན་ན་ཡང་འབྲུག་སྒྲའི་འཚོ་བའི་ནང་གཏམ་རྒྱུད་གསར་པ་དུ་མ་ད་དུང་ཡོད་ངེས་རེད། མི་རེའི་སེམས་ནང་གི་འབྲུག་རེ་རེ་ཡིས་འཛམ་གླིང་ཡོངས་ལ་བདེ་སྐྱིད་དང་བརྩེ་བའི་འོད་སྣང་འཕྲོ་བར་སྨོན། བཀྲ་ཤིས་བདེ་ལེགས།
