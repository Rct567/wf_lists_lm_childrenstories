རྩྭ་ཐང་ཡངས་པོ་ཞིག་གི་ནང་། གཡག་རྒོད་ཅིག་དང་། བུ་ཆུང་བཀྲ་ཤིས་ཟེར་བ་ཞིག་འདུག །

བཀྲ་ཤིས། “ཨ་ཕ། ང་ད་དུང་ཡང་དགེ་རྒན་མེད་པའི་འཛིན་གྲྭར་འགྲོ་དགོས་བྱུང་།”

གཡག་རྒོད། “ཨ་ཁ། བཀྲ་ཤིས་ལགས། ད་ལྟ་སེམས་སྡུག་བྱེད་མ་དགོས། ཁྱེད་ཀྱི་དགེ་རྒན་ནི་ང་ཡིན་། ང་ཁྱེད་རང་གི་འགྲོགས་པོ་ཡིན།”

བཀྲ་ཤིས། “ངས་ཤེས་ཀྱི་ཡོད། འོན་ཀྱང་། ང་རྩེད་མོ་རྩེ་ཆེན་པོ་དེ་འདྲ་ཞིག་རྩེ་མི་ཐུབ་”

གཡག་རྒོད། “རྩེད་མོ་ནི་དགའ་རོགས་གང་མང་དང་མཉམ་དུ་རྩེ་བ་དང་དེ་བཞིན་ཤེས་བྱ་གསར་པ་སློབ་སྦྱོང་བྱེད་རྒྱུ་ཞིག་ཡིན། ཡོང་། ང་ཚོ་ཡུལ་དུ་སྐྱོད་ནས་བྱ་དགའ་བཙལ་ན་ཅི་འདྲ་རེད།”

བཀྲ་ཤིས། “བྱ་དགའ་? ངོ་མ་རེད་པས།”

བཀྲ་ཤིས། “ཡག་པོ་འདུག། ཡིན་ནའང་ང་སོང་ན་ག་རེ་བྱེད་ཀྱི་ཡོད།”

གཡག་རྒོད། “ང་ཚོས་འགྲུལ་བཞུད་བྱ་དགའ་རྩེ་ཆེན་པོ་ཞིག་གི་དོན་དུ་བྱེད་དགོས། དང་ཐོག་ཏུ་ཤར་ཕྱོགས་སུ་འགྲོ་དགོས། དེར་འབྲོག་གི་ཚོགས་སྤྱིར་མཇལ་འཕྲད་བྱེད་དགོས།”

བཀྲ་ཤིས། “འབྲོག་གི་ཚོགས་སྤྱི། མོས་དགོས་གསུངས་སམ། ང་མ་ཤེས།”

གཡག་རྒོད། “མོ་དགའ། ཁོ་ཚོ་ཤིན་ཏུ་དགའ་སྤྲོ་ཡོད་ཅིང་། ཁོང་ཚོས་མཐོང་བ་ཙམ་གྱིས་འཛུམ་མུག་བྱེད་ངེས།”

བཀྲ་ཤིས། “ང་དེར་སྤྲོ་པོ་བྱེད་ཀྱི་ཡོད།”

དེ་ནས་གཡག་རྒོད་དང་བཀྲ་ཤིས་གཉིས་ཤར་ཕྱོགས་སུ་བསྐྱོད། ཉིན་འགའ་ཤས་ཀྱི་རྗེས་སུ་ཁོང་ཚོ་འབྲོག་གི་ཚོགས་སྤྱིར་སླེབས་པ་རེད།

བཀྲ་ཤིས། “གཟིགས་དང་། ཚོགས་སྤྱི་གང་འདྲ་རེད།”

གཡག་རྒོད། “ཧྲིན། ཡག་པོ་འདུག ཡོང་། ང་ཚོས་བརྗོད་ལབ་འགའ་ཤས་ཞུ།”

གཡག་རྒོད་དང་བཀྲ་ཤིས་གཉིས་འབྲོག་གི་ཚོགས་སྤྱིའི་འཁྲིས་སུ་ཕྱིན།

གཡག་རྒོད། “འབྲོག་ཆེན་མོ། ང་ཚོ་གཡག་རྒོད་དང་བཀྲ་ཤིས་ཡིན། ཁྱེད་རང་ལ་མཇལ་འཕྲད་ཞུ་བར་ཡོང་བ་ཡིན།”

འབྲོག་གི་ཚོགས་སྤྱིས། “ཨོ། གཡག་རྒོད་རྒན་པོ། བཀྲ་ཤིས་ལགས། ཁྱེད་རང་གཉིས་ལ་འདི་རུ་བསུ་བར་དགའ་པོ་ཡོད།”

བཀྲ་ཤིས། “ཐུགས་རྗེ་ཆེ།”

འབྲོག་གི་ཚོགས་སྤྱིས། “འདི་ག་རེ་བྱས་ནས་ཡོང་བ་ཡིན།”

གཡག་རྒོད། “ང་ཚོ་འགྲུལ་བཞུད་བྱེད་ཀྱི་ཡོད། ང་ཚོ་སྙན་ངག་རྩོམ་མཁན་ཞིག་ལ་འཕྲད་པར་འགྲོ་བཞིན་ཡོད། ཁོང་ལ་མཇལ་འཕྲད་ཞུ་བ་ཡིན།”

འབྲོག་གི་ཚོགས་སྤྱིས། “ཧ་ཅང་ཡག་པོ་འདུག། ང་ཚོ་ལ་ཤེས་བྱ་གང་མང་ཡོད་། ངས་ཁྱེད་ཚོ་ལ་བྱ་དགའ་ཅིག་སྤྲོད་ཆོག། ང་ཚོ་ལ་ཤེས་བྱ་ཆེན་པོ་ཡོད། ཁྱེད་རང་གིས་གསན་ན་ཅི་འདྲ་རེད།”

བཀྲ་ཤིས། “གསན་ན་འདོད།”

འབྲོག་གི་ཚོགས་སྤྱིས། “ང་རང་ཚོས་ཁུ་ལུག་གི་གཏམ་རྒྱུད་གསུངས་ཀྱི་ཡོད། ཁོང་ཚོས་བུ་དང་སྤྲུལ་ཕྲུག་དུ་བརྗོད་པའི་སྐོར་ཡིན།”

བཀྲ་ཤིས་དང༌གཡག་རྒོད་རྣམས་ཡ་མཚན་ཆེན་པོས་ཉན་བསྡད་པ་རེད།

གཡག་རྒོད། “ངོ་མཚར་རེད། ཁྱེད་ཀྱི་ལོ་རྒྱུས་ངོ་མ་ལེགས་པོ་འདུག།”

འབྲོག་གི་ཚོགས་སྤྱིས། “ད་དུང་ཡང་མུ་མཐུད་གསུངས་ཆོག། ང་ཚོ་ནུབ་ཕྱོགས་སུ་སྐྱོད་པ་ཡིན། དེར་ཆུ་མིག་གི་རིགས་བཞག་སྟེ་བྱ་དགའ་བསྡུ་དགོས།”

བཀྲ་ཤིས། “རྗེས་སུ་འབྲང་།”

དེ་རྗེས་གཡག་རྒོད་དང༌། བཀྲ་ཤིས་གཉིས་ནུབ་ཕྱོགས་སུ་སྐྱོད་པ་རེད། དཀའ་ལས་ཁག་པོ་འདུག་ཀྱང་། ཁོང་གཉིས་ཀྱི་སེམས་ལ་འཆམ་མཐུན་དང༌། བྱ་དགའ་བསྡུ་བའི་བསམ་བློ་ཡོད་པས་དཀའ་ངལ་མེད་པར་ཕྱིན་ཐུབ།

བཀྲ་ཤིས། “གཡག་རྒོད། མཐོང་སོང་ངམ། ང་ཚོའི་ཆུ་མིག་གང་དུ་བསླེབས།”

གཡག་རྒོད། “ང་མཐོང་། འདིའི་གནས་ནི་བྲག་ཆེན་པོའི་འཁྲིས་སུ་ཡོད་། དེ་རུ་སྨན་པ་མཁས་པས་སེམས་ལ་གསོན་ཤུགས་སྦྱིན་པའི་གཏམ་བཤད་དགོས་།”

བཀྲ་ཤིས། “ངས་སེམས་ལ་འགྱུར་བ་གཏོང་དགོས་ཀྱི་རེད་དམ།”

གཡག་རྒོད། “དེ་ལྟར་མིན་པར་འདུག། སྨན་པས་ཁྱེད་ལ་གདམས་པ་གསུངས་གསོན།”

ཁོང་གཉིས་བྲག་འོག་ཏུ་སླེབས་ནས། སྨན་པ་མཁས་པ་མཇལ་ཡོད།

བཀྲ་ཤིས། “སྨན་པ། ཁྱེད་རང་གིས་ང་ཚོའི་བྱ་དགའ་བཀྲལ་ཐུབ་བམ།”

སྨན་པས། “བདེན་པ་ཡིན། ངའི་ཆུ་མིག་ཆེ་རུ་གཏོང་གི་ཡིན། ངའི་ཤེས་བྱ་རྣམས་ཁྱེད་ལ་སྤྲོད་ཀྱི་ཡོད། ངས་སྙན་ངག་གི་གཏམ་ཡང་མང་པོ་བཤད་ཀྱི་ཡོད། ང་ཚོ་ནས་གཏམ་བཤད་བྱས་རྗེས་བྱ་དགའ་བླངས་ཆོག།”

བཀྲ་ཤིས་སེམས་དགའ་བས་ཁེངས་ནས། སྨན་པའི་གསུང་ལ་ཉན་སྡད་པར་གྱུར།

བཀྲ་ཤིས། “ཐུགས་རྗེ་ཆེ། སྨན་པ། ཁྱེད་རང་གི་གཏམ་དེས་ང་ལ་བདེ་སྐྱིད་སྤྲད།”

སྨན་པས། “ཁྱེད་རང་ལ་དགའ། བཀྲ་ཤིས། བྱ་དགའ་འདིའི་ཐོག་ནས་སློབ་སྦྱོང་མང་པོ་བྱ་ཆོག།”

དེ་རྗེས་ཁོང་གཉིས་ལྷོ་ཕྱོགས་སུ་ཕྱིན་ཏེ། དགེ་རྒན་ཐོགས་མཁས་དང་འཕྲད།

གཡག་རྒོད། “དགེ་རྒན་ལགས། ང་ཚོའི་འགྲུལ་བཞུད་ཀྱི་རིན་ཐང་མང་པོ་ཐོབ། དེའི་རྒྱུ་མཚན་གྱིས་ཁྱེད་ཀྱིས་བྱ་དགའ་འགའ་ཤས་སྤྲོད་རོགས།”

དགེ་རྒན་ཐོགས་མཁས་ཀྱིས། “བཀྲ་ཤིས་ལགས། ངས་ཤེས་ཡོད། ཁྱེད་ལ་བྱ་དགའ་སྤྲད་ཆོག། གཅིག་ནི་སྙིང་རྗེ། གཉིས་པ་བདེན་པ། གསུམ་པ་མཛའ་བརྩེ་བཅས་ཡིན། འདིས་ཁྱེད་ཀྱི་མི་ཚེར་གཅེས་པའི་ནུས་པ་ཐོབ།”

བཀྲ་ཤིས། “ཐུགས་རྗེ་ཆེ། དགེ་རྒན། ཁྱེད་རང་གིས་ང་ལ་སློབ་གསོ་གནང་བར་སྙིང་དུ་རང་སེམས་ཡོད།”

དེ་ནས་གཡག་རྒོད་དང༌བཀྲ་ཤིས་གཉིས་སྤྲོ་སེམས་འཁོལ་པོའི་ངང་ཕྱིར་ཕ་ཡུལ་དུ་ལོག །

བཀྲ་ཤིས། “གཡག་རྒོད། ང་ཚོས་འགྲུལ་བཞུད་འདི་ནི་དྲན་གསོ་བྱེད་ངེས་རེད།”

གཡག་རྒོད། “རེད། ང་ནི་ཁྱེད་རང་ལ་ཧ་ཅང་དགའ། ཁྱེད་ནི་སྙན་ངག་ལ་དགའ་ཞིང་། དྲན་ཤེས་ཡོད།”

བཀྲ་ཤིས། “ང་ཡང་ཁྱེད་རང་ལ་དགའ། ཁྱེད་ནི་ངའི་མཛའ་བརྩེའི་གྲོགས་པོ་ཡིན། ཁྱེད་ཀྱིས་ངའི་མི་ཚེར་བྱ་དགའ་མང་པོ་བསྣན་སོང་།”

གཡག་རྒོད། “དེ་ནི་བྱ་དགའ་མང་པོ་སྤྲོད་པའི་གཏམ་རྒྱུད་མ་རེད། འོན་ཀྱང་ང་ཚོས་མཉམ་དུ་དེ་འདྲའི་ཡུལ་ལུང་རང་མཐོང་བྱས་པ་རེད།”

བཀྲ་ཤིས། “ད་ནི་ངས་ངའི་གྲོགས་པོ་ཚོ་ལ་ངའི་བྱ་དགའ་མང་པོ་གསུངས་ཆོག།”

བཀྲ་ཤིས་དང༌། གཡག་རྒོད་རྣམས་ཕན་ཚུན་ལ་འཛུམ་བཞིན་ཕྱིར་ཁོང་གཉིས་ཀྱི་ཁྱིམ་ལ་ལོག་སོང་།
