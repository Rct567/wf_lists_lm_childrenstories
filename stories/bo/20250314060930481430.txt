ཨ་ཕས་མགྱོགས་པོར་དགེ་རྒན་ལ་བཤད་རྒྱུར། “དགེ་རྒན་ལགས། ང་རང་ནི་སློབ་གྲྭར་མ་སོང་། འོན་ཀྱང་། སློབ་གྲྭར་འགྲོ་མཁན་ལ་དཀའ་ངལ་འདུག་པ་འདྲ།”

སྐལ་བཟང་གིས་ཧ་ལས་པའི་ངང་ནས། “འོ་ལགས། དེ་ག་རེ་རེད།” ཞེས་དྲིས།

དགེ་རྒན་གྱིས། དྲི་བ་དེར་ལན་མ་བཏབ། ཁོང་གིས་ཅི་ཞིག་བསམ་བཞིན་འདུག ཁོང་གིས་འཇིགས་མེད་དབྱངས་ཅན་ལ་སྤྲང་རྟགས་གང་ཡིན་དྲིས་པར། “བུ་མོ། ཁྱེད་ཀྱིས་བཤད་པ་ལྟར་བྱས་ན། སྤྲང་རྟགས་ཟེར་བ་ག་རེ་ཡིན།” ཞེས་བཤད།

འཇིགས་མེད་དབྱངས་ཅན་གྱིས་ཡིད་ཆེས་མེད་པའི་རྣམ་པས། “དེ་ག་རེ་རེད་དམ་ཞེ་ན། སྤྲང་རྟགས་ནི་སྐར་མ་མང་པོ་ཡོད་པའི་བསླབ་དེབ་རྙིང་པ་རེད། ཁོང་ཚོར་འབྲི་དགོས་པའི་ལས་ཀ་ཡོད་ཚད་དྲག་སྐྱེས་སུ་བྱེད་དགོས་ཀྱི་འདུག ཁོང་ཚོས་དུས་ཡུན་གང་ཡིན་རུང་ཡོང་གི་ཡོད། མཚན་མོ་ལའང་ཡོང་། གལ་སྲིད་ང་ཚོས་ལས་ཀ་མ་བྱས་ན་། ཁོང་ཚོས་གང་དགའ་བྱེད་ཀྱི་རེད།” ཅེས་བཤད་པ་རེད།

དེ་གོ་མ་ཐག་ཏུ། དགེ་བཤེས་བསོད་ནམས་ཀྱིས་ཤེས་རྟོགས་བྱུང་བ་སྟེ། “ཨ་ཕ། ང་རང་གིས་ཤེས་སོང་། སྤྲང་རྟགས་ཟེར་བ་དེ་ནི་སློབ་མའི་ལས་བྱེད་སྙིང་པོ་མེད་པ་རེད། ཁོང་ཚོ་འབྲི་རྒྱུར་སྙིང་རྗེ་མེད་པ། ཧ་ཅང་ཁག་པོ་འདུག ང་ཚོར་དེའི་ཐད་ཐབས་ཤེས་བྱེད་དགོས་ཀྱི་འདུག” ཟེར།

དགེ་རྒན་གྱིས། “བདེན་པ་རེད། ང་ཚོས་དེ་ལ་ཐབས་ཤེས་བཏོན་ན་ཆོག་གི་རེད།” དེ་ནས་དགེ་རྒན་གྱིས། “བུ་མོ། ཁྱེད་རང་དངོས་གནས་ཡག་པོ་འདུག ད་ཁྱེད་ཀྱིས་ཐབས་ཤེས་གང་འདྲ་ཞིག་ཡོད་དམ་ཤོད་དང་། གང་ལྟར་ངས་ཁྱེད་རང་ལ་གནས་ལུགས་གསལ་པོར་བཤད་ཆོག” ཅེས་གསུངས།

འཇིགས་མེད་དབྱངས་ཅན་གྱིས། “ང་ཚོར་གལ་ཏེ་རང་ཉིད་ཀྱི་ལས་ཀ་དམ་པོར་བྱས་ན་ཆོག་གི་རེད། ཡང་ན་མི་ཚང་མས་མཉམ་དུ་བསྡད་དེ་ལས་ཀ་བྱེད་ཀྱི་ཡིན། བར་བར་དུ་བརྡ་ལན་གཏོང་ཆོག དེ་ནས་སྤྲང་རྟགས་དེ་ཚོ་ཡོང་ཐུབ་ཀྱི་མ་རེད། གལ་ཏེ་དེ་འདྲ་བྱས་ཚེ། ང་ཚོ་འཇིགས་སྣང་བྱེད་དགོས་དོན་མེད།” ཅེས་བཤད།

དེ་གོ་མ་ཐག་དགེ་རྒན་གྱིས་ཧ་ཅང་དགའ་བའི་ངང་ནས་“ཡག་པོ་འདུག ཁྱེད་ཀྱི་ཐབས་ཤེས་ནི་ཡག་ཤོས་རེད། ད་དུང་གཅིག་བཤད་རོགས་།” ཞེས་གསུངས།

འཇིགས་མེད་དབྱངས་ཅན་གྱིས། “ཡང་ཐབས་ཤེས་གཞན་ཞིག་ནི། ང་ཚོས་སློབ་གྲྭའི་ནང་དུ་དགའ་སྤྲོ་ཡོང་བ་བྱེད་དགོས། སློབ་ཚན་ལ་སྤྲོ་སེམས་སྐྱེ་དགོས། བྱིས་པ་ཚོས་རང་གི་ལས་ཀ་ལ་དགའ་བོ་བྱེད་དགོས། གལ་སྲིད་སྤྲོ་སེམས་སྐྱེ་དགོས་ན་སློབ་ཚན་སློབ་ཁྲིད་བྱེད་སྟངས་བསྒྱུར་ན་ཆོག”

དགེ་རྒན་གྱིས་ཧ་ཅང་ཡིད་སྨོན་གྱིས་ “བདེན་པ་རེད། ཁྱེད་ཀྱི་ཐབས་ཤེས་འདི་ཧ་ཅང་ཕན་ཐོགས་ཡོང་ངེས་རེད།” ཞེས་བཤད།

དེ་རྗེས་ཁོང་ཚོས་ཐབས་ཤེས་གཞན་དག་མང་པོ་ཞིག་བཤད་པ་རེད། འཇིགས་མེད་དབྱངས་ཅན་གྱིས་གཏམ་བཤད་ཧ་ཅང་མཁས་པོ་བྱས་པ་དང་། ཐབས་ཤེས་སྣ་ཚོགས་བཤད་དེ་སློབ་སྦྱོང་བྱེད་སྟངས་གསར་པ་མང་པོ་ཞིག་བསླབས་པ་རེད། དེ་ནི་སྤྲང་རྟགས་དེ་ཚོའི་དཀའ་ངལ་སེལ་ཐབས་རེད་ཟེར་བས། དེ་ནས་འཇིགས་མེད་དབྱངས་ཅན་ལ་སྔགས་པ་ཆེན་མོའི་བྱམས་པའི་མཐོང་སྣང་གནང་བས་ཧ་ཅང་དགའ་སོང་། མཐར་དགེ་རྒན་ཆེན་མོ་སེམས་དགའ་པོ་དང་བཅས་ཕེབས་སོང་།

ཉིན་དེ་ནས་བཟུང་། སློབ་གྲྭའི་ནང་གི་སློབ་མ་ཚང་མས་ལས་ཀ་དུས་ཚོད་ལྟར་བྱེད་ཀྱི་ཡོད་ལ། ཁོ་ཚོས་སྤྲོ་སྣང་ཆེན་པོས་སློབ་སྦྱོང་བྱེད་ཀྱི་ཡོད་པས། སློབ་གྲྭའི་སྤྲང་རྟགས་ཚང་མ་ནུས་མེད་དུ་གྱུར་སོང་། དེར་བརྟེན་འཇིགས་མེད་དབྱངས་ཅན་ཧ་ཅང་དགའ་པོ་བྱུང་། ཁོ་མོས་དགེ་རྒན་གྱི་གསུང་ཡོད་དོ་ཅོག་ཉན་པ་དང་སློབ་སྦྱོང་ལ་བརྩོན་པ་བྱས་པས། ཁོ་མོ་དགེ་རྒན་གྱིས་བསྔགས་བརྗོད་བྱེད་ཡུལ་དུ་གྱུར།

དེའི་རྗེས་སུ་འཇིགས་མེད་དབྱངས་ཅན་སློབ་མ་ཡག་ཤོས་དེ་ཆགས་པ་རེད། ཁོ་མོ་ནི་སློབ་གྲྭའི་བྱིས་པ་ཚང་མར་སེམས་གདེངས་འཁེལ་བ་དང་། དུས་ནམ་ཡིན་རུང་མི་གཞན་ལ་ཕན་ཐོགས་ཡོང་བའི་ལས་ཀ་བྱེད་ཀྱི་རེད། མི་ཚང་མས་ཁོ་མོ་ལ་བརྩེ་བ་ཆེན་པོས་བལྟ་བཞིན་ཡོད།
