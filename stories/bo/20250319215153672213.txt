རྡོ་རྗེས་དགེ་རྒན་གྱི་བཀའ་གསུང་ལ་ཉན་པ་དང་དམ་བཅས་ཏེ། མི་གཞན་ལ་ཕན་འདོགས་བྱེད་ཀྱི་སེམས་བསྐྱེད་བསམ་བློ་བྱེད། ཁོས་དགེ་རྒན་གྱི་གསུངས་པ་དེ་ནམ་ཡང་བརྗེད་མི་སྲིད།

ཉིན་གཅིག་ལ། ལྷ་ཡུལ་གྱི་ཕྱོགས་གང་སར་འབྲུག་སྒྲ་དྲག་པོ་ལྡིར་ཞིང་། ནམ་མཁའ་ནི་མུན་ནག་ཏུ་གྱུར་པ་དང་། ཆར་པ་དྲག་པོ་བབས་ཏེ། མི་རྣམས་ཧ་ལས་ཤིང་སྐྲག་དགོས་བྱུང་། མི་རྣམས་རང་ཁྱིམ་ནང་དུ་སྐུམ་ནས་བསྡད་ཡོད་ལ་། རྡོ་རྗེ་ནི་སེམས་འཇགས་པོ་བྱས་ནས། དགེ་རྒན་གྱིས་གསུངས་པའི་གཏམ་བཤད་ལ་དྲན་གསོས་པ་རེད། ཁོས་དགོངས་པ་དང་སེམས་ལ་བསམ་བློ་བྱེད་པའི་ཚུལ་དུ་སྨོན་ལམ་བཏབ་པ་རེད། སེམས་ལ་ཡིད་ཆེས་དང་སྤོབས་པ་ཆེན་པོ་བསྐྱེད་དེ་འཚོ།

དགེ་རྒན་ཡང་མི་རྣམས་ལ་རོགས་རམ་བྱེད་དུ་བསྐྱོད་པ་དང་། ཁོང་གིས་རྡོ་རྗེ་ལ་བཀའ་བཀོད་གནང་། དྲ་བ་འདི་དག་མྱུར་དུ་བཅོས་དགོས་པའི་བཀའ་གནང་། རྡོ་རྗེས་ཀྱང་དེའི་བཀའ་བཞིན་ལག་ལེན་དུ་བསྟར། རྡོ་རྗེས་ཀྱིས་ཁོའི་བློ་གྲོས་དང་སྤོབས་པ་ལ་བརྟེན་ནས་འབྲུག་སྒྲ་སྙན་མོ་ཇེ་དུལ་དུ་གྱུར་བ་རྟོགས་བྱུང་།

རྡོ་རྗེ་སྒྲོལ་མའི་སར་ཕྱིན་ནས་། "སྒྲོལ་མ་ལགས། མ་འཇིགས་རོགས། ང་ཚོས་དཀའ་ངལ་ཁྱད་དུ་བསད་དེ་གྲུབ་འབྲས་ཐོབ་ངེས་ཡིན།" ཞེས་བཤད།

སྒྲོལ་མ་ཡིས་རྡོ་རྗེར་བཤད་པ། "ངས་ཁྱོད་ལ་ཡིད་ཆེས་ཡོད།"

རྡོ་རྗེ་དང་སྒྲོལ་མ་གཉིས་ཀྱིས་མཉམ་དུ་འབད་བརྩོན་བྱས་པ་དང་། མཐར་འབྲུག་སྒྲ་ཡང་ཞི་དུལ་དུ་གྱུར། ཉི་མ་ཤར་བ་དང་ལྷ་ཡུལ་དུ་ཞི་བདེ་སླར་གསོ། མི་རྣམས་དགའ་བས་ཁེངས་ཤིང་། དགེ་རྒན་གྱིས་རྡོ་རྗེ་དང་སྒྲོལ་མ་ལ་བསྟོད་བསྔགས་དང་བྱ་དགའ་ཆེན་པོ་གནང་།

དེ་རྗེས་ཀྱི་ཉིན་མང་པོའི་རིང་ལ། རྡོ་རྗེས་མུ་མཐུད་སློབ་སྦྱོང་དང་ཉམས་ཞིབ་བྱས་པ་དང་། ཁོང་རང་ཉིད་བླ་མ་ཆེན་པོ་ཞིག་ཏུ་གྱུར་ཏེ། མི་མང་པོ་ཞིག་ལ་བསླབ་བྱ་བྱས་པ་དང་། ཆོས་ཀྱི་བཀའ་གདམས་སྤེལ་བས་། ཁོས་ལྷ་ཡུལ་ཧྲིལ་པོར་སྙན་གྲགས་ཁྱབ། རྡོ་རྗེ་དགེ་ཚུལ་ནི་ནམ་ཡང་བཀའ་དྲིན་ཆེ་བའི་མི་སྣ་ཞིག་ཏུ་གྱུར།

མཇུག་ཏུ་། རྡོ་རྗེ་ཡིས་དགེ་རྒན་གྱི་བཀའ་དྲིན་དང་སྒྲོལ་མའི་རོགས་རམ་ལ་བརྗེད་མི་སྲིད། ལྷ་ཡུལ་ཡང་ནམ་ཡང་ཞི་བདེ་ལྡན་པ་ཞིག་ཏུ་གྱུར། རྡོ་རྗེ་ཡིས་ཉིན་ལྟར་སྨོན་ལམ་འདེབས་པ་དང་། དཀའ་ངལ་ལ་བློ་སྤོབས་བསྐྱེད་པ། མི་གཞན་ལ་བྱམས་སེམས་བྱེད་པ་སོགས་ཀྱིས་རང་ཉིད་དང་གཞན་ལ་ཕན་ཐོགས་བྱེད་པ་ནི་ཁོའི་མི་ཚེའི་སྙིང་པོ་ཡིན་པ་རེད། ལྷ་ཡུལ་གྱི་བྱིས་པ་ཚང་མ་རྡོ་རྗེ་ཡིས་གསུངས་པའི་གཏམ་ལ་ཉན་པས་ཡ་རབས་ཀྱི་མི་བཟང་པོར་གྱུར་ཐུབ།
