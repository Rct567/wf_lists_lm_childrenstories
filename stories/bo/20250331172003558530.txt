བྱིས་པའི་གཏམ་རྒྱུད་དེ་ཚེས་བསླབ་བྱ་རེད།

གནས་ཚུལ་འགོ་འཛུགས་ཀྱི་སྒྲུང་གི་མིང་ལ་ “བསམ་བློའི་འཇིག་རྟེན་དང་ལུས་སྦྱོང་བྱེད་ཕྱོགས།” ཟེར་བ་རེད།

ཉི་མ་གཅིག་ལ་ཁུ་ཤུ་དང་པདྨ་གཉིས་ཀྱིས་རི་ཁུག་ཅིག་ལ་རྐང་ཐང་དུ་འགྲོ་གི་ཡོད། ཁོ་གཉིས་ཀ་རང་ཉིད་ཀྱི་རང་བཞིན་གྱི་གླིང་ཀ་ལ་མཛའ་བརྩེ་ཆེན་པོ་ཡོད་པའི་བྱིས་པ་ཡིན། ཁོ་གཉིས་ཀྱིས་རྡོ་རྗེ་དང་འབྲོག་པའི་ཁྱིམ་ཚང་དུ་དྲན་གསོ་མང་པོ་བྱེད།

ཁུ་ཤུ། “པདྨ་ལགས། ང་ཚོས་བསམ་བློ་འདི་དག་ལ་གླེང་མོལ་བྱེད་ཐུབ་བམ། ང་རང་བསམ་བློ་བཏང་ནས་དཀའ་ངལ་མང་པོ་ཡོད། གང་ཡིན་ཟེར་ན། བསམ་བློ་ནི་འཇིག་རྟེན་རེད།”

པདྨ། “ཁུ་ཤུ་ལགས། ཁྱེད་རང་བདེན་པ་རེད། ངས་ཀྱང་དེ་རིང་དེ་ལྟར་བསམ་བློ་བཏང་། ངས་ང་ཚོའི་བསམ་བློ་དེ་འཇིག་རྟེན་རང་དུ་བརྩི་ཡོད། དེ་ནི་ཁྱེད་རང་གིས་སྨྲས་པ་ལྟར་ཡིན་པ་རེད།”

ཁུ་ཤུ། “ངས་ཁྱེད་ཀྱི་སེམས་འདུན་ལས་དཀའ་ངལ་འདུག་ལ། ངའི་བསམ་བློའི་ནང་གི་ཚེ་གཅིག་འདི་ལ་རྒྱུན་དུ་འཇིགས་སྣང་སྐྱེས་ཀྱི་འདུག”

པདྨ། “འཇིགས་སྣང་། འོ་ན། ང་ཚོར་དེའི་ཐོག་ལ་ཅི་ཞིག་བཤད་རྒྱུ་ཡོད་དམ། ངས་ནམ་རྒྱུན་ངའི་ཕ་ཡུལ་ལ་བསམ་བློ་གཏོང་སྐབས། རྩེ་མོ་མཐོན་པོའི་རི་བོ་དང་དེ་དག་ལ་བརྩེ་དུང་འཆང་གི་ཡོད། ང་ཚོའི་སེམས་ནང་དུ་ཡང་གནས་སོ་སོ་ཡོད་དམ་”

ཁུ་ཤུ། “ཡང་དག་རེད། ངས་ཤེས་པ་ནི། ང་ཚོའི་སེམས་ལ་སོ་སོའི་ཁ་དོག་ཡོད། དེ་བཞིན་འཚོ་བའི་ཁྲོད་ཀྱི་སྣང་བ་འདི་ལ་འདོད་པ་ཡོད་དོ། ང་དུས་རྒྱུན་དུ་བསམ་བློ་མང་དུ་བཏང་།”

པདྨ། “ཁྱེད་ཀྱིས་དཀའ་ངལ་ལ་གདོང་ལེན་བྱེད་ཐུབ་དགོས་པ་རེད། ངས་དེ་ཤེས་ཡོད། ལུས་པོ་བདེ་ཐང་ཡིན་ན་སེམས་ཀྱང་བདེ་ཐང་ཡོང་གི་རེད། དེའི་ཕྱིར། ང་ཚོས་རྒྱུན་དུ་ལུས་སྦྱོང་བྱེད་དགོས།”

ཁུ་ཤུ། “པདྨ་ལགས། ཁྱེད་ཀྱིས་ལུས་སྦྱོང་ཇི་ལྟར་བྱེད་ཀྱི་ཡོད་དམ། ངའི་ཕ་མ་ཡང་དེ་ལྟར་བཤད་སོང་། ང་སྤྲོ་སེམས་འཁོལ་པོས་སེམས་སྦྱོང་བྱེད་པ་ནི་འཇོན་ཐང་ལྡན་པ་ཞིག་ཡིན་པར་ངེས།”

པདྨ། “དེ་ལྟར་རེད། ང་ཚོས་ཐོག་མར་གོམ་སྦྱོང་བྱེད་ཀྱི་ཡིན། འབྲི་རྐང་ལ་འགྲོ་རྒྱུ་དང་སྤྲོ་སྣང་ཡོང་གི་རེད། ཡང་ན་ཤིང་ཏོག་སྒྲུག་པའམ་ཆུ་ཀླུང་ལ་ཆུ་རྩེད་བྱེད་རྒྱུ་ཡིན།”

ཁུ་ཤུ། “ཆུ་རྩེད་། དེ་ཧ་ཅང་ཡག་པོ་འདུག དེ་ནི་བསམ་བློའི་ནང་གི་འཚོ་བ་བདེ་ཐང་ལ་འགྱུར་བ་གཏོང་བར་ཧ་ཅང་ཕན་ཐོགས་ཆེན་པོ་ཡོད་”

པདྨ། “དེ་བདེན་པ་རེད། ལུས་སྦྱོང་བྱེད་སྐབས། ང་ཚོའི་ལུས་པོར་སྐྱོན་ཆད་ཆུང་དུ་གཏོང་བ་མ་ཟད་བདེ་ཐང་དུ་འགྱུར་བ་དང་སེམས་ཀྱི་འཇིགས་སྣང་ཡང་ཇེ་ཉུང་དུ་གཏོང་གི་རེད།”

ཁུ་ཤུ། “ད་དུང་། ངེད་གཉིས་ཀྱིས་མཉམ་དུ་ལུས་སྦྱོང་བྱེད་ཐུབ་བམ། ངས་ནམ་རྒྱུན་རྐང་རྩེད་སྤོ་ལོ་རྩེད་ཀྱི་ཡོད། ཁྱེད་ཀྱིས་དགའ་པོ་བྱེད་ཀྱི་ཡོད་དམ།”

པདྨ། “དེ་ཧ་ཅང་བཟང་། ང་རྐང་རྩེད་སྤོ་ལོ་ལ་དགའ་པོ་ཡོད། ད་དུང་ངས་ནམ་རྒྱུན་སྡོང་པོའི་འོག་ལ་བསྡད་དེ་བསམ་བློ་གཏོང་གི་ཡོད། ལུས་སྦྱོང་བྱེད་རྒྱུ་ནི་སེམས་ལ་རོགས་བྱེད་ཐབས་ཡག་ཤོས་རེད།”

ཁུ་ཤུ། “ཁྱེད་རང་ཚད་ལྡན་རེད། ངས་རྗེས་སུ་འབྲང་། ངས་འདི་ལྟར་བསམ་བློ་བཏང་། ངས་བསམ་བློ་མང་པོ་གཏོང་དགོས་ཀྱི་ཡོད་མོད། ལུས་སྦྱོང་བྱེད་རྒྱུར་ཡང་དེ་བཞིན་སེམས་ནས་བསམ་གཞིགས་བྱེད་ཀྱི་ཡོད།”

པདྨ། “དེས་ན། ང་ཚོས་འདི་རིང་གི་ཉི་མ་ལམ་སྟོན་དུ་འཇོག་དགོས། ཁྱེད་ཀྱིས་དངོས་གནས་བདེན་པ་རེད། ད་ཆ་ང་ཚོས་ཐག་གཅོད་པ་ནི་རྩལ་སྦྱོང་ལ་དུས་ཚོད་འགའ་ཞིག་བཏང་བ་དེ་རེད།”

ཁུ་ཤུ། “ཡང་དག་རེད། ང་ཚོས་དགེ་རྒན་འཚོལ་དགོས། སློབ་སྦྱོང་བྱེད་དགོས་པར་མ་ཟད་སེམས་ལ་སྦྱོང་བརྡར་བྱེད་དགོས།”

པདྨ། “ད་ང་ཚོས་དེའི་སྐོར་ལ་མུ་མཐུད་གླེང་མོལ་བྱེད་ཆོག ངས་བསམ་པར་བསམ་བློ་ནི་འཇིག་རྟེན་ཞིག་དང་འདྲ་བ་དང་། ལུས་སྦྱོང་ནི་ང་ཚོའི་དེ་དག་གི་ལམ་ལྷོང་ལ་བསླབ་བྱ་གལ་ཆེན་པོ་ཞིག་རེད།”

ཁུ་ཤུ། “གལ་ཏེ་ཁྱེད་རང་འཐད་པ་ཡིན་ན། ང་ཚོས་འདི་རིང་ཐོག་མར་སེམས་ལ་ཞུགས་ནས་ད་ལྟ་གོམ་འགྲོས་སློབ་དགོས། ངས་མྱུར་དུ་ལུས་སྦྱོང་བྱེད་པའི་ལམ་དུ་འགྲོ་བར་སེམས་ནས་ཡིད་ཆེས་བྱེད་ཀྱི་ཡོད།”

པདྨ། “ང་ཡང་ཁྱེད་ཀྱི་བསམ་བློ་དང་མཐུན་གྱི་ཡོད། ཁྱེད་ལ་སྐྱིད་པོ་ཡོང་། ང་ཚོས་དེའི་སྐོར་ལ་སྔོན་ལ་བསམ་བློ་བཏང་བ་དེ་ཧ་ཅང་ཡག་པོ་འདུག དེ་ནས་མཉམ་དུ་ལུས་སྦྱོང་བྱེད་ཆོག་པ་རེད།”

ཁུ་ཤུ། “དེ་རིང་གི་གཏམ་རྒྱུད་ཀྱིས་ངའི་སེམས་ལ་ཕན་ཐོགས་བྱུང་། པདྨ་ལགས། ཐུགས་རྗེ་ཆེ།”

པདྨ། “དགའ་པོ་བྱུང་། ཁྱེད་ཀྱིས་ངེས་པར་དུ་བསམ་བློ་དག་ལ་རྒྱུས་ལོན་བྱེད་དགོས། ང་ཚོས་ལུས་སྦྱོང་བྱེད་རྒྱུ་དང་བསམ་བློ་བཏང་བས་སེམས་ཀྱི་སྡུག་བསྔལ་འགོག་ཐུབ།”

དེ་ནས་བྱིས་པ་གཉིས་པོ་དགའ་སྤྲོ་དང་བཅས་ཕན་ཚུན་ལ་གཅིག་ལས་གཅིག་ལ་གཏམ་བརྗོད་བྱེད་བཞིན་འགྲོ་ཞིང་བཞུགས། ཁོ་གཉིས་ཀྱིས་རང་ཉིད་ཀྱི་སེམས་དང་ལུས་པོ་གཉིས་ཀར་བདེ་སྐྱིད་སྤྲོད་པར་ཐག་གཅོད་བྱས་པ་རེད།
