ཁོང་གཉིས་ཀྱིས་སྤྲིན་པའི་སྟེང་ནས་རྩེད་མོར་རོལ་བཞིན་ནམ་མཁའ་ལ་འཕུར་སྐབས། ཉིན་གུང་ནམ་མཁའ་ནས་སྐར་ཚོགས་མང་པོ་མཐོང་བྱུང་། ཁོང་ཚོས་ནམ་ཡང་བརྗེད་ཐབས་མེད་པའི་མཛེས་སྡུག་ལྡན་པའི་གནམ་གཤིས་མྱོང་བ་རེད། སྐར་ཚོགས་རེ་རེས་ཁོང་ཚོའི་སེམས་ལ་རེ་བའི་གསོན་ཤུགས་སྦྱིན་བྱུང་།

ཉིན་གཅིག་ལ། བསོད་ནམས་ཀྱིས་སླར་ཡང་སྤྲིན་པའི་ནང་དུ་འཕུར་རྒྱུ་མཚམས་བཞག་སྟེ་གནས་ཚུལ་འགའ་ཤས་ལ་བསམ་བློ་བཏང་། ཁོས་གྲོགས་པོ་དོན་གྲུབ་ལའང་ཁོ་རང་དང་མཉམ་དུ་འབོད་བསྐུལ་བྱས། ཁོ་གཉིས་ཀྱིས་མཉམ་དུ་རྩེད་མོ་རྩེ་ཞིང་རང་གི་བློ་ཕེར་བའི་ལས་ཀ་བྱེད་རྒྱུ་གཏན་འཁེལ་བྱས། ཁོ་ཚོས་བདེན་པ་དང་བརྩེ་བ་ལ་དགའ་པོ་བྱེད། དེ་ནི་ཁོང་ཚོའི་མི་ཚེའི་ལམ་ཕྱོགས་ཡིན།

ཉིན་ཞིག་ལ། བསོད་ནམས་ཀྱིས་དཔེ་དེབ་ཁྲོད་ཀྱི་བདེན་དོན་མང་པོ་ཤེས་བྱུང་། ཁོས་སྤྲིན་པ་ནང་དུ་འཕུར་བ་དེས་བརྩེ་བ་དང་ཕན་ཚུན་གཅེས་སྤྲས་བྱེད་རྒྱུར་སྐུལ་མ་བྱེད། ཁོའི་མིག་མདངས་ནི་འོད་ཟེར་འཕྲོ་བཞིན་འདུག་ལ། ཁོའི་སེམས་ནང་དུ་ཡང་དགའ་སྤྲོའི་ཚོར་བ་ཡོད།

དེ་ནས་བསོད་ནམས་ཀྱིས་གཞན་ལ་རོགས་རམ་བྱེད་འགོ་བརྩམས། ཁོས་སློབ་གྲྭའི་ནང་དུ་རྒན་རྒོན་དག་ལ་དཀའ་ངལ་སེལ་རོགས་བྱས། ཁོས་ཁོང་ཚོར་བརྩེ་བ་དང་བྱམས་བརྩེ་མཚོན་པར་བྱས། ཁོང་ཚོའི་མིག་ཆུ་གདོང་དུ་ཤོར་བའི་དུས་སུ། ཁོས་རང་ཉིད་ཀྱི་མི་ཚེ་དོན་སྙིང་ལྡན་པ་ཞིག་ཡིན་པ་ཚོར།

བསོད་ནམས་ཀྱིས་ཤེས་ཡོན། གནས་ཚུལ། རང་མཚན་དང་ཕན་ཚུན་བརྩེ་བའི་དོན་དག་དག་སློབ་སྦྱོང་བྱས། ཁོས་སེམས་གཏིང་ནས་དེ་དག་རང་གི་མི་ཚེའི་ནང་དུ་བེད་སྤྱོད་བྱས་ཏེ། རང་གི་རྨི་ལམ་དངོས་དོན་ལ་བསྒྱུར། རྨི་ལམ་གྱི་ནང་དོན་བདེ་སྐྱིད་དུ་བསྒྱུར་བས།

དེ་ལྟར་བསོད་ནམས་ནི་ནམ་རྒྱུན་དུ་སྤྲིན་པའི་སྟེང་ནས་འཕུར་འདོད་བྱེད། ཁོའི་རྨི་ལམ་གྱི་གཏམ་རྒྱུད་ཡུན་རིང་པོར་འཚེར་སོང་། འོན་ཀྱང་ཁོས་ནམ་རྒྱུན་གྲོགས་པོ་དོན་གྲུབ་དང་མཉམ་དུ་མི་ཚེ་རོལ་པ་དང་སྤྲོ་སྣང་སྐྱེད། ཁོང་ཚོའི་མཛའ་མཐུན་ཡུན་རིང་དུ་གནས་པ་དང་བརྩེ་བ་ཟབ་མོའི་སེམས་ཚོར་སླར་ཡང་མྱོང་ཐུབ། ཁོང་ཚོའི་འཚོ་བ་ནི་དགའ་སྐྱིད་དང་བརྩེ་བས་ཕྱུག་པ་ཞིག་ཡིན། ཁོང་གཉིས་ནམ་ཡང་ཁ་བྲལ་དུ་འགྲོ་གི་མ་རེད། རང་ཉིད་ཀྱི་སེམས་ནང་དུ་ཡོད་པའི་རེ་བ་དང་འདུན་མར་རྒྱུན་དུ་དགའ་སྣང་ཡོད་ཅིང་རྩེད་མོར་རོལ་བ་རེད། དེ་ནི་ཁོང་ཚོའི་མི་ཚེའི་བདེ་སྐྱིད་ཡིན། །
