བདེ་སྐྱིད་ཀྱིས་བསམ་བློ་བཏང་། གལ་སྲིད་ངའི་ཕ་ཡུལ་ལ་ཕྱིན་ན། ང་ལ་ཇོན་ལྟ་བུའི་གྲོགས་པོ་ཞིག་ཡོད་ན་ཅི་མ་རེད། ཇོན་ནི་ཧ་ཅང་ཡག་པོ་འདུག ཁོས་བོད་ལ་དགའ་པོ་ཡོད། བོད་ཀྱི་རིག་གཞུང་སློབ་སྦྱོང་བྱེད་པར་དགའ་པོ་ཡོད།

ཉིན་གཅིག་ལ། བདེ་སྐྱིད་ཀྱི་ཕ་ཡུལ་ནས་འབྱོར་བའི་ཡིག་གེ་ཞིག་འབྱོར། དེའི་ནང་དུ་ཕ་ཡུལ་གྱི་མཁན་པོས་འདི་ལྟར་བྲིས་ཡོད། “བུ་མོ། ཁྱེད་རང་མྱུར་དུ་ཕ་ཡུལ་དུ་ལོག་ཤོག འདིར་བསླབ་བྱའི་འཛིན་གྲྭ་གསར་པ་ཞིག་འགོ་བརྩམས་ཡོད།”

བདེ་སྐྱིད་སེམས་ཧ་ཅང་སྐྱིད་པོ་བྱུང་། མོའི་ཕ་ཡུལ་ལ་བསླབ་བྱའི་འཛིན་གྲྭ་ཡོད་རེད། དེ་ནི་ཧ་ཅང་བཟང་པོ་རེད། མོས་བསམ་བློ་བཏང་། ངེད་གཉིས་མཉམ་དུ་འགྲོ་ཆོག

བདེ་སྐྱིད་ཀྱིས་ཇོན་དང་བསོད་ནམས་གཉིས་ལ་གསལ་བཤད་བྱས་པ་རེད། བསོད་ནམས་དང་ཇོན་གཉིས་ཀྱང་སེམས་སྐྱིད་པོ་བྱུང་།

ཇོན། “བདེ་སྐྱིད་ལགས། ཁྱེད་རང་ལ་ལམ་འགྲོ་བདེ་མོ།”

བསོད་ནམས། “བདེ་སྐྱིད་ལགས། ཁྱེད་རང་སེམས་གསོ་གནང་།”

བདེ་སྐྱིད། “ཁྱེད་རང་གཉིས་ལའང་བཀྲ་ཤིས་བདེ་ལེགས།”

བདེ་སྐྱིད་ཕ་ཡུལ་ལ་ལོག་སོང་། ཁོ་མོ་ཕ་ཡུལ་ལ་སླེབས་རྗེས། མཁན་པོས་མོར་དགའ་བསུ་བྱས། བདེ་སྐྱིད་ཀྱིས་ཇོན་དང་བསོད་ནམས་ཀྱིས་བཤད་པའི་སྐད་ཆ་དག་མཁན་པོར་བཤད་པས། མཁན་པོ་ཧ་ཅང་དགའ་པོ་བྱུང་།

མཁན་པོས། “ད་རེད། སྤྱིར་བཏང་དུ་ཡུལ་གཞན་ནས་ཡོང་བའི་མི་དག་བོད་ཀྱི་སྐད་ཡིག་དང་རིག་གཞུང་ལ་སྦྱངས་ན་ཧ་ཅང་ལེགས། དེས་བོད་ཀྱི་སློབ་གསོ་ཡང་ཡར་རྒྱས་འགྲོ་བར་ཕན་ཐོགས་ཡོད། ང་ཚོས་འདི་འདྲའི་བྱ་བཞག་ལ་རྒྱབ་སྐྱོར་བྱེད་དགོས།”

བདེ་སྐྱིད། “ལེགས་སོ། ང་ཚོའི་དགེ་རྒན་ཚང་མས་དེ་ལྟར་གསུངས་སོང་།”

ཉིན་རེའི་འཚོ་བའི་ནང་། བདེ་སྐྱིད་བོད་སྐད་དང་། བོད་ཀྱི་རིག་གཞུང་། བོད་ཀྱི་ཆོས་ལུགས་ཀྱི་ཐོག་ལ་མུ་མཐུད་སྦྱང་པ་བྱས་པ་རེད། མོས་ཇོན་གྱིས་བསླབས་པའི་སྦྱང་ཚུལ་ལ་གཞིར་བཟུང་ནས། བསོད་ནམས་དང་ཇོན་དང་མཉམ་དུ་མུ་མཐུད་ནས་འབྲེལ་བ་བྱས་ཏེ། བོད་སྐད་ལ་སྦྱངས་པ་བྱས་པས་མོས་མཁན་པོས་བཤད་པའི་ཚིག་འདི་དྲན་བཞིན་ཡོད། “བོད་ནི་རང་གི་ཕ་ཡུལ་ཡིན་པར་མ་ཟད། འཛམ་གླིང་གི་ནང་ལ་དགེ་མཚན་ཆེ་བའི་རིག་གཞུང་ཞིག་ཡིན། དེར་བརྟེན་བོད་མི་ཡིན་པ་ཚང་མས་རང་ཉིད་ཀྱི་སྐད་ཡིག་དང་རིག་གཞུང་ལ་སྦྱང་པ་བྱེད་དགོས།”

མཐའ་མཇུག་ཏུ་། བསོད་ནམས་དང་བདེ་སྐྱིད་གཉིས་ཀྱིས་ཇོན་གྱི་སློབ་སྦྱོང་ལ་རོགས་རམ་བྱས། ཇོན་གྱིས་བོད་སྐད་གསན་པའི་དུས་ཚོད་ལ། ཁོང་ཧ་ཅང་སེམས་སྐྱིད་པོ་བྱུང་། ཁོང་གིས་དུས་རྒྱུན་དུ་འདི་ལྟར་བཤད་ཀྱི་ཡོད། “ངས་བོད་ཀྱི་སྐད་ཡིག་ཤེས་པ་ནི་ངའི་མི་ཚེའི་ནང་གི་དོན་ཆེན་ཞིག་རེད། བོད་ཀྱི་སྐད་ཡིག་ལ་སྦྱངས་པས་ངས་བོད་ཀྱི་རིག་གཞུང་ཧ་ཅང་ཡག་པོར་ཤེས་ཀྱི་འདུག” དེ་ལྟར་བྱས་ནས་ཁོང་གིས་བོད་དུ་ཡོད་པའི་ལོ་མང་པོའི་རིང་ལ་བོད་ཀྱི་མི་ཚེའི་རིན་ཐང་དང་། འཛམ་གླིང་ལ་བརྩེ་བ་བྱེད་སྟངས་སོགས་ལ་རྒྱུས་ལོན་ཟབ་མོ་བྱུང་བ་རེད།
