དེ་ནས་ཟངས་མེད་ཀྱིས་ཁོའི་ཕྱག་གཡས་ཀྱིས་བར་སྣང་ལ་ལག་པ་བརྐྱངས་ནས་སེམས་ཀྱིས་བསམ་བློ་བཏང་། སྐར་མ་ཁ་ཤས་ཀྱི་རྗེས་སུ་བག་ལེབ་ཧ་ཅང་ཆེན་པོ་ཞིག་ཁོང་གི་མདུན་དུ་ཐོན་བྱུང་། ཟླ་འོད་ཀྱིས་ཧ་ཅང་ཧང་སངས་པའི་ངང་ནས། “ཨ་ལགས་ལགས། འདི་ག་འདྲ་བྱས་ནས་ཡིན།”ཟེར།

དེ་ནས་ཉིན་གཅིག་ལ། ཟླ་འོད་རྫ་ཁར་འགྲོ་བཞིན་པའི་སྐབས། ཁོ་མོས་འབྲོང་ཞིག་མཐོང་བྱུང་། འབྲོང་དེ་ནི་བག་ལེབ་ཟངས་མེད་ཀྱིས་མངོན་དུ་བཏང་བའི་འབྲོང་དེ་རེད། ཟླ་འོད་དགོད་ནས་འབྲོང་དེར་“འབྲོང་ལགས། ཁྱོད་ཀྱིས་ཡང་བག་ལེབ་དགོས་སམ།”ཟེར། འབྲོང་དེས་མགོ་བོ་བསྒུལ་ནས་མོས་ལ“མི་དགོས། ངའི་ཁོང་འདུག་”ཟེར། ཟླ་འོད་ཀྱིས། “ངས་ཁྱོད་ལ་འདྲི་བ་ཞིག་ཡོད། བག་ལེབ་ག་ཚོད་བཟས་པ་ཡིན།”ཟེར།

འབྲོང་གིས་མགོ་བོ་སྒུར་ནས་དེས་རྣམ་པར་དྲན་པའི་གཤིས་ཀ་བསྟན། དེ་ནས་མདུན་དུ་འདུག་ཅིང་“དགོངས་དག་ཞུ། ང་འདི་ག་ནས་ཡིན། བག་ལེབ་མང་པོ་ཡང་ཡིན། དེ་ནི་ངོ་མ་ཡིན། འོན་ཀྱང་། ཁྱོད་ཀྱི་བླ་མ་དེ་སུ་ཡིན་ནམ།”ཟེར།

ཟླ་འོད་ཧ་ལས་པར་གྱུར། “ཨ་ལགས་ཟངས་མེད། ཁོང་རང་ག་འདྲ་བྱེད་དགོས་སམ། ཁོང་གིས་བག་ལེབ་ཡག་པོ་འདི་སྤྲད་པ་རེད།”ཟེར།

འབྲོང་དེ་སེམས་ཆུང་དུ་གྱུར། “ངས་བཤད་ཚར་སོང་། ངས་བདེན་པ་བཤད་ཀྱི་ཡིན། བླ་མ་ཟངས་མེད་ནི་ང་ལ་ཧ་ཅང་ཡིད་ཆེས་ཆེན་པོ་ཡོད། ངས་ཁོང་གི་སྐད་ཆ་ལ་ཉན་པ་ཡིན། ངས་གདོན་འདྲེ་སྤྱོད་ཚུལ་གང་ཡང་མི་འདུག ངས་ལམ་འདི་བརྒྱུད་ནས་ཕྱིན་པས་ཕན་ཐོགས་ཆེན་པོ་ཐོབ་བྱུང་།”ཟེར། ཟླ་འོད་དགོད་ཞོར་“ངས་ཁྱོད་ལ་ཡིད་ཆེས་ཡོད། ང་ཡང་ལམ་འདི་ནས་ཕྱིན་ཆོག་གམ།”ཟེར།

འབྲོང་གིས་“འོ་རེད། ཁྱོད་ལ་དབང་ཆ་ཡོད། ང་ཚོ་མཉམ་དུ་ཕྱིན་ཆོག”ཟེར། ཟླ་འོད་དགའ་སྤྲོའི་ངང་ཕྱིར་ལོག་ནས་ཁོང་གི་ཕྱག་ལ་འབྲེལ་བ་བྱས་པས། མཁའ་རླུང་དུ་འཕུར་སོང་། ཁོ་མོ་ཟངས་མེད་ཀྱི་འཁྲིས་སུ་ཕྱིན་ཏེ་ཚང་མ་བཤད་པ་རེད། 

ཟངས་མེད་ཀྱིས་ཟླ་འོད་ཀྱི་གཏམ་ཉན་ནས་ཧ་ཅང་དགོད་པར་གྱུར། “འོ་དེ་འདྲ་རེད་འདུག་ག།”ཟེར།

“ད་རེས་ཀྱི་གཏམ་རྒྱུད་དེ་ཧ་ཅང་ངོ་མཚར་པོ་འདུག་”ཟེར།

ཟླ་འོད་ཀྱིས་“ཨ་ལགས་ལགས། གལ་སྲིད་ངས་མུ་མཐུད་གཏམ་རྒྱུད་དེ་ལ་ཉན་ན་འདོད་ཡོད།”ཟེར།

“བུ་མོ། ངས་ཁྱོད་ལ་ཉིན་ལྟར་གཏམ་རྒྱུད་ཤོད་ཀྱི་ཡིན།”ཟངས་མེད་ཀྱིས་དགོད་ཞོར་དེ་ལྟར་སྨྲས། དེ་ལྟར་ཁོ་རང་གཉིས་ཀྱིས་སྡེ་བའི་མི་མང་ལ་དགོད་གཏམ་མང་པོ་ཞིག་བཤད། བླ་རྒོལ་ཟངས་མེད་དང་ཟླ་འོད་གཉིས་ཀྱི་མཛའ་མཐུན་གྱི་འབྲེལ་བ་དེ་ནམ་ཡང་མི་འགྱུར་བ་ཞིག་ཡིན་པར་བཤད་ཆོག། དེ་ནས་བཟུང་ཟངས་མེད་ཀྱི་གཏམ་རྒྱུད་དེས་ཡུལ་གྲོང་ཡོངས་ལ་ཁྱབ་སོང་། 

སྡེ་བའི་བྱིས་པ་ཚང་མས་ཟླ་འོད་ལྟར་ཟངས་མེད་ཀྱི་གཏམ་རྒྱུད་ལ་ཉན་པར་དགའ་པོ་བྱེད། ཁོང་གིས་སྤྲད་པའི་བག་ལེབ་ཀྱང་ཟ་འདོད་པོ་བྱེད་ཀྱི་ཡོད། དེ་ནི་ཟླ་འོད་དང་ཟངས་མེད་གཉིས་ཀྱི་འབྲེལ་བའི་སྙིང་པོ་དེ་ཡིན། སུ་ཞིག་གིས་ཀྱང་འདི་ལྟ་བུའི་དགོད་གཏམ་ཞིག་བརྗེད་མི་སྲིད།
