བདེ་ལེགས་ཀྱིས་སྒྲུང་གཏམ་བྲིས་ཚར་བ་དང་བུད་མེད་རྒན་མོར་ཕུལ། བུད་མེད་རྒན་མོས་སྒྲུང་དེར་བལྟས་རྗེས་ཧ་ཅང་དགའ། མོས་སྒྲུང་དེ་བླངས་ཏེ་བུམ་པའི་ནང་བཅུག་པ་དང་བུམ་པའི་ནང་གི་ཆུ་མཐའ་དག་དྭངས་གསལ་ལ་འགྱུར། “བདེ་ལེགས་ལགས། ཁྱེད་རང་ལ་ཐུགས་རྗེ་ཆེ། ཁྱེད་རང་གིས་ང་ཚོར་སྒྲུང་གཏམ་ཡག་པོ་ཞིག་བྲིས་བྱུང་། ང་ཚོས་སྒྲུང་འདི་འཛམ་གླིང་ཡོངས་ལ་བཤད་ཆོག”

དེ་ནས་བདེ་ལེགས་ཀྱིས་རང་གི་མི་ཚེའི་རིང་དུ་སྒྲུང་བྲིས་ཤིང་། མོས་བྱིས་པ་མང་པོར་སྒྲུང་བཤད་པ་རེད། ཁོ་མོས་བཤད་པའི་སྒྲུང་གཏམ་གྱིས་མི་རྣམས་ལ་བརྩེ་བ་དང་བྱམས་སེམས་ཀྱི་བསམ་པ་སྦྱིན་ཡོད་ལ། མི་རྣམས་ཀྱིས་ཁོ་མོའི་སྒྲུང་གཏམ་ལ་ཉན་དུས་སེམས་སྐྱིད་པོ་འདུག བདེ་ལེགས་ནི་སྒྲུང་གཏམ་འབྲི་མཁན་གྲགས་ཅན་ཞིག་ཏུ་གྱུར་པ་རེད། 

ཉིན་ཞིག་བདེ་ལེགས་ནི་རྒས་སོང་བ་དང་ཁོ་མོ་གནམ་ལ་འགྲུལ་བཞུད་བྱེད་དུས་ལོ་ན་ཆུང་དུས་ཀྱི་སྒྲུང་དེ་ཡང་བསྐྱར་དྲན་གསོ་བྱས། མོས་བསམ་པ། “ང་འདི་ནས་ཕྱིར་ལོག་དགོས་འདུག”

དེ་ནས་མོས་མཁའ་ལ་མིག་བལྟས་ཏེ་དེ་རིང་གི་འགྲུལ་བཞུད་དེ་དྲན་པ་དང་འདྲ་པོར་ཡིད་ལ་དྲན། བུད་མེད་རྒན་མོ་དང་མཇལ་བའི་གནས་ཚུལ་དྲན་པ་དང༌། མོ་རང་སྒྲུང་གཏམ་སྤྱི་ཁྱབ་ཀྱི་ཚོགས་པའི་འོག་ཏུ་ལས་ཀ་བྱས་ཚུལ་སོགས་ཡིད་ལ་འཁོར། བདེ་ལེགས་ཀྱིས་བཤད་པ། “ངས་མི་ཚེ་ཧྲིལ་བོར་གནས་དོན་བདེན་པ་ལ་སྲུང་སྐྱོབ་བྱས་པ་དང་བརྩེ་བ་ལ་དགའ། དེས་མི་ཡང་མང་པོར་དགའ་སྤྲོ་ཐོབ་པར་རོགས་རམ་བྱས་ཡོད།”

མཐའ་མཇུག་ཏུ། བདེ་ལེགས་རྒན་མོས་བུད་མེད་རྒན་མོ་དེ་ཡང་བསྐྱར་མཇལ། བུད་མེད་རྒན་མོས་མོར་བཤད་པ། “བདེ་ལེགས་ལགས། ཁྱེད་རང་གིས་ལས་དོན་ཧ་ཅང་བཟང་པོ་བསྒྲུབས་སོང་། ཁྱེད་ཀྱིས་མི་རྣམས་ལ་ཡ་རབས་བཟང་སྤྱོད་སློབ་ཡོད། ཁྱེད་ཀྱིས་མི་རྣམས་ལ་བྱམས་སེམས་ཀྱི་བསམ་པ་སྐྱེས་ཡོད། ཁྱེད་ཀྱིས་ང་ཚོར་བརྩེ་བ་བསླབ་ཡོད་པས། ད་ནི་ང་ཚོའི་སྒྲུང་མཛད་ཀྱི་གྲས་སུ་འཛུལ་བར་དགའ་བསུ་ཞུ།”

བདེ་ལེགས་ཧ་ཅང་དགའ། མོས་བཤད་པ། “ང་དགའ་པོ་བྱུང།” དེ་ནས་ཁོ་མོས་ཡུན་རིང་ནས་འགྲུལ་བཞུད་གསར་པ་ཞིག་མགོ་བརྩམས། དེ་ནི་གཏམ་རྒྱུད་མཛད་མཁན་ཚོགས་པའི་ནང་དུ་ཞུགས་ནས་དུས་རྒྱུན་དུ་སྒྲུང་བཤད་དེ་ཡུན་རིང་དུ་གནས་རྒྱུ་དེ་རེད།

དེར་བརྟེན། བོད་ཀྱི་བྱིས་པ་ཚོ། ཁྱེད་རྣམས་ཀྱིས་སྒྲུང་ལ་དགའ་ཞེན་བྱེད། སྒྲུང་གཏམ་མང་པོ་བཀླགས། སྒྲུང་གཏམ་ཡག་པོ་བརྩམས། ནམ་ཞིག་ཁྱེད་རང་གིས་ཀྱང་བདེ་ལེགས་བཞིན་ནམ་མཁའ་ལ་འགྲུལ་བཞུད་བྱེད་ཐུབ།
