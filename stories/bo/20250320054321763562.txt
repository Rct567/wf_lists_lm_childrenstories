དུས་དེ་ནས་བཟུང་། རྡོ་རྗེ་དང་རྒན་པོ་གཉིས་ནམ་ཡང་མཉམ་དུ་འཚོ་གི་ཡོད་ལ། རྡོ་རྗེ་ནི་རྒན་པོའི་སློབ་མ་ཆགས་པ་དང་། རྒན་པོས་རྡོ་རྗེར་མི་ཚེ་སྐྱེལ་སྟངས་སློབ། རྡོ་རྗེས་ཀྱང་རྒན་པོར་རི་འགྲོའི་སྐོར་སློབ་དགོས་པ་བྱུང་། ཁོ་གཉིས་ཀྱི་མཛའ་གྲོགས་དེ་དག་འགྱུར་བ་མེད་པར་གནས་ཡོད་ལ། ཁོ་ཚོས་ནམ་ཡང་རང་གི་མཛའ་གྲོགས་ཀྱི་ཆེད་དུ་ཅི་ཞིག་བྱེད་རྒྱུ་དང་། གཞན་ལ་རོགས་རམ་ཇི་ལྟར་བྱེད་དགོས་མིན་སོགས་སློབ་སྦྱོང་བྱེད་ཀྱི་ཡོད།

ཉིན་ཞིག་ལ། རྡོ་རྗེ་རྒན་པོར་བརྗོད་དོན། “ང་སླར་ཡང་གངས་རི་བརྒལ་ནས་ཡུལ་གཞན་ལ་འགྲོ་འདོད་ཡོད། ང་ཁྱེད་ལ་བཞག་ནས་འགྲོ་མི་ཕོད།”

རྒན་པོས་འཛུམ་དམུལ་དམུལ་བྱས་ནས་བརྗོད་དོན། “རྡོ་རྗེ་ཁྱོད་ཀྱིས་འགྲོ་ན་ཆོག ངས་ཁྱོད་ལ་འགོག་ཐབས་མེད། ཡིན་ནའང་ང་ཁྱོད་ཀྱི་རྗེས་སུ་འབྲངས་ནས་སྡོད་ཆོག གལ་ཏེ་ཁྱོད་ལ་དཀའ་ངལ་འཕྲད་ཚེ་ང་ཡོད་དོ།”

དེ་ནས་རྡོ་རྗེ་གངས་རི་བརྒལ་ནས་འགྲོ་སྐབས། རྒན་པོ་ཡང་ཁོའི་རྗེས་སུ་འབྲངས་ནས་སོང་། དེ་ལྟར་ལམ་བར་དུ་ཁོ་ཚོས་མཛའ་བརྩེའི་དགའ་སྤྲོ་མང་པོ་བྱུང་བ་དང་། ཕན་ཚུན་ལ་རོགས་རམ་ཡང་བྱས།

ལོ་མང་པོའི་རྗེས་སུ། རྡོ་རྗེ་རང་ཉིད་རྒན་པོའི་དགོངས་པ་རྫོགས་རྗེས་ཡུལ་ལུང་དུ་ལོག་ནས་བསྡད། ཁོས་ནམ་ཡང་རང་གི་མཛའ་གྲོགས་ཀྱི་སྐོར་དྲན་པ་དང་། གཞན་ལ་བརྩེ་བ་དང་བྱམས་སེམས་བྱེད་དགོས་པ་ཤེས། དེ་བཞིན་དུ་རང་གི་མི་ཚེ་ནི་གལ་ཆེན་པོ་ཡིན་པས་སྙིང་རུས་བྱས་ནས་འཚོ་དགོས་པར་ཡང་རྟོགས།

དེ་བས་རྡོ་རྗེ་ནམ་ཡང་སྤྱང་པོ་དང་བློ་རིག་བཀྲ་པོ་ཡིན་ལ། ཁོས་གངས་རི་བརྒལ་ནས་ཡུལ་ལུང་མང་པོ་བསྐོར་བ་དང་། མི་རིགས་སྣ་ཚོགས་ཀྱི་ལོ་རྒྱུས་དང་རིག་གཞུང་ཡང་ཤེས་བྱུང་། ཁོས་མི་ཚེ་གང་པོར་མཛའ་བརྩེ་སྲུང་སྐྱོབ་དང་། སྙིང་རུས་བྱེད་དགོས་པ་ནམ་ཡང་བརྗེད་མ་སོང་།

དེ་ནི་གངས་རི་བརྒལ་བའི་རི་འགྲོའི་མཛའ་གྲོགས་ཀྱི་이야기རེད། དེའི་མཇུག་ཏུ་བདེན་པའི་བརྩེ་བ་ནི་སུ་དང་སུས་ཀྱང་བརྗེད་མི་སྲིད། བྱམས་སེམས་དང་མཛའ་བརྩེ་ནི་ནམ་ཡང་གནས་དགོས།
