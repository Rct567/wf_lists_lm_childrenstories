དེ་བས། དགེ་རྒན་གྱིས་སླར་ཡང་བཀའ་གནང་དོན། "འོ་ལེ། ད་ཐེངས་ལམ་བཟོ་རོགས་བྱེད་དགོས།"

དེ་བས། ཕྲུག་གུ་དེ་ཚོས་རྡོ་མཁྲེགས་དང་། རྡོ་རྩི་སོགས་བསྡུ་བའི་ལས་ཀ་བསྒྲུབས། དེ་དག་ལམ་བཟོ་བའི་རྒྱུ་ཆ་ཡིན། ལས་ཀའི་ཁྲོད་དུ་བཀྲ་ཤིས་ནི་ཧ་ཅང་ནུས་པ་ལྡན་པར་གྱུར་ཡོད། ཁོས་ཕྲུག་གུ་གཞན་དག་ལ་མདུན་དུ་བསྐྱོད་དེ་འཁྲིད་སྟོན་བྱས་པ་རེད། བཀྲ་ཤིས་ཀྱིས་ཁོ་ཚོར་གོམ་སྟབས་བསྟན་པའི་དུས་སུ། མགོ་སྐོར་མི་ཐེབས་པ་དང་ལས་ཀ་ལས་སླ་པོར་གྲུབ་ཡོད། དགའ་སྤྲོའི་ངང་ནས་ལས་ཀ་བྱེད་ཀྱི་ཡོད་དུས་སྐབས། ཁོ་ཚོའི་ངག་ནས་དགོད་སྒྲ་དང་བྲེལ་བྲེལ་གྱི་སྐད་ཆ་ཐོས་ཡོང་།

ལམ་བཟོ་བའི་དུས་ཚོད་ཡོངས་རྫོགས་སྐྱིད་པོ་ཡིན་པར་མ་ཟད། ཕྲུག་གུ་རེ་རེས་རང་གི་ནུས་པ་གང་ཡིན་མིན་ལ་རྟོགས་པ་དང་མཉམ་དུ། དུས་ཚོད་ཀྱི་འགྱུར་ལྡོག་གིས་ཁོ་ཚོའི་འཚོ་བར་ཤུགས་རྐྱེན་ཐེབས་ཀྱི་ཡོད།

ཉིན་གུང་ལ་ལམ་བཟོ་མཚམས་བཞག་སྟེ་ལྟོ་ཆས་བཟས། བདེ་ལེགས་ཀྱིས་ལམ་བཟོ་བའི་སྐོར་བཤད་དོན། "ང་ཚོས་ཕྲུག་གུ་གཞན་དག་ལ་རོགས་བྱས་ནས་ལམ་ཡག་པོ་ཞིག་བཟོ་ཐུབ་ཀྱི་རེད།"

བསོད་ནམས་ཀྱིས་དེར་མོས་མཐུན་བྱེད་བཞིན་བཤད་དོན། "དེ་བདེན་པ་རེད། ང་ཚོས་ལམ་བཟོས་པ་དེས་ཡུལ་མི་རྣམས་ལ་དཀའ་ངལ་མེད་པར་འགྲོ་བསྐྱོད་བྱེད་ཐུབ་པ་དང་། ཉེན་ཁ་ཆུང་དུ་འགྲོ་ཡི་རེད།"

ཁོ་ཚོའི་ཟ་མ་ཟིན་རྗེས་ལམ་བཟོ་བསྐྱར་དུ་བྱས་ཤིང་། བདེ་སྐྱིད་ཀྱིས་རྡོ་ལེན་པར་རོགས་རམ་བྱས་པ་རེད། བཀྲ་ཤིས་ཀྱིས་རྡོ་རྩི་དགོས་མཁོ་ལྟར་སྤྱད་པ་རེད། ཁོ་ཚོས་ཧུར་ཐག་གིས་ལས་ཀ་བྱེད་བཞིན་ཡོད།

ཉིན་དེའི་དགོང་མོར་ལམབཟོའི་ལས་ཀ་ཡོངས་སུ་གྲུབ་སོང། ལམ་ཆེན་པོ་ཞིག་ཆགས་ཡོད། ཡུལ་མི་ཚང་མ་དགའ་སྤྲོའི་ངང་ནས་ལམ་གསར་པར་བསྐྱོད་ཀྱི་ཡོད། ཕྲུག་གུ་དག་ལ་ཐུགས་རྗེ་ཆེ་ཞུས་ཤིང་། ཁོ་ཚོས་ཀྱང་དེ་ལས་སྤྲོ་སྣང་ཆེན་པོ་བསྐྱེད་པ་རེད།

ལོ་གཅིག་འགོར་རྗེས། དབྱར་ཁ་སླེབས་སོང། དགའ་བདེ་ཅན་གྲོང་དུ་མཚར་སྡུག་ལྡན་པའི་རྩྭ་ཐང་དུ་དཔག་ཏུ་མེད་པའི་མེ་ཏོག་དང་། དེ་བཞིན་བསིལ་བསིལ་གྱི་ཆུ་བོ་འབབ་ཀྱི་ཡོད་པས། དགའ་སྐྱིད་ཀྱིས་ཁེངས་པ་དང་། གྲོང་དེར་གནས་པའི་དུད་འགྲོ་རྣམས་ལྟོ་རྒྱགས་སུ་འཚོ་བར་བྱེད།

ཕྲུག་གུ་ཚོས་གནམ་གྱི་སྐར་ཚོགས་ལ་བལྟས་ནས་གྲོང་དུ་ཡོང་སོང་། དགེ་རྒན་དང་ལྷན་དུ་ཡུལ་སྐོར་ལ་བསྐྱོད་དུས། བཀྲ་ཤིས་ཀྱིས་བཤད་དོན། "ངས་ལམ་བཟོ་བར་སྐུལ་འདེབས་བྱས་ཡོད། དེར་བརྟེན་ལམ་བཟང་པོ་ཆགས་ཡོད་"

བསོད་ནམས་ཀྱིས་མུ་མཐུད་དེ་བཤད་དོན། "ངས་ཆུ་གསོག་རྒྱུར་རོགས་བྱས་ཡོད། དེར་བརྟེན་རྩྭ་ཐང་ཧ་ཅང་ཡག་པོ་ཡོད།"

ཁོ་ཚོས་ཚང་མས་གཅིག་གིས་གཅིག་ལ་བརྩེ་བ་བྱེད་པ་དང་། ཁོ་ཚོའི་ལས་ཀའི་གྲུབ་འབྲས་ལ་ཡིད་ཆེས་བྱེད་ཀྱི་ཡོད། གྲོང་དེའི་མི་རྣམས་ཧ་ཅང་བདེ་སྐྱིད་ཡིན་ལ། གྲོང་དེར་འཚོ་བ་སྐྱེལ་བ་ནི་ཧ་ཅང་བདེ་སྐྱིད་ལྡན་པར་བརྗོད།

ཉིན་རེ་བཞིན་འཚོ་བ་འདི་ལྟར་བསྐྱེལ་བཞིན་པས། ཕྲུག་གུ་ཚོས་མུ་མཐུད་དུ་གྲོང་དེར་རོགས་རམ་བྱེད་ཀྱི་ཡོད། དགའ་སྤྲོའི་ངང་ནས་འཚོ་བ་བསྐྱེལ་བ་དང་། ལས་ཀ་བྱེད་པའི་དུས་སྐབས་སུ་འཛོམས་པར་གྱུར་ཏེ། ཕན་ཚུན་ལ་བརྩེ་བ་ཆེན་པོས་ཁེངས་ནས་འཚོ་བ་རོལ་བ་རེད། དེ་ནི་ཧ་ཅང་ངོ་མཚར་ལྡན་པ་ཞིག་རེད། ། །
