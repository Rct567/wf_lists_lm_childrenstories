ལོ་འགའ་ཞིག་འགོར་རྗེས། བཀྲ་ཤིས་པ་དར་མ་དང་ལང་ཚོ་རྒྱས་པའི་ན་གཞོན་ཞིག་ཏུ་གྱུར་སོང་། ཁོས་ནམ་ཡང་སྒྲོལ་མ་དང་ཁོའི་མཛའ་བརྩེའི་སྐོར་བསམ་བློ་བཏང་། ཉིན་ཞིག་ཁོས་གསན་བསམ་རྒྱ་ཆེན་བྱེད་ཀྱིན་པའི་སྐབས། དགེ་རྒན་རྒན་པོ་ཞིག་གིས་བཀྲ་ཤིས་པ་ལ་“ཁྱོད་ཀྱིས་ནམ་ཡང་བསམ་བློ་ཡག་པོ་ཞིག་རྒྱག་དགོས། མི་རྣམས་ཀྱི་སེམས་ནང་དུ་བརྩེ་བ་དྲན་གསོ་བྱེད་དགོས།”ཞེས་བཤད། དེ་ནས་བཀྲ་ཤིས་པས་རང་གི་སེམས་ལ་བརྩེ་བ་ཡོད་པའི་བུད་མེད་དེ་ག་རེ་ཡིན་མིན་བསམ་བློ་བཏང་། མཐར་སྒྲོལ་མ་དྲན་བྱུང་།

བཀྲ་ཤིས་པས་སྒྲོལ་མ་འཚོལ་བའི་ཐག་གཅོད་བྱས། ཁོས་རང་གི་ཕ་དང་མ་ལ་བཤད་དེ་ཕ་ཡུལ་ནས་ཕྱིར་ཐོན་ཏེ་སྒྲོལ་མ་འཚོལ་དུ་ཕྱིན། ལམ་བར་དུ་ཁོས་དཀའ་ལས་མང་པོ་བརྒྱབ། གནས་ཚུལ་མང་པོ་ཐོས། མཐར་ལོ་མང་པོའི་རྗེས་སུ། ཁོས་སྒྲོལ་མའི་ཁྱིམ་གྱི་གནས་ཚུལ་ཤེས། ཁོ་མོ་ནི་གྲོང་གསེབ་ཀྱི་མི་ཚང་ཞིག་ལ་བག་མ་བརྒྱབ་ནས་བསྡད་ཡོད་པ་རེད། བཀྲ་ཤིས་པ་གྲོང་གསེབ་དེར་བསྐྱོད་དེ་སྒྲོལ་མའི་ཁྱིམ་དུ་བསྐྱོད། སྒྲོལ་མ་ཡང་ཤིན་ཏུ་དགའ་སྤྲོ་ཆེན་པོས་བཀྲ་ཤིས་པ་ལ་ཐུག སྒྲོལ་མ་ནི་བཀྲ་ཤིས་པའི་མདུན་དུ་ཡོང་དུས། ཁོ་གཉིས་ཀྱིས་ཕན་ཚུན་མིག་ཆུ་འདོན་བཞིན་ཡོད་། བཀྲ་ཤིས་པས་སྒྲོལ་མ་ལ་“ཁྱོད་ནི་ང་ཡི་མཛའ་གྲོགས་རྙིང་པ་ཡིན་པས་ང་ནམ་ཡང་ཁྱོད་ལ་བརྩེ་དུང་ཆེན་པོ་ལྡན། ངས་ཁྱོད་བསམ་བློ་བཏང་མྱོང་།”ཞེས་བཤད།

སྒྲོལ་མས་བཀྲ་ཤིས་པ་ལ་བཤད་རྒྱུར། “ང་ཡང་ནམ་ཡང་ཁྱོད་ལ་བརྩེ་བ་ཆེན་པོ་ཡོད། ང་རང་ནི་ཁྱོད་ཀྱི་རྡོ་རིལ་ཆུང་ཆུང་དེ་ནམ་ཡང་བསྲུངས་ཡོད། དེ་ནི་ང་ཚོའི་མཛའ་བརྩེགས་ཀྱི་དྲན་རྟེན་ཡིན། ངས་ནམ་ཡང་ཁྱོད་བསམ་བློ་བཏང་།” སྒྲོལ་མས་རྡོ་རིལ་ཆུང་ཆུང་དེ་ཕྱིར་བཏོན་ནས་བཀྲ་ཤིས་པ་ལ་སྤྲད། ཉིན་དེ་ནས་བཟུང་། ཁོང་གཉིས་ཀྱིས་བསྐྱར་དུ་མཉམ་དུ་སྐད་ཆ་བཤད་པ་དང་། བརྩེ་དུང་སྤེལ་ཞིང་། མཛའ་མཐུན་མུ་མཐུད་ཡིན། ཁོང་གཉིས་ཀྱིས་གནས་ཚུལ་དེ་འདྲ་མང་པོ་མཐོང་ཡོད་ཀྱང་། ཁོང་གཉིས་ཀྱི་མཛའ་བརྩེ་དེ་ནི་ནམ་ཡང་འགྱུར་བ་མི་འགྲོ་བ་རེད། གཏམ་རྒྱུད་ཀྱི་མཇུག་ཏུ། ཁོང་གཉིས་ནི་ནམ་ཡང་མཉམ་དུ་བསྡད། དགོད་སྒྲས་ཁེངས་པའི་འཚོ་བ་རོལ་བཞིན་ཡོད། བཀྲ་ཤིས་པ་དང་སྒྲོལ་མ་ནི་མཛའ་བརྩེགས་ཀྱི་མིག་དཔེ་ཡིན།
