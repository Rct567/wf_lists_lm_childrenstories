ཕ་མ་གཉིས་ཀྱིས་ཡིད་རྟོན་བྱས་ཏེ་བཀྲ་ཤིས་དང་མཉམ་དུ་ཞིང་ཁར་བལྟ་བར་སོང་། དེར་ཁོང་ཚོས་ངོ་མ་ཞིང་ཁར་ཆུ་ཡོང་གི་ཡོད་པ་མཐོང་། ཁོང་ཚོས་ཆུ་ཡོང་བཞིན་པའི་ཉེན་ཁ་ཡོད་པར་བརྟེན། འཕྲལ་དུ་ཞིང་ཁའི་ནང་གི་ལས་ཀ་མཚམས་བཞག་ནས། ཁྱིམ་ལ་ལོག་སོང་།

དེ་ནི་འཚོ་བའི་ཉིན་ཞིག་ཡིན་ལ། ཡིན་ནའང་ཁོང་ཚོས་བྱམས་བརྩེ་དང་བརྩི་བཀུར་གྱི་རིན་ཐང་བརྒྱུད་ནས་ལས་དོན་བསྒྲུབས་པ་ཡིན། ཉིན་དེར་ཞིང་ཁར་ཆུ་ལྷུང་། ཡིན་ནའང་བཀྲ་ཤིས་ཀྱིས་མི་ཚང་ལ་ཉེན་བརྡ་བཏང་བའི་རྐྱེན་གྱིས་། མི་ཚང་སྲོག་ཉེན་ལས་ཐར་སོང་།

ཉིན་དེའི་ཕྱི་ཉིན་ལ། རྒན་པོ་བསོད་ནམས་བཀྲ་ཤིས་ཀྱི་ཁྱིམ་དུ་ཡོང་ནས་། བཀྲ་ཤིས་ལ་སེམས་གསོ་བྱས། "བཀྲ་ཤིས། ཁྱེད་ཀྱིས་ང་ལ་བརྩེ་བའི་རིན་ཐང་སྟོན་སོང་། ཁྱེད་ཀྱིས་བཟང་སྤྱོད་བྱེད་ནུས་པར་བརྟེན་ངའི་སེམས་ལ་དགའ་སྐྱིད་འཆིང་བྱུང།"

བཀྲ་ཤིས་ཧ་ཅང་སྤྲོ་སེམས་སྐྱེས་ནས་རྒན་པོར་འུར་ཆེན་བྱས་ཏེ་བཤད། "རྒན་ལགས། ངའི་སེམས་ནས་ཁྱེད་རང་བཀུར་བཀུར་བྱེད་ཀྱི་ཡོད། ངས་རྟག་པར་ཁྱེད་ཀྱི་བཀའ་སློབ་ལ་ཉན་གྱི་ཡིན།"

དེ་ནས་བཀྲ་ཤིས་དང་རྒན་པོ་གཉིས་ཀྱིས་ཞིང་ཁར་ཆུ་ཡོང་བཞིན་པའི་སྐོར་བཤད་པ་རེད། བཀྲ་ཤིས་ཀྱིས་བརྩེ་བ་བཏང་བ་ལ་བརྟེན། ཞིང་པ་ཚང་མ་མི་འགྱོད་པར་སེམས་དང་མཐུན་པའི་ཐབས་ཤེས་ལ་བརྟེན། ཆུའི་རྐྱེན་ངན་སེལ་ཐུབ་བྱུང།

བཀྲ་ཤིས་ཀྱི་ཨ་མས་བཤད་རྒྱུར། "བཀྲ་ཤིས། ཁྱེད་ནི་ངའི་བུ་ཆུང་ཡིན་ལ། ཁྱེད་ཀྱིས་ངའི་སེམས་སུ་སྐྱིད་སྣང་བསྐྲུན། བརྩེ་བ་ནི་མི་ཚེའི་རིན་ཐང་གལ་ཆེན་ཡིན་པ་ཁྱེད་ཀྱིས་ཤེས་སོང།"

བཀྲ་ཤིས་ཀྱིས་མགོ་བོ་བསྒུར་བཞིན། "ཨ་མ། ང་ནི་ཁྱེད་ལ་བརྩེ་བ་བྱེད་པར་དགའ། དེ་ནི་ངའི་ལས་འགན་ཡིན།" ཞེས་ལན་བཏབ།

དེ་ནི་བྱིས་པའི་ལོ་རྒྱུས་ཡིན་ལ། དེའི་ནང་དུ་བྱམས་བརྩེ་དང་བརྩི་བཀུར་ཡོད་པ་མ་ཟད། དཀའ་ངལ་སེལ་བའི་སྙིང་སྟོབས་ཀྱང་ཡོད། བཀྲ་ཤིས་ནི་ལོ་རེ་བཞིན་སྐྱེས་ཚེ། ཁོང་གི་མི་ཚེའི་ནང་གི་འཚོ་བར་སྙིང་སྟོབས་དང་སྤྲོ་སེམས་འཛིན་པར་ཤེས་བྱུང། དེ་ནི་འཚོ་བའི་གོ་རྟོགས་ཡིན་ལ། ཁྱེད་རང་གིས་གཞན་ལ་བྱམས་བརྩེ་དང་བརྩི་བཀུར་སྤྲོད་ན། ཁྱེད་ཀྱིས་ཕྱིར་ཡང་དེ་ལྟ་བུའི་བྱམས་བརྩེ་དང་བརྩི་བཀུར་ལེན་ཐུབ།

མཇུག་ཏུ། བཀྲ་ཤིས་ནི་བུ་ཆུང་སྤྱང་པོ་ཞིག་ཏུ་གྱུར་ལ། ཁོང་གིས་གཞན་ལ་རོགས་རམ་བྱེད་ཀྱི་ཡོད། ཁོང་གིས་ཕ་མ་ལ་བརྩི་བཀུར་བྱེད་ཀྱི་ཡོད། ཁོང་ནི་འཇོན་ཐང་ཅན་གྱི་མི་ཞིག་ཏུ་གྱུར་སོང་། མི་ཚེ་ཧྲིལ་པོའི་ནང་། བྱམས་བརྩེ་དང་བརྩི་བཀུར་གྱི་སྙིང་སྟོབས་ཀྱིས་མི་ཚང་དང་ཁོའི་མཐའ་འཁོར་གྱི་མི་རྣམས་ལ་དཀའ་སྡུག་སེལ་བར་རོགས་རམ་བྱས།
