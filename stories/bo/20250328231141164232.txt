ཉི་མ་ནུབ་ལ་ཁད་དེ། ཁོང་གསུམ་མདའ་རྩེད་རྩེ་སའི་ས་ཆ་རུ་སླེབས་སོང་། པཎ་ཆེན་རིན་པོ་ཆེས་བསོད་ནམས་དང་བཀྲ་ཤིས་གཉིས་ལ་མདའ་རྩེད་སྦྱོང་དུ་བཅུག ། བསོད་ནམས་ལ་ནུས་པ་ཆེན་པོ་ཡོད་པ་མ་ཟད། མདའ་མདེའུ་རིང་པོའི་དམིགས་འབེན་ཡང་ཡག་པོ་ཐུབ་ཀྱི་ཡོད་རེད། བཀྲ་ཤིས་ནི་ཐེངས་དང་པོར་རྩེ་བཞིན་ཡོད་མོད། ཡིན་ན་ཡང་། ཁོང་གིས་གཟབ་ནན་དང་བརྩོན་འགྲུས་ཆེན་པོས་རྩེ་བ་རེད།

བསོད་ནམས། "ང་ཧ་ཅང་སྤྲོ་པོ་བྱུང་། པཎ་ཆེན་རིན་པོ་ཆེ། ཁྱེད་ཀྱིས་ང་ལ་རྩེད་མོ་མང་པོ་ཞིག་སློབ་གནང་བྱུང་།"

བཀྲ་ཤིས། "ཡང་དག་རེད། ང་ཡང་རོལ་རྩེད་སྣ་ཚོགས་ཤེས་པར་དགའ་པོ་བྱུང་། ད་དུང་རོལ་རྩེད་གཞན་ཡོད་དམ།"

པཎ་ཆེན་རིན་པོ་ཆེ། "ང་ལ་རོལ་རྩེད་མང་པོ་བསྟན་ཆོག་མོད། ཉིན་འདི་ག་རང་ལ་ལུས་ངལ་གསོ་དགོས། སང་ཉིན་ཕྱིར་ཡོང་། དེ་དུས་ངས་ཁྱེད་ཚོར་བོད་ཀྱི་གཞས་དང་གར་སྐོར་སློབ་ཆོག"

བསོད་ནམས། "བཟང་ངེས། ངས་ནམ་རྒྱུན་བོད་ཀྱི་གཞས་དང་གར་ལ་དགའ་བོ་ཡོད།"

བཀྲ་ཤིས། "ང་ཡང་རོལ་རྩེད་དེ་ལ་སློབ་སྤྲོ་ཡོད།"

པཎ་ཆེན་རིན་པོ་ཆེ། "ཡག་པོ། ད་ང་ཚོ་མཉམ་དུ་འདི་རུ་འགྲོ་སོང་།" ཁོང་གསུམ་ཕོ་བྲང་གི་ནང་དུ་ཕྱིན།

ཉིན་དེའི་མཚན་མོར། ཁོང་གཉིས་ནང་དུ་ལོག་སྐབས། བསོད་ནམས་ཀྱིས་བཀྲ་ཤིས་ལ་བཤད།

བསོད་ནམས། "བཀྲ་ཤིས་ལགས། ང་དེ་རིང་ཧ་ཅང་དགའ་པོ་བྱུང་།"

བཀྲ་ཤིས། "ངས་ཀྱང་དེ་ལྟར་ཡིན། ང་ལ་མགྲོན་པོ་དེ་འདྲ་ཡོང་རེད།"

བསོད་ནམས། "རྗེས་ཕྱོགས་སུ། པཎ་ཆེན་རིན་པོ་ཆེ་ལ་བཀའ་དྲིན་ཆེ་ཞུ་དགོས་ཀྱི་རེད།"

བཀྲ་ཤིས། "ལགས་ཡིན། སང་ཉིན་ལ་ང་ཚོས་ཁོང་ལ་གུས་ཕྱག་འབུལ་ཆོག"

ཉི་མ་ཕྱེད་ལ་སླེབས། བསོད་ནམས་དང་བཀྲ་ཤིས་གཉིས་ཕོ་བྲང་གི་སྒོ་ཁར་སླེབས་བྱུང་། པཎ་ཆེན་རིན་པོ་ཆེ་ཡང་སྒོ་ཁར་སྒུག་ཡོད་པ་རེད།

པཎ་ཆེན་རིན་པོ་ཆེ། "བུ་ཚོ། ཡོང་། དེ་རིང་ང་ཚོ་བོད་ཀྱི་གཞས་དང་གར་སྦྱོང་གི་ཡིན།"

བཀྲ་ཤིས། "ང་ལ་སྤྲོ་བ་ཡོད། ང་གར་སྦྱོང་སྤྲོ་ཡོད།"

ཁོང་ཚོས་གར་གྱི་དུས་ཚོད་དེ་དགའ་སྤྲོའི་ངང་བསྐྱལ་བ་རེད། པཎ་ཆེན་རིན་པོ་ཆེས་ཁོང་གཉིས་ལ་བོད་ཀྱི་གཞས་དང་གར་གྱི་རྣམ་གྲངས་མང་པོ་ཞིག་བསླབས་པས། ཁོང་གཉིས་དེ་ལས་ཀྱང་སྤྲོ་པོ་བྱུང་། ཉིན་མཐའ་མར་ཁོང་གཉིས་ཀྱིས་པཎ་ཆེན་རིན་པོ་ཆེ་ལ་གུས་ཕྱག་ཕུལ་ཏེ། ཕྱིར་རང་གི་ཕ་ཡུལ་དུ་ལོག་པ་རེད།
