དེར་མ་ཟད། ཁོས་གྲོགས་པོ་རྣམས་ལ་“ང་ཚོས་ཁོང་ཚོར་ཁང་པ་བཟོ་རོགས་བྱེད་དགོས་རེད། ཁོང་ཚོར་དགོས་པའི་ཟ་ཆ་དག་གྲ་སྒྲིག་བྱེད་དགོས།”ཞེས་བཤད། རྡོ་རྗེ་ཡི་བྱེད་སྟངས་དེས་སློབ་མ་གཞན་དག་ཀྱང་ཧ་ཅང་སེམས་ཚོར་ཆེན་པོ་སྐྱེས། དེ་ནི་མི་ཚང་མ་འདི་ལྟར་འགྱུར་དགོས་ཟེར་བའི་བསླབ་བྱ་ལྟ་བུར་གྱུར་སོང་།

སློབ་གྲྭའི་ནང་གི་སློབ་མ་ཚང་མས་རང་ནུས་ལྟར་དཀའ་ངལ་ཅན་ཚོར་རོགས་བྱེད་ཀྱི་ཡོད། ཁ་ཤས་ཀྱིས་ཁང་པ་བཟོ་རོགས་བྱས་པ་དང་། ཁ་ཤས་ཀྱིས་འཚོ་བའི་དགོས་མཁོ་གྲ་སྒྲིག་བྱས། རྡོ་རྗེ་ཡིས་འགོ་ཁྲིད་དེ་སློབ་གྲྭའི་དགེ་རྒན་དང་གྲོགས་པོ་རྣམས་མཉམ་དུ་མི་དམངས་དཀའ་ངལ་ཅན་ལ་རོགས་རམ་བྱེད་པའི་ལས་དོན་དེ་ལེགས་པར་གྲུབ་པ་རེད། ཁོས་ཐོག་མར་གྲོགས་པོ་ཚོ་ལ་ལས་ཀ་གང་དག་བགོ་བཤའ་རྒྱག་དགོས་མིན་སོགས་བསླབ།

མི་དམངས་དཀའ་ངལ་ཅན་ཡང་སླེབས་བྱུང་། རྡོ་རྗེ་དང་ཁོའི་གྲོགས་པོ་ཚོས་དགའ་སྤྲོའི་ངང་ཁོང་ཚོར་ཕེབས་བསུ་ཞུས། རྡོ་རྗེས་ཁོང་ཚོའི་ཆེད་དུ་དྲོ་སྐྱིད་ལྡན་པའི་ཁང་པ་དང་ཟ་ཆས་སྦྱར་ནས་ཁོང་ཚོར་རང་ཉིད་ཀྱི་ཕ་ཡུལ་ལྟ་བུའི་སྣང་བ་བསྐྲུན་པར་བྱས། མི་དམངས་དཀའ་ངལ་ཅན་ཚང་མས་རྡོ་རྗེ་དང་ཁོའི་གྲོགས་པོ་རྣམས་ལ་བཀའ་དྲིན་ཞུས་པ་དང་། ཁོང་ཚོས་བླ་མེད་ཀྱི་སྙིང་སྟོབས་ལ་ཡི་རངས་བྱས། དཀའ་ངལ་ཅན་མང་པོ་ཞིག་གིས་སེམས་གསོ་དང་བདེ་སྐྱིད་ཚོར་བྱུང་། སློབ་གྲྭའི་ནང་དུ་ཡང་བསྐྱར་དགོད་སྒྲ་དེ་ལས་ཀྱང་ཡང་དུ་གྲགས་པ་རེད།

མི་ཚང་མས་འཆམ་མཐུན་དང་མཐུན་སྒྲིལ་ཡོང་བར་བྱེད་དགོས་པའི་གལ་ཆེན་པོ་དེའང་ཁོང་ཚོས་མཐོང་། དགེ་རྒན་གྱིས་ཀྱང་“འདི་ནི་རང་ཉིད་ལ་སློབ་སྦྱོང་ཡག་པོ་བྱེད་པ་དང་། གཞན་ལ་ཕན་ཐོགས་བྱེད་པའི་སློབ་གྲྭ་ཡག་པོ་ཞིག་རེད།”ཅེས་གསུངས་པ་རེད།

ལོ་གཅིག་འགོར་རྗེས། དབྱར་ཁ་སླེབས་ཤིང་སློབ་གྲྭ་གྲོལ། རྡོ་རྗེས་ཁོ་རང་གི་མི་ཚེའི་ནང་གི་བྱ་བ་སྣ་ཚོགས་དྲན་གསོ་བྱས་པ་དང་སེམས་ཀྱི་གཏིང་ནས་བདེ་སྐྱིད་ཡོད། ཁོས་ཐོག་མར་ནོར་འཁྲུལ་བཟོས་སྐབས་གྲོགས་པོ་དང་འབྲེལ་བ་བྱུང་བ་དང་། རང་གི་ནོར་འཁྲུལ་ལ་ངོས་ལེན་བྱེད་ཤེས་པར་བསླབས་པ་མ་ཟད། རང་ཉིད་ཀྱི་འཚོ་བ་ལ་འགྱུར་བ་ཐེབས་སོང་།

དེའི་རྗེས་སུ་མི་ལ་བརྩེ་སེམས་དང་རོགས་རམ་བྱེད་ཤེས་པར་སྦྱངས་པ་རེད། སློབ་གྲྭ་དེ་ནི་དགོད་སྒྲས་ཁེངས་པ་དང་། སེམས་གསོ་ཐོབ་པའི་སློབ་གྲྭ་ཞིག་ཏུ་གྱུར་པ་མ་ཟད། ལ་དྭགས་ཡུལ་གྱི་ཕྲུ་གུ་ཚང་མས་དེ་འདྲའི་སློབ་གྲྭར་སློབ་སྦྱོང་བྱེད་རྒྱུའི་རེ་བ་བཅངས་སོ།

དེར་མ་ཟད། རྡོ་རྗེ་ཡིས་གྲོགས་པོ་ཚོ་དང་མཉམ་དུ་འབྱུང་འགྱུར་གྱི་འཆར་གཞི་བཟོས་ནས་སློབ་གྲྭར་ཕན་པའི་བྱ་བ་མང་པོ་སྒྲུབ་རྒྱུའི་ཐག་བཅད། ཁོས་ཤེས་པ་ནི། དཀའ་སྡུག་གང་འདྲ་ཡིན་རུང་། གལ་ཏེ་མཐུན་སྒྲིལ་ཡོད་ཚེ། མི་ཚང་མས་མཉམ་དུ་ལས་དོན་མང་པོ་སྒྲུབ་ཐུབ་པ་དང་སེམས་གསོའང་རྙེད་ཐུབ་པ་ཡིན།
